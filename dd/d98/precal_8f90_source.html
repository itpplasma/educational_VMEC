6SUBROUTINE precal
7 USE vacmod
8
9 use dbgout
10
11 IMPLICIT NONE
12
13 REAL(rprec), PARAMETER :: p25 = p5*p5
14 REAL(rprec), PARAMETER :: bigno = 1.e50_dp
15 REAL(rprec), PARAMETER :: epstan = epsilon(one)
16
17 INTEGER :: kp, ku, kuminus, kv, kvminus, i, m, n, mn, &
18 imn, jmn, kmn, l, istat1, smn
19 REAL(rprec), DIMENSION(0:mf + nf,0:mf,0:nf) :: cmn
20 REAL(rprec) :: argu, argv, argp, dn1, f1, f2, f3, alp_per
21
22 ! THIS ROUTINE COMPUTES INITIAL CONSTANTS AND ARRAYS
23
24 pi2 = 8*atan(one) ! 2 pi
25 pi3 = p5*pi2**3 ! 4 pi^3 = pi * (2 pi)^2
26 pi4 = 2*pi2 ! 4 pi
27 onp = one/nfper ! 1/nfp
28 onp2 = onp*onp ! 1/nfp^2
29 alu = pi2/nu ! (2 pi)/nu
30 alv = pi2/nv ! (2 pi)/nv
31 alp = pi2*onp ! (2 pi)/nfp
32 alvp = onp*alv ! (2 pi)/(nv * nfp)
33
34 alp_per = pi2/nvper ! (2 pi)/nvper
35
36 DO kp = 1, nvper
37 cosper(kp) = cos(alp_per*(kp - 1))
38 sinper(kp) = sin(alp_per*(kp - 1))
39 END DO
40
41 ! IMIRR(I) GIVES THE INDEX OF THE POINT (TWOPI-THETA(I), TWOPI-ZETA(I))
42 ! k(u,v)minus count from the other side of the grid
43 ! imirr is then the linear index correponding to (kuminus, kvminus)
44 DO ku = 1, nu
45 kuminus = mod(nu - (ku-1), nu) + 1
46 DO kv = 1, nv
47 kvminus = mod(nv - (kv-1), nv) + 1
48
49 i = kv + nv*(ku - 1)
50 imirr(i) = kvminus + nv*(kuminus - 1)
51
52 cosuv(i) = cos(alvp*(kv - 1))
53 sinuv(i) = sin(alvp*(kv - 1))
54 END DO
55 END DO
56
57 ! NOTE: ANGLE DIFFERENCE IS PI2*{[NUV + (KUP-1)] - (KU-1)}
58 ! THIS DIFFERENCE IS ACCOUNTED FOR BY THE OFFSET IUOFF IN GREENF ROUTINE
59 !
60 ! THE KP SUM BELOW IS USED ONLY FOR NV == 1. IT PERFORMS THE V-INTEGRAL
61 ! IN AN AXISYMMETRIC PLASMA
62
63 if (nv .eq. 1) then
64 ! Tokamak; nvper = 64
65 DO kp = 1, nvper
66 argp = p5*alp_per*(kp-1)
67 IF (abs(argp - p25*pi2) < epstan) THEN
68 tanv_1d(kp) = bigno
69 ELSE
70 tanv_1d(kp) = 2*tan(argp)
71 ENDIF
72 end do
73 else
74 ! Stellarator
75 DO kv = 1, nv
76 argv = p5*alv*(kv - 1)
77 IF (abs(argv - p25*pi2) < epstan) THEN
78 tanv_1d(kv) = bigno
79 ELSE
80 tanv_1d(kv) = 2*tan(argv)
81 ENDIF
82 end do
83 end if
84
85 DO ku = 1, 2*nu
86 argu = p5*alu*(ku - 1)
87 IF (abs(argu - p25*pi2)<epstan .or. abs(argu - 0.75_dp*pi2) < epstan) THEN
88 tanu_1d(ku) = bigno
89 ELSE
90 tanu_1d(ku) = 2*tan(argu)
91 ENDIF
92 end do
93
94 i = 0
95 DO kp = 1, nvper
96 IF (kp.gt.1 .and. nv.ne.1) EXIT
97 argp = p5*alp_per*(kp-1)
98 DO ku = 1, 2*nu
99 argu = p5*alu*(ku - 1)
100 DO kv = 1, nv
101 i = i + 1
102 argv = p5*alv*(kv - 1) + argp
103
104 IF (abs(argu - p25*pi2)<epstan .or. abs(argu - 0.75_dp*pi2) < epstan) THEN
105 tanu(i) = bigno
106 ELSE
107 tanu(i) = 2*tan(argu)
108 ENDIF
109
110 IF (abs(argv - p25*pi2) < epstan) THEN
111 tanv(i) = bigno
112 ELSE
113 tanv(i) = 2*tan(argv)
114 ENDIF
115
116 ! print *, i, kp, argp, ku, argu, kv, argv, tanu(i), tanv(i)
117 END DO
118 END DO
119 END DO
120
121 ! poloidal Fourier transform basis functions
122 DO m = 0, mf
123 l40: DO ku = 1, nu
124 cosu(m,ku) = cos(alu*(m*(ku - 1)))
125 sinu(m,ku) = sin(alu*(m*(ku - 1)))
126 DO kv = 1, nv
127 i = kv + nv*(ku - 1)
128 IF (i > nuv2) then
129 ! This crops the linear index i to the stellarator-symmetric half of a toroidal module.
130 ! ku == poloidal is the slow direction, so there the index ku will only run up to nu2.
131 ! Above ku loop was probably not directly ended at nu2 to have the full set of Fourier basis functions in cosu, sinu.
132 cycle l40
133 end if
134 cosu1(i,m) = cosu(m,ku)
135 sinu1(i,m) = sinu(m,ku)
136 END DO
137 END DO l40
138 DO ku = 1, nu2
139 cosui(m,ku) = cosu(m,ku)*alu*alv*2
140 sinui(m,ku) = sinu(m,ku)*alu*alv*2
141 IF (ku.eq.1 .or. ku.eq.nu2) then
142 ! wint-like functionality to take stellarator-symmetric
143 ! "folding-over" into first half-period into account
144 cosui(m,ku) = p5*cosui(m,ku)
145 end if
146
147 ! alternative, maybe more reasonable way:
148 ! endpoints weighted by 1, inner points weighted by 2
149 ! --> no need to have a product 2*0.5 in here...
150 END DO
151 END DO
152
153 ! toroidal Fourier transform basis functions
154 DO n = -nf, nf
155 dn1 = alvp*(n*nfper)
156 csign(n) = sign(one,dn1)
157 l50: DO ku = 1, nu
158 DO kv = 1, nv
159 i = kv + nv*(ku - 1)
160 cosv(n,kv) = cos(dn1*(kv - 1))
161 sinv(n,kv) = sin(dn1*(kv - 1))
162 IF (i.gt.nuv2 .or. n.lt.0) then
163 ! This crops the linear index i to the stellarator-symmetric half of a toroidal module.
164 ! ku == poloidal is the slow direction, so there the index ku will only run up to nu2.
165 ! Above ku loop was probably not directly ended at nu2 to have the full set of Fourier basis functions in cosv, sinv.
166 cycle l50
167 end if
168 cosv1(i,n) = cosv(n,kv)
169 sinv1(i,n) = sinv(n,kv)
170 END DO
171 END DO l50
172 END DO
173
174 ! Fourier mode number arrays for potsin, potcos
175 mn = 0
176 DO n = -nf, nf
177 DO m = 0, mf
178 mn = mn + 1
179 xmpot(mn) = m
180 xnpot(mn) = n*nfper
181 END DO
182 END DO
183
184 ! COMPUTE CMNS AND THE COEFFICIENTS OF T+- IN EQ (A14 AND A13) IN J.COMP.PHYS PAPER (PKM)
185 ! NOTE: HERE, THE INDEX L IN THE LOOP BELOW IS THE SUBSCRIPT OF T+-. THEREFORE,
186 ! L = 2L' + Kmn (L' = INDEX IN EQ. A14, Kmn = |m-n|), WITH LMIN = K AND LMAX = Jmn == m+n.
187 !
188 ! THE FOLLOWING DEFINITIONS PERTAIN (NOTE: kmn <= L <= jmn):
189 !
190 ! F1 = [(L + jmn)/2]! / [(jmn - L)/2]! == [(jmn + kmn)/2 + L']!/[(jmn - kmn)/2 + L']!
191 !
192 ! F2 = [(L + kmn)/2]! == (L' + kmn)!
193 !
194 ! F3 = [(L - kmn)/2]! == (L')!
195
196 DO m = 0, mf
197 DO n = 0, nf
198 jmn = m + n
199 imn = m - n
200 kmn = abs(imn)
201
202 ! Integer: J+K always even
203 smn = (jmn + kmn)/2
204 f1 = 1
205 f2 = 1
206 f3 = 1
207 DO i = 1, kmn
208 f1 = f1*(smn + 1 - i)
209 f2 = f2*i
210 END DO
211 cmn(0:mf+nf,m,n) = 0
212 DO l = kmn, jmn, 2
213 cmn(l,m,n) = f1/(f2*f3)*((-1)**((l - imn)/2))
214 f1 = f1*p25*((jmn + l + 2)*(jmn - l))
215 f2 = f2*p5*(l + 2 + kmn)
216 f3 = f3*p5*(l + 2 - kmn)
217 END DO
218 END DO
219 END DO
220
221 ! Now combine these into a single coefficient (cmns), Eq. A13).
222 ! NOTE: The ALP=2*pi/nfper factor is needed to normalize integral over field periods
223
224 DO m = 1,mf
225 DO n = 1,nf
226 cmns(0:mf+nf,m,n) = p5*alp*( cmn(0:mf+nf,m,n ) + cmn(0:mf+nf,m-1,n ) &
227 + cmn(0:mf+nf,m,n-1) + cmn(0:mf+nf,m-1,n-1))
228 END DO
229 END DO
230 cmns(0:mf+nf,1:mf,0) = (p5*alp)*(cmn(0:mf+nf,1:mf,0) + cmn(0:mf+nf,:mf-1,0))
231 cmns(0:mf+nf,0,1:nf) = (p5*alp)*(cmn(0:mf+nf,0,1:nf) + cmn(0:mf+nf,0,:nf-1))
232 cmns(0:mf+nf,0,0) = (p5*alp)*(cmn(0:mf+nf,0,0) + cmn(0:mf+nf,0,0))
233
234 precal_done = .true.
235
236 if (open_dbg_context("vac1n_precal", id=icall)) then
237
238 call add_int("nvper", nvper)
239 call add_int("nuv_tan", nuv_tan)
240
241 call add_real_1d("cosper", nvper, cosper)
242 call add_real_1d("sinper", nvper, sinper)
243
244 call add_real_1d("tanu_1d", 2*nu, tanu_1d)
245 call add_real_1d("tanv_1d", nuv_tan/(2*nu), tanv_1d)
246
247 call add_real_2d("tanu", 2*nu, nuv_tan/(2*nu), tanu)
248 call add_real_2d("tanv", 2*nu, nuv_tan/(2*nu), tanv)
249
250 call add_real_3d("cmns", mf+nf+1, mf+1, nf+1, cmns)
251
252 call close_dbg_out()
253 end if
254
255END SUBROUTINE precal

24 logical, intent(in) :: lasym

30 ! THIS ROUTINE COMPUTES THE SURFACE VALUES OF R,Z AND DERIVATIVES
31 !
32 ! Compute R & Z (and their derivatives) on surface
33 !
34 ! R = SUM [RC(m,n)*COS(mu - nv) + RS(m,n)*SIN(mu - nv)]
35 ! Z = SUM [ZS(m,n)*SIN(mu - nv) + ZC(m,n)*COS(mu - nv)]
36 !
37 ! NOTE: u, v here are actual angles (0, 2pi), NOT the normalized
38 ! variables used in PKM paper
39
40 r1b = 0; rub = 0; rvb = 0; ruu = 0; ruv = 0; rvv = 0
41 z1b = 0; zub = 0; zvb = 0; zuu = 0; zuv = 0; zvv = 0
42
43 DO mn = 1, mnmax
44 m = nint(xm(mn))
45 n = nint(xn(mn)/(nfper))
46 n1 = abs(n)
47 DO i = 1, nuv2
48 cosmn1 = cosu1(i,m)*cosv1(i,n1) + csign(n)*sinu1(i,m)*sinv1(i,n1)
49 sinmn1 = sinu1(i,m)*cosv1(i,n1) - csign(n)*cosu1(i,m)*sinv1(i,n1)
50 r1b(i) = r1b(i) + rc(mn) * cosmn1
51 rub(i) = rub(i) - xm(mn) * rc(mn) * sinmn1
52 rvb(i) = rvb(i) + xn(mn) * rc(mn) * sinmn1
53 z1b(i) = z1b(i) + zs(mn) * sinmn1
54 zub(i) = zub(i) + xm(mn) * zs(mn) * cosmn1
55 zvb(i) = zvb(i) - xn(mn) * zs(mn) * cosmn1
56 ruu(i) = ruu(i) - xm(mn) * xm(mn) * rc(mn) * cosmn1
57 ruv(i) = ruv(i) + xm(mn) * xn(mn) * rc(mn) * cosmn1
58 rvv(i) = rvv(i) - xn(mn) * xn(mn) * rc(mn) * cosmn1
59 zuu(i) = zuu(i) - xm(mn) * xm(mn) * zs(mn) * sinmn1
60 zuv(i) = zuv(i) + xm(mn) * xn(mn) * zs(mn) * sinmn1
61 zvv(i) = zvv(i) - xn(mn) * xn(mn) * zs(mn) * sinmn1
62 IF (lasym) THEN
63 r1b(i) = r1b(i) + rs(mn) * sinmn1
64 rub(i) = rub(i) + xm(mn) * rs(mn) * cosmn1
65 rvb(i) = rvb(i) - xn(mn) * rs(mn) * cosmn1
66 z1b(i) = z1b(i) + zc(mn) * cosmn1
67 zub(i) = zub(i) - xm(mn) * zc(mn) * sinmn1
68 zvb(i) = zvb(i) + xn(mn) * zc(mn) * sinmn1
69 ruu(i) = ruu(i) - xm(mn) * xm(mn) * rs(mn) * sinmn1
70 ruv(i) = ruv(i) + xm(mn) * xn(mn) * rs(mn) * sinmn1
71 rvv(i) = rvv(i) - xn(mn) * xn(mn) * rs(mn) * sinmn1
72 zuu(i) = zuu(i) - xm(mn) * xm(mn) * zc(mn) * cosmn1
73 zuv(i) = zuv(i) + xm(mn) * xn(mn) * zc(mn) * cosmn1
74 zvv(i) = zvv(i) - xn(mn) * xn(mn) * zc(mn) * cosmn1
75 END IF
76 END DO
77 END DO
78
79 !print *, "max R = ", maxval(r1b)
80
81 ! COMPUTE METRIC COEFFICIENTS GIJ_B AND SURFACE NORMAL COMPONENTS
82 ! [SNR, SNV, SNZ] = NP*[Xu cross Xv]
83 !
84 ! NOTE: These should be multiplied by -signgs to point OUTWARD from vacuum INTO plasma
85 ! for either handed-ness of the coordinate system
86 !
87 ! Eq. 2.4 in PKM has wrong sign for a left-handed coordinate system
88 !
89 ! NOTE: guv = .5*np guv_b; gvv = np*np* gvv_b, where GUV, GVV are the
90 ! REAL metric elements. CAP(A), etc. defined in Eq. (2.13) of PKM paper
91 !
92 ! AUU == NP*CAP(A) = .5*Xuu dot [Xu cross Xv] * NP
93 !
94 ! AUV == 2*NP*CAP(B) = Xuv dot [Xu cross Xv] * NP
95 !
96 ! AVV == NP*CAP(C) = .5*Xvv dot [Xu cross Xv] * NP
97 DO i = 1,nuv2
98 guu_b(i) = rub(i)*rub(i) + zub(i)*zub(i)
99 guv_b(i) = (rub(i)*rvb(i)+ zub(i)*zvb(i))*onp*2
100 gvv_b(i) = (rvb(i)*rvb(i)+ zvb(i)*zvb(i)+r1b(i)*r1b(i))*onp2
101 rzb2(i) = r1b(i)*r1b(i) + z1b(i)*z1b(i)
102 snr(i) = signgs*r1b(i)*zub(i)
103 snv(i) = signgs*(rub(i)*zvb(i) - rvb(i)*zub(i))
104 snz(i) =-signgs*r1b(i)*rub(i)
105 drv(i) = -(r1b(i)*snr(i) + z1b(i)*snz(i))
106 auu(i) = p5*(snr(i)*ruu(i) + snz(i)*zuu(i))
107 auv(i) = (snr(i)*ruv(i) + snv(i)*rub(i) + snz(i)*zuv(i))*onp
108 avv(i) = (snv(i)*rvb(i) + p5*(snr(i)*(rvv(i) - r1b(i)) + snz(i)* zvv(i)))*onp2
109 END DO
110
111 IF (.NOT.lasym) THEN
112 DO i = 1 + nv, nuv2 - nv
113 rzb2(imirr(i)) = rzb2(i)
114 r1b(imirr(i)) = r1b(i)
115 z1b(imirr(i)) =-z1b(i)
116 END DO
117 END IF
118
119 DO i = 1,nuv
120 rcosuv(i) = r1b(i)*cosuv(i)
121 rsinuv(i) = r1b(i)*sinuv(i)
122 END DO
123
124 if (open_dbg_context("vac1n_surface", id=icall)) then
125 call add_real_2d("r1b", nv, nu, r1b) ! nu !
126 call add_real_2d("rub", nv, nu3, rub)
127 call add_real_2d("rvb", nv, nu3, rvb)
128 call add_real_2d("ruu", nv, nu3, ruu)
129 call add_real_2d("ruv", nv, nu3, ruv)
130 call add_real_2d("rvv", nv, nu3, rvv)
131
132 call add_real_2d("z1b", nv, nu, z1b) ! nu !
133 call add_real_2d("zub", nv, nu3, zub)
134 call add_real_2d("zvb", nv, nu3, zvb)
135 call add_real_2d("zuu", nv, nu3, zuu)
136 call add_real_2d("zuv", nv, nu3, zuv)
137 call add_real_2d("zvv", nv, nu3, zvv)
138
139 call add_real_2d("guu_b", nv, nu3, guu_b)
140 call add_real_2d("guv_b", nv, nu3, guv_b)
141 call add_real_2d("gvv_b", nv, nu3, gvv_b)
142
143 call add_real_2d("rzb2", nv, nu, rzb2) ! nu !
144
145 call add_real_2d("snr", nv, nu3, snr)
146 call add_real_2d("snv", nv, nu3, snv)
147 call add_real_2d("snz", nv, nu3, snz)
148
149 call add_real_2d("drv", nv, nu3, drv)
150
151 call add_real_2d("auu", nv, nu3, auu)
152 call add_real_2d("auv", nv, nu3, auv)
153 call add_real_2d("avv", nv, nu3, avv)
154
155 call add_real_2d("rcosuv", nv, nu, rcosuv) ! nu !
156 call add_real_2d("rsinuv", nv, nu, rsinuv) ! nu !
157
158 call close_dbg_out()
159 end if

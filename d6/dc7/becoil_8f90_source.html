13  INTEGER, SAVE :: icount = 0

15  REAL(rprec) :: rad0, zee0, ri, zj, pr, qz, w22, w21, w12, w11

17  ! DETERMINE THE CYLINDRICAL COMPONENTS OF THE EXTERNAL
18  ! MAGNETIC FIELD (BR, BP, BZ) AT A FIXED PHI PLANE BY
19  ! USING 2-D INTERPOLATION BASED ON THE FOUR POINT FORMULA
20  ! IN ABRAMOWITZ AND STEGUN, EQ. 25.2.66
21  !
22  ! BRVAC, BPVAC, BZVAC: CYLINDRICAL COMPONENTS OF VACUUM B-FIELD
23  ! STORED ON R, Z, PHI GRID
24  !
25  ! RAD, ZEE: ARRAY OF R, Z VALUES (ON FLUX SURFACE)
26  ! AT WHICH B-FIELD IS TO BE DETERMINED
27 
28  icount = icount + 1
29 
30  DO i = 1, nuv2
31 
32  ! CHECK THAT BOUNDARY POINTS ARE INSIDE VACUUM GRID.
33  ! IF NOT, SET THEM EQUAL TO LARGEST (OR SMALLEST) VACUUM GRID POINTS
34  rad0 = min(rad(i), rmaxb)
35  rad0 = max(rad0, rminb)
36  zee0 = min(zee(i), zmaxb)
37  zee0 = max(zee0, zminb)
38 
39  ! DETERMINE PHI-PLANE, KV (MUST LIE IN FIRST FIELD PERIOD)
40  kv = 1 + mod(i - 1,nv)
41 
42  ! Axi-symmetric special case
43  kv = min(kv, np0b)
44 
45  ! DETERMINE INTEGER INDICES (IR,JZ) FOR LOWER LEFT R, Z CORNER GRID POINT
46  ir = int((rad0 - rminb)/delrb) + 1
47  jz = int((zee0 - zminb)/delzb) + 1
48  ir1 = min(nr0b,ir + 1)
49  jz1 = min(nz0b,jz + 1)
50 
51  ! COMPUTE RI, ZJ AND PR , QZ AT GRID POINT (IR , JZ)
52  ! ALSO, COMPUTE WEIGHTS WIJ FOR 4 CORNER GRID POINTS
53  ri = rminb + (ir - 1)*delrb
54  zj = zminb + (jz - 1)*delzb
55  pr = (rad0 - ri)/delrb
56  qz = (zee0 - zj)/delzb
57  w22 = pr*qz ! p * q
58  w21 = pr - w22 ! p *(1-q)
59  w12 = qz - w22 ! (1-p)* q
60  w11 = 1 + w22 - (pr + qz) ! (1-p)*(1-q)
61 
62  ! COMPUTE B FIELD AT R, PHI, Z BY INTERPOLATION
63  brad(i) = w11*brvac(ir,jz,kv) + w22*brvac(ir1,jz1,kv) + w21*brvac(ir1,jz,kv) + w12*brvac(ir,jz1,kv)
64  bphi(i) = w11*bpvac(ir,jz,kv) + w22*bpvac(ir1,jz1,kv) + w21*bpvac(ir1,jz,kv) + w12*bpvac(ir,jz1,kv)
65  bz(i) = w11*bzvac(ir,jz,kv) + w22*bzvac(ir1,jz1,kv) + w21*bzvac(ir1,jz,kv) + w12*bzvac(ir,jz1,kv)
66  END DO
67 
68  IF (mod(icount,25).eq.0) THEN
69  ! PRINT INFO IF R, Z OUT OF BOX
70  i = 0
71  rad0 = maxval(rad)
72  zee0 = maxval(zee)
73  IF (rad0 .gt. rmaxb) i = 1
74  IF (zee0 .gt. zmaxb) i = i + 2
75  ri = minval(rad)
76  zj = minval(zee)
77  IF (ri .lt. rminb) i = i + 4
78  IF (zj .lt. zminb) i = i + 8
79  IF (i .ne. 0) THEN
80  print *, warning
81  ! WRITE (nthreed, *) warning
82  IF (i/8 .ne. 0) print *,' zmin = ', zj
83  i = mod(i,8)
84  IF (i/4 .ne. 0) print *,' rmin = ', ri
85  i = mod(i,4)
86  IF (i/2 .ne. 0) print *,' zmax = ', zee0
87  i = mod(i,2)
88  IF (i .ne. 0) print *,' rmax = ', rad0
89  END IF
90  ENDIF

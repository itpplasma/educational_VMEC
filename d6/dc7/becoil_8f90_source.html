30  ! DETERMINE THE CYLINDRICAL COMPONENTS OF THE EXTERNAL
31  ! MAGNETIC FIELD (BR, BP, BZ) AT A FIXED PHI PLANE BY
32  ! USING 2-D INTERPOLATION BASED ON THE FOUR POINT FORMULA
33  ! IN ABRAMOWITZ AND STEGUN, EQ. 25.2.66
34  !
35  ! BRVAC, BPVAC, BZVAC: CYLINDRICAL COMPONENTS OF VACUUM B-FIELD
36  ! STORED ON R, Z, PHI GRID
37  !
38  ! RAD, ZEE: ARRAY OF R, Z VALUES (ON FLUX SURFACE)
39  ! AT WHICH B-FIELD IS TO BE DETERMINED
40 
41  icount = icount + 1
42 
43  DO i = 1, nuv2
44 
45  ! CHECK THAT BOUNDARY POINTS ARE INSIDE VACUUM GRID.
46  ! IF NOT, SET THEM EQUAL TO LARGEST (OR SMALLEST) VACUUM GRID POINTS
47  rad0 = min(rad(i), rmaxb)
48  rad0 = max(rad0, rminb)
49  zee0 = min(zee(i), zmaxb)
50  zee0 = max(zee0, zminb)
51 
52  ! DETERMINE PHI-PLANE, KV (MUST LIE IN FIRST FIELD PERIOD)
53  kv = 1 + mod(i - 1,nv)
54 
55  ! Axi-symmetric special case
56  kv = min(kv, np0b)
57 
58  ! DETERMINE INTEGER INDICES (IR,JZ) FOR LOWER LEFT R, Z CORNER GRID POINT
59  ir = int((rad0 - rminb)/delrb) + 1
60  jz = int((zee0 - zminb)/delzb) + 1
61  ir1 = min(nr0b,ir + 1)
62  jz1 = min(nz0b,jz + 1)
63 
64  ! COMPUTE RI, ZJ AND PR , QZ AT GRID POINT (IR , JZ)
65  ! ALSO, COMPUTE WEIGHTS WIJ FOR 4 CORNER GRID POINTS
66  ri = rminb + (ir - 1)*delrb
67  zj = zminb + (jz - 1)*delzb
68  pr = (rad0 - ri)/delrb
69  qz = (zee0 - zj)/delzb
70  w22 = pr*qz ! p * q
71  w21 = pr - w22 ! p *(1-q)
72  w12 = qz - w22 ! (1-p)* q
73  w11 = 1 + w22 - (pr + qz) ! (1-p)*(1-q)
74 
75  ! COMPUTE B FIELD AT R, PHI, Z BY INTERPOLATION
76  brad(i) = w11*brvac(ir,jz,kv) + w22*brvac(ir1,jz1,kv) + w21*brvac(ir1,jz,kv) + w12*brvac(ir,jz1,kv)
77  bphi(i) = w11*bpvac(ir,jz,kv) + w22*bpvac(ir1,jz1,kv) + w21*bpvac(ir1,jz,kv) + w12*bpvac(ir,jz1,kv)
78  bz(i) = w11*bzvac(ir,jz,kv) + w22*bzvac(ir1,jz1,kv) + w21*bzvac(ir1,jz,kv) + w12*bzvac(ir,jz1,kv)
79  END DO

91  IF (zj .lt. zminb) i = i + 8
92  IF (i .ne. 0) THEN
93  print *, warning

95  IF (i/8 .ne. 0) print *,' zmin = ', zj

98  i = mod(i,4)
99  IF (i/2 .ne. 0) print *,' zmax = ', zee0
100  i = mod(i,2)
101  IF (i .ne. 0) print *,' rmax = ', rad0
102  END IF
103  ENDIF

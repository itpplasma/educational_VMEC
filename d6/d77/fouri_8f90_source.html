17 REAL(rprec), DIMENSION(mnpd,nv,nu3,ndim), INTENT(in) :: grpmn
18 REAL(rprec), DIMENSION(nuv), INTENT(in) :: gsource
19 REAL(rprec), DIMENSION(mnpd,mnpd,ndim**2), INTENT(out) :: amatrix
20 REAL(rprec), DIMENSION(mnpd2,mnpd2), INTENT(out) :: amatsq

31 ! AMATRIX(,1) = A(Sin)(Sin'); AMATRIX(,2) = A(Sin)(Cos');
32 ! AMATRIX(,3) = A(Cos)(Sin'); AMATRIX(,4) = A(Cos)(Cos')
33 !
34 ! ARG OF TRIG FUNCTIONS: m'u' - n'v' CORRESPONDING TO THE PRIMED MESH IN
35 ! PKM EQ.(2.14), AND mu - nv (UNPRIMED MESH) IN EQ. (2.16)
36 !
37 ! ON ENTRY, GRPMN(MN,...) HAS BEEN FOURIER-TRANSFORMED OVER THE UNPRIMED
38 ! COORDINATES (IN FOURP), SO THE FIRST INDEX OF GRPMN CORRESPONDS TO THE FIRST
39 ! INDEX OF AMATRIX. THUS, THE FOURIER TRANSFORMS OF GRPMN HERE ARE OVER THE PRIMED MESH.
40 !
41 ! IN CONTRAST, THE INTEGRAL OF THE SOURCE TERM OVER THE PRIMED MESH WAS ALREADY
42 ! DONE (IN SCALPOT), SO HERE THE FT ARE OVER THE UNPRIMED MESH FOR THE SOURCE.
43
44 ! SYMMETRIZE SOURCE TERMS (with respect to u,v and -u,-v)
45 ! INDEX (1) IS ANTI-SYMMETRIC, INDEX (2) IS SYMMETRIC, PART
46 !
47 ! GSOURCE = - (2pi)**2 B * dS (h - hsing) * WINT
48 !
49 ! WINT: needed to normalize integral over angles to unity
50 k = 0
51 DO i = 1, nu2
52 DO j = 1, nv
53 k = k + 1
54
55 ! anti-symmetric part
56 source(j,i,1) = gsource(k) - gsource(imirr(k))
57
58 ! gsource_sym.dat
59 ! print *, i, j, source(j,i,1)
60
61 IF (lasym) then
62 ! symmetric part
63 source(j,i,2) = gsource(k) + gsource(imirr(k))
64 end if
65 END DO
66 END DO
67
68 source = p5*onp*source ! TODO: why is this required?
69
70 ! INITIALIZE RUNNING-SUM ARRAYS
71 bcos = 0
72 bsin = 0
73 actemp = 0
74 astemp = 0
75
76 amatrix = 0
77
78 ! PERFORM KV (TOROIDAL ANGLE) TRANSFORM
79 DO n = 0, nf
80 DO kvi = 1, nv
81 cosn = cosv(n,kvi)
82 sinn = sinv(n,kvi)
83 DO isym = 1, ndim
84 bcos(:,n,isym) = bcos(:,n,isym) + cosn*source(kvi,:,isym)
85 bsin(:,n,isym) = bsin(:,n,isym) + sinn*source(kvi,:,isym)
86
87 actemp(:,n,:,isym) = actemp(:,n,:,isym) + cosn*grpmn(:,kvi,:,isym)
88 astemp(:,n,:,isym) = astemp(:,n,:,isym) + sinn*grpmn(:,kvi,:,isym)
89
90 IF (n .ne. 0) THEN
91 bcos(:,(-n),isym) = bcos(:,n,isym)
92 bsin(:,(-n),isym) = -bsin(:,n,isym)
93
94 actemp(:,(-n),:,isym) = actemp(:,n,:,isym)
95 astemp(:,(-n),:,isym) = -astemp(:,n,:,isym)
96 ENDIF
97 END DO
98 END DO
99 END DO
100
101 ! PERFORM KU (POLOIDAL ANGLE) TRANSFORM
102 DO m = 0, mf
103 DO kui = 1, nu2
104 cosm = cosui(m,kui)
105 sinm = sinui(m,kui)
106 bvec(m,-nf:nf,1) = bvec(m,-nf:nf,1) + bcos(kui,-nf:nf,1)*sinm - bsin(kui,-nf:nf,1)*cosm
107 IF (lasym) THEN
108 bvec(m,-nf:nf,2) = bvec(m,-nf:nf,2) + bcos(kui,-nf:nf,2)*cosm + bsin(kui,-nf:nf,2)*sinm
109 END IF
110 END DO
111
112 ! RECALL, LAST INDEX OF A(S,C)TEMP
113 ! = 1 CORRESPONDS TO SIN (UNPRIMED) TRANSFORM (FIRST INDEX OF AMATRIX)
114 ! = 2 CORRESPONDS TO COS (UNPRIMED) TRANSFORM
115 DO kui = 1, nu3
116 cosm = cosu(m,kui)*wint(kui*nv) ! wint only varies in the poloidal direction --> only need every nv-th entry
117 sinm = sinu(m,kui)*wint(kui*nv)
118
119 ! SIN SIN'
120 amatrix(:,m+1:mnpd:mf1,1) = amatrix(:,m+1:mnpd:mf1,1) + sinm*actemp(:,-nf:nf,kui,1) - cosm*astemp(:,-nf:nf,kui,1)
121
122 IF (lasym) then
123 ! SIN COS'
124 amatrix(:,m+1:mnpd:mf1,2) = amatrix(:,m+1:mnpd:mf1,2) + cosm*actemp(:,-nf:nf,kui,1) + sinm*astemp(:,-nf:nf,kui,1)
125
126 ! COS SIN'
127 amatrix(:,m+1:mnpd:mf1,3) = amatrix(:,m+1:mnpd:mf1,3) + sinm*actemp(:,-nf:nf,kui,2) - cosm*astemp(:,-nf:nf,kui,2)
128
129 ! COS COS'
130 amatrix(:,m+1:mnpd:mf1,4) = amatrix(:,m+1:mnpd:mf1,4) + cosm*actemp(:,-nf:nf,kui,2) + sinm*astemp(:,-nf:nf,kui,2)
131 end if
132
133 END DO
134 END DO
135
136
137 ! debugging: first focus on above Fourier transforms
138 ! return
139
140
141 ! The code below is not related to Fourier transforms anymore,
142 ! so it should probably go into the main vacuum routine...
143
144 amatrix = (pi2*pi2)*amatrix ! TODO: can this be cancelled with some factors of 2*pi in the rest of NESTOR?
145
146
147
148
149
150 ! ZERO BVEC(0,n) FOR n < 0 (TODO: why ?)
151 ! Fixed SPH081515: had -nf:0 before
152 bvec(0,-nf:-1,1:ndim) = 0
153
154 ! ZERO AMATRIX(0,n,m',n') M = 0 MODES FOR n < 0 (TODO: why ?)
155 ! Index of m=0,n=0
156 mn0 = 1+mf1*nf
157 ! SPH082415: mn0-mf1: (m=0,n=-1 index)
158 amatrix(1:mn0-mf1:mf1, :, 1:ndim*ndim) = 0
159
160
161
162
163 ! ADD DIAGONAL TERMS TO AMATRIX [THE FIRST TERM IN EQ(3.2) OF PKM]
164 DO mn = 1, mnpd
165 amatrix(mn,mn,1) = amatrix(mn,mn,1) + pi3*int_ext
166 END DO
167
168 IF (lasym) THEN
169 DO mn = 1, mnpd
170 ! last index == 4 is the lower-right corner of amatrix
171 ! and this corner is crossed by the main diagnoal as well
172 ! --> no need to touch 2 and 3 (off-diagonal blocks of amatrix)
173 amatrix(mn,mn,4) = amatrix(mn,mn,4) + pi3*int_ext
174 END DO
175
176 ! COS(0-0) mode *2 (TODO: why is this required ?)
177 amatrix(mn0,mn0,4) = amatrix(mn0,mn0,4) + pi3*int_ext
178 END IF
179
180
181 ! PUT ELEMENTS INTO SQUARE MATRIX
182
183 ! Sin-Sin'
184 amatsq(:mnpd,:mnpd) = amatrix(:,:,1)
185
186 IF (lasym) then
187 ! Sin-Cos'
188 amatsq(:mnpd,1+mnpd:mnpd2) = amatrix(:,:,2)
189
190 ! Cos-Sin'
191 amatsq(1+mnpd:mnpd2,:mnpd) = amatrix(:,:,3)
192
193 ! Cos-Cos'
194 amatsq(1+mnpd:mnpd2,1+mnpd:mnpd2) = amatrix(:,:,4)
195 end if

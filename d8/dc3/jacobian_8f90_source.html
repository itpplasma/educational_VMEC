23  ! SQRT(G) = R*TAU IS COMPUTED IN BCOVAR

32  zs(l) = ohs*( z1(l,meven) - z1(l-1,meven) + shalf(l)*(z1(l,modd) - z1(l-1,modd)) ) ! dZ/ds on half grid (?)
33  rs(l) = ohs*( r1(l,meven) - r1(l-1,meven) + shalf(l)*(r1(l,modd) - r1(l-1,modd)) ) ! dR/ds on half grid (?)
34  r12(l) = p5*( r1(l,meven) + r1(l-1,meven) + shalf(l)*(r1(l,modd) + r1(l-1,modd)) ) ! 0.5*R on half grid (?)
35  ru12(l) = p5*( ru(l,meven) + ru(l-1,meven) + shalf(l)*(ru(l,modd) + ru(l-1,modd)) ) ! 0.5*dR/du on half grid (?)
36  zu12(l) = p5*( zu(l,meven) + zu(l-1,meven) + shalf(l)*(zu(l,modd) + zu(l-1,modd)) ) ! 0.5*dZ/du on half grid (?)
37 
38  ! TODO: the lower four lines of below expression could be split off into some separate variables...
39  tau(l) = ru12(l)*zs(l) - rs(l)*zu12(l) + dshalfds* &
40  ( ru(l,modd) *z1(l,modd) + ru(l-1,modd) *z1(l-1,modd) &
41  - zu(l,modd) *r1(l,modd) - zu(l-1,modd) *r1(l-1,modd) &
42  + ( ru(l,meven)*z1(l,modd) + ru(l-1,meven)*z1(l-1,modd) &
43  - zu(l,meven)*r1(l,modd) - zu(l-1,meven)*r1(l-1,modd) )/shalf(l) )
44  END DO
45 
46  ! TEST FOR SIGN CHANGE IN JACOBIAN

51  IF (taumax*taumin .lt. zero) then
52  ! bad jacobian !

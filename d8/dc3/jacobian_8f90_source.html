8 USE vmec_params, ONLY: meven, modd
9 use vmec_input, only: input_extension, nzeta
10 USE realspace
11 USE vmec_dim, ONLY: ns, ntheta3
12 USE vforces, r12 => armn_o, ru12 => azmn_e, zu12 => armn_e, &
13 rs => bzmn_e, zs => brmn_e, tau => azmn_o
14

19 REAL(rprec), PARAMETER :: zero=0, p5=0.5_dp, p25=p5*p5
20
21 INTEGER :: l, js, ku, lk
22 REAL(rprec) :: taumax, taumin, dshalfds=p25, temp(nrzt/ns)
23
24 ! (RS, ZS)=(R, Z) SUB S, (RU12, ZU12)=(R, Z) SUB THETA(=U)
25 ! AND TAU=SQRT(G)/R ARE DIFFERENCED ON HALF MESH
26 !
27 ! SQRT(G) = R*TAU IS COMPUTED IN BCOVAR
28 !
29 ! HERE, TAU = (Ru * Zs - Rs * Zu).
30
31 ! THE DERIVATIVES OF SHALF = SQRT(s) WERE COMPUTED EXPLICITLY AS: d(shalf)/ds = .5/shalf
32 ! This can be understood by noting shalf(i) = sqrt(s_{i-1/2}),
33 ! leading to d/ds sqrt(s_{i-1/2}) = 1/2 * (s_{i-1/2})^{-1/2} = 0.5 / shalf.
34
35 ! initially, all good
36 first = 1
37
38 DO l = 2,nrzt
39 r12(l) = p5*( r1(l,meven) + r1(l-1,meven) + shalf(l)*(r1(l,modd) + r1(l-1,modd)) ) ! R on half grid
40
41 ru12(l) = p5*( ru(l,meven) + ru(l-1,meven) + shalf(l)*(ru(l,modd) + ru(l-1,modd)) ) ! dR/du on half grid
42 zu12(l) = p5*( zu(l,meven) + zu(l-1,meven) + shalf(l)*(zu(l,modd) + zu(l-1,modd)) ) ! dZ/du on half grid
43
44 rs(l) = ohs*( r1(l,meven) - r1(l-1,meven) + shalf(l)*(r1(l,modd) - r1(l-1,modd)) ) ! dR/ds on half grid
45 zs(l) = ohs*( z1(l,meven) - z1(l-1,meven) + shalf(l)*(z1(l,modd) - z1(l-1,modd)) ) ! dZ/ds on half grid
46
47 ! TODO: the lower four lines of below expression could be split off into some separate variables...
48 tau(l) = ru12(l)*zs(l) - rs(l)*zu12(l) + dshalfds* &
49 ( ru(l,modd) *z1(l,modd) + ru(l-1,modd) *z1(l-1,modd) &
50 - zu(l,modd) *r1(l,modd) - zu(l-1,modd) *r1(l-1,modd) &
51 + ( ru(l,meven)*z1(l,modd) + ru(l-1,meven)*z1(l-1,modd) &
52 - zu(l,meven)*r1(l,modd) - zu(l-1,meven)*r1(l-1,modd) )/shalf(l) )
53 END DO
54
55 ! extrapolate as constant to axis
56 temp(:) = tau(2:nrzt:ns)
57 tau(1:nrzt:ns) = temp(:)
58
59 ! check output from jacobian()
60 if (dump_jacobian .and. iter2.le.max_dump) then
61 write(dump_filename, 998) ns, iter2, trim(input_extension)
62998 format('jacobian_',i5.5,'_',i6.6,'.',a,'.json')
63
64 call open_dbg_out(dump_filename)
65
66 call add_real_3d("r12", ns, nzeta, ntheta3, &
67 reshape(r12, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
68 call add_real_3d("ru12", ns, nzeta, ntheta3, &
69 reshape(ru12, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
70 call add_real_3d("zu12", ns, nzeta, ntheta3, &
71 reshape(zu12, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
72 call add_real_3d("rs", ns, nzeta, ntheta3, &
73 reshape(rs, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
74 call add_real_3d("zs", ns, nzeta, ntheta3, &
75 reshape(zs, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
76 call add_real_3d("tau", ns, nzeta, ntheta3, &
77 reshape(tau, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
78
79 call close_dbg_out()
80 end if
81
82! TEST FOR SIGN CHANGE IN JACOBIAN
83 taumax = maxval(tau(2:nrzt))
84 taumin = minval(tau(2:nrzt))
85 IF (taumax*taumin .lt. zero) then
86 ! bad jacobian !
87 first = 2
88 end if
89

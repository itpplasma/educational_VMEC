466 ! Scaling of cos**2 in alias (4*r0scale**2)
467
468! print *, "tcon_mul = ", tcon_mul

472 aznorm = sum(wint(js:nrzt:ns)*zu0(js:nrzt:ns)**2)
473 IF (arnorm.eq.zero .or. aznorm.eq.zero) then
474 stop 'arnorm or aznorm=0 in bcovar'
475 end if

477 tcon(js) = min(abs(ard(js,1)/arnorm), abs(azd(js,1)/aznorm)) * tcon_mul*(32*hs)**2
478 END DO

480! print *, "tcon profile: ", tcon

515 ENDIF ! MOD(iter2-iter1,ns4).eq.0

519 bsubv_e = -lamscale*bsubv_e
520 bsubu_o(:nrzt) = sqrts(:nrzt)*bsubu_e(:nrzt)

527 gvv(2:nrzt) = bsupv(2:nrzt)*bsupv(2:nrzt)*lvv(2:nrzt)
528 lv(2:nrzt,0) = bsq(2:nrzt)*tau(2:nrzt)
529 lu(2:nrzt,0) = bsq(2:nrzt)*r12(2:nrzt)

532 write(dump_filename, 988) ns, iter2, trim(input_extension)
533988 format('lulv_comb_',i5.5,'_',i6.6,'.',a,'.json')
534
535 call open_dbg_out(dump_filename)
536
537 call add_real_3d("bsubu_e", ns, nzeta, ntheta3, &
538 reshape(bsubu_e, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
539 call add_real_3d("bsubv_e", ns, nzeta, ntheta3, &
540 reshape(bsubv_e, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
541 call add_real_3d("bsubu_o", ns, nzeta, ntheta3, &
542 reshape(bsubu_o, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
543 call add_real_3d("bsubv_o", ns, nzeta, ntheta3, &
544 reshape(bsubv_o, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
545 call add_real_3d("lvv", ns, nzeta, ntheta3, &
546 reshape(lvv, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
547 call add_real_3d("guu", ns, nzeta, ntheta3, &
548 reshape(guu, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
549 call add_real_3d("guv", ns, nzeta, ntheta3, &
550 reshape(guv, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
551 call add_real_3d("gvv", ns, nzeta, ntheta3, &
552 reshape(gvv, (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
553 call add_real_3d("lv_even_m", ns, nzeta, ntheta3, &
554 reshape(lv(:, 0), (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
555 call add_real_3d("lu_even_m", ns, nzeta, ntheta3, &
556 reshape(lu(:, 0), (/ ns, nzeta, ntheta3 /), order=(/ 2, 3, 1 /) ) )
557
558 call close_dbg_out()
559 end if
560
561 ELSE ! (iequi .eq. 0)
562
563 ! iequi == 1 --> final iter for fileout()
564
565 ! NOTE THAT lu=>czmn, lv=>crmn externally
566 ! SO THIS STORES bsupv in czmn_e, bsupu in crmn_e
567 ! only executed at end of equilibrium
568 lu(:nrzt,0) = bsupv(:nrzt)
569 lv(:nrzt,0) = bsupu(:nrzt)
570
571 ! COMPUTE COVARIANT BSUBU,V (EVEN, ODD) ON HALF RADIAL MESH
572 ! FOR FORCE BALANCE AND RETURN (IEQUI=1)
573
574 ! final call from fileout --> compute additional stuff
575 DO js = ns-1,2,-1
576 DO l = js, nrzt, ns
577 bsubvh(l) = 2*bsubv_e(l) - bsubvh(l+1)
578 END DO
579 END DO
580
581 ! ADJUST <bsubvh> AFTER MESH-BLENDING
582 DO js = 2, ns
583 curpol_temp = fpsi(js) - sum(bsubvh(js:nrzt:ns)*wint(js:nrzt:ns))
584 DO l = js, nrzt, ns
585 bsubvh(l) = bsubvh(l) + curpol_temp
586 END DO
587 END DO
588
589 bsubu_e(:nrzt) = bsubuh(:nrzt)
590 bsubv_e(:nrzt) = bsubvh(:nrzt)
591
592 bsubu_o(:nrzt) = shalf(:nrzt)*bsubu_e(:nrzt)
593 bsubv_o(:nrzt) = shalf(:nrzt)*bsubv_e(:nrzt)
594 END IF ! (iequi .eq. 0)
595
596END SUBROUTINE bcovar

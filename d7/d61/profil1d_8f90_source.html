9SUBROUTINE profil1d()
10 USE vmec_main
11 USE vmec_params, ONLY: signgs, lamscale, rcc, pdamp
12 USE realspace, ONLY: shalf, sqrts
13 IMPLICIT NONE
14
15 REAL(rprec), PARAMETER :: c1p5 = 1.5_dp
16
17 INTEGER :: i
18 REAL(rprec) :: Itor, si, tf, pedge, vpnorm, torflux_edge, polflux_edge
19
20 REAL(rprec), EXTERNAL :: pcurr, pmass, piota, &
21 torflux, torflux_deriv, polflux, polflux_deriv
22
23 character(len=255) :: dump_filename
24 logical, parameter :: dump_profil1d = .false.
25
26 ! COMPUTE PHIP, IOTA PROFILES ON FULL-GRID
27 ! COMPUTE MASS PROFILE ON HALF-GRID BY READING INPUT COEFFICIENTS.
28 ! PRESSURE CONVERTED TO INTERNAL UNITS BY MULTIPLICATION BY mu0 = 4*pi*10**-7
29
30 torflux_edge = signgs * phiedge / twopi
31 si = torflux(one)
32 IF (si .ne. zero) torflux_edge = torflux_edge/si
33
34 polflux_edge = torflux_edge
35 si = polflux(one)
36 IF (si .ne. zero) polflux_edge = polflux_edge/si
37
38 ! z00 gets assiged in reset_params and in funct3d
39 r00 = rmn_bdy(0,0,rcc)
40
41 ! zero fluxes and current at magnetic axis
42 phips(1) = 0
43 chips(1) = 0
44 icurv(1) = 0
45
46 ! half-grid quantities: s_i are shifted inwards by 0.5 grid points
47 DO i = 2,ns
48 si = hs*(i-c1p5)
49 tf = min(one, torflux(si))
50 phips(i) = torflux_edge * torflux_deriv(si)
51 chips(i) = torflux_edge * polflux_deriv(si)
52 iotas(i) = piota(tf) ! evaluate iota profile
53 icurv(i) = pcurr(tf) ! evaluate current profile
54 END DO
55 phips(ns+1) = 2*phips(ns)-phips(ns-1) ! virtual point outside the LCFS
56
57 ! Compute lamscale factor for "normalizing" lambda (needed for scaling hessian)
58 lamscale = sqrt(hs*sum(phips(2:ns)**2))
59 IF (lamscale .EQ. 0) stop 'PHIP == 0: ERROR!'
60
61 IF (lflip) THEN
62 iotas = -iotas
63 chips = -chips
64 END IF
65
66 ! full-grid quantities
67 DO i = 1,ns
68 si = hs*(i-1)
69 tf = min(one, torflux(si))
70 phipf(i) = torflux_edge * torflux_deriv(si)
71 chipf(i) = torflux_edge * polflux_deriv(si)
72 iotaf(i) = piota(tf)
73 ENDDO
74
75 ! SCALE CURRENT TO MATCH INPUT EDGE VALUE, CURTOR
76 ! FACTOR OF SIGNGS NEEDED HERE, SINCE MATCH IS MADE TO LINE
77 ! INTEGRAL OF BSUBU (IN GETIOTA) ~ SIGNGS * CURTOR
78 pedge = pcurr(one) ! pcurr gives current enclosed between axis and given parameter
79 ! --> pedge is temporarily re-used to store the radial integral over current density profile (still in Amperes)
80 itor = 0
81 ! TODO: what is this? might have something to do with preventing division-by-zero
82 ! when computing scaled toroidal current profile from pcurr and curtor...
83 IF (abs(pedge) .gt. abs(epsilon(pedge)*curtor)) then
84 itor = signgs*currv/(twopi*pedge)
85 end if
86 icurv(2:ns) = itor*icurv(2:ns)
87
88 ! POSSIBLE PRESSURE PEDESTAL FOR S >= SPRES_PED
89 spres_ped = abs(spres_ped)
90 DO i = 2,ns
91 si = hs*(i - c1p5)
92 ! NORMALIZE mass so dV/dPHI (or dV/dPSI) in pressure to mass relation
93 ! See line 195 of bcovar: pres(2:ns) = mass(2:ns)/vp(2:ns)**gamma
94 tf = min(one, torflux(si))
95 vpnorm = torflux_edge * torflux_deriv(si)
96 IF (si .gt. spres_ped) THEN
97 pedge = pmass(spres_ped)
98 ELSE
99 pedge = pmass(tf)
100 END IF
101 mass(i) = pedge*(abs(vpnorm)*r00)**gamma
102 END DO
103
104 pres(:ns+1) = 0
105
106 DO i = 1, ns
107
108 ! sqrt(s_i) for half-grid s values (shifted inwards by -0.5 grid points)
109 si = hs*abs(i-1.5_dp)
110 shalf(i:nrzt:ns) = sqrt(si)
111
112 ! sqrt(s_i) for full-grid s values
113 si = hs*(i-1)
114 sqrts(i:nrzt:ns) = sqrt(si)
115
116 ! TODO: what is this _exactly_?
117 ! just observing, this is a linear profile on the half-grid
118 ! which is at 2*pdamp == 0.1 at the axis
119 ! and at ca. 0 at the LCFS
120 bdamp(i) = 2*pdamp*(1-si)
121 END DO
122
123 ! Avoid round-off
124 sqrts(ns:nrzt:ns) = 1 ! boundary value
125 shalf(nrzt+1) = 1 ! no scaling for the weird hidden value at the end of shalf (see ndim in allocate_ns)
126 sqrts(nrzt+1) = 1 ! no scaling for the weird hidden value at the end of sqrts (see ndim in allocate_ns)
127
128 ! sm, sp used mainly in 1d preconditioner?
129 DO i = 2,ns
130 sm(i) = shalf(i)/sqrts(i)
131 sp(i) = shalf(i+1)/sqrts(i)
132 ENDDO
133
134 sm(1) = 0
135 sp(0) = 0
136 sp(1) = sm(2)
137
138 ! dump all relevant output to a text file
139 if (dump_profil1d) then
140 write(dump_filename, 999) ns, trim(input_extension)
141999 format('profil1d_',i5.5,'.',a)
142
143 open(unit=42, file=trim(dump_filename), status="unknown")
144
145 write(42, *) "# torflux_edge polflux_edge r00 lamscale"
146 write(42, *) torflux_edge, polflux_edge, r00, lamscale
147
148 write(42, *) "# currv Itor"
149 write(42, *) currv, itor
150
151 write(42, *) "# shalf"
152 write(42, *) shalf(:ns)
153
154 write(42, *) "# phips"
155 write(42, *) phips
156
157 write(42, *) "# chips"
158 write(42, *) chips
159
160 write(42, *) "# iotas"
161 write(42, *) iotas
162
163 write(42, *) "# icurv"
164 write(42, *) icurv
165
166 write(42, *) "# mass"
167 write(42, *) mass
168
169 write(42, *) "# sqrts"
170 write(42, *) sqrts(:ns)
171
172 write(42, *) "# phipf"
173 write(42, *) phipf
174
175 write(42, *) "# chipf"
176 write(42, *) chipf
177
178 write(42, *) "# iotaf"
179 write(42, *) iotaf
180
181 write(42, *) "# bdamp"
182 write(42, *) bdamp
183
184 write(42, *) "# sm"
185 write(42, *) sm
186
187 write(42, *) "# sp"
188 write(42, *) sp
189
190 close(42)
191
192 print *, "dumped profil1d output to '"//trim(dump_filename)//"'"
193 end if

681  SUBROUTINE compute_currents(ierror)
682  USE stel_constants, ONLY: mu0
683  IMPLICIT NONE
684  INTEGER, INTENT(out) :: ierror
685 !-----------------------------------------------
686 ! L o c a l V a r i a b l e s
687 !-----------------------------------------------
688  INTEGER :: js
689  REAL(rprec) :: ohs, hs, shalf(ns), sfull(ns)
690  REAL(rprec), DIMENSION(mnmax_nyq) :: bu1, bu0, bv1, bv0, t1, t2,
691  & t3
692 !-----------------------------------------------
693 !
694 ! Computes current harmonics for currXmn == sqrt(g)*JsupX, X = u,v
695 ! [Corrected above "JsubX" to "JsupX", JDH 2010-08-16]
696 
697 ! NOTE: bsub(s,u,v)mn are on HALF radial grid
698 ! (in earlier versions, bsubsmn was on FULL radial grid)
699 
700 !
701  ohs = (ns-1)
702  hs = 1._dp/ohs
703 
704  DO js = 2, ns
705  shalf(js) = sqrt(hs*(js-1.5_dp))
706  sfull(js) = sqrt(hs*(js-1))
707  END DO
708 
709  ALLOCATE (currumnc(mnmax_nyq,ns), currvmnc(mnmax_nyq,ns), &
710  & stat=ierror)
711  IF (ierror .ne. 0) RETURN
712 
713  DO js = 2, ns-1
714  WHERE (mod(int(xm_nyq),2) .EQ. 1)
715  t1 = 0.5_dp*(shalf(js+1)*bsubsmns(:,js+1) + &
716  & shalf(js) *bsubsmns(:,js)) /sfull(js)
717  bu0 = bsubumnc(:,js )/shalf(js)
718  bu1 = bsubumnc(:,js+1)/shalf(js+1)
719  t2 = ohs*(bu1-bu0)*sfull(js)+0.25_dp*(bu0+bu1)/sfull(js)
720  bv0 = bsubvmnc(:,js )/shalf(js)
721  bv1 = bsubvmnc(:,js+1)/shalf(js+1)
722  t3 = ohs*(bv1-bv0)*sfull(js)+0.25_dp*(bv0+bv1)/sfull(js)
723  ELSEWHERE
724  t1 = 0.5_dp*(bsubsmns(:,js+1)+bsubsmns(:,js))
725  t2 = ohs*(bsubumnc(:,js+1)-bsubumnc(:,js))
726  t3 = ohs*(bsubvmnc(:,js+1)-bsubvmnc(:,js))
727  ENDWHERE
728  currumnc(:,js) = -xn_nyq(:)*t1 - t3
729  currvmnc(:,js) = -xm_nyq(:)*t1 + t2
730  END DO
731 
732  WHERE (xm_nyq .LE. 1)
733  currvmnc(:,1) = 2*currvmnc(:,2) - currvmnc(:,3)
734  currumnc(:,1) = 2*currumnc(:,2) - currumnc(:,3)
735  ELSEWHERE
736  currvmnc(:,1) = 0
737  currumnc(:,1) = 0
738  ENDWHERE
739 
740  currumnc(:,ns) = 2*currumnc(:,ns-1) - currumnc(:,ns-2)
741  currvmnc(:,ns) = 2*currvmnc(:,ns-1) - currvmnc(:,ns-2)
742  currumnc = currumnc/mu0; currvmnc = currvmnc/mu0
743 
744  IF (.NOT.lasym) RETURN
745 
746  ALLOCATE (currumns(mnmax_nyq,ns), currvmns(mnmax_nyq,ns), &
747  & stat=ierror)
748 
749  DO js = 2, ns-1
750  WHERE (mod(int(xm_nyq),2) .EQ. 1)
751  t1 = 0.5_dp*(shalf(js+1)*bsubsmnc(:,js+1) &
752  & + shalf(js) *bsubsmnc(:,js)) / sfull(js)
753  bu0 = bsubumns(:,js )/shalf(js+1)
754  bu1 = bsubumns(:,js+1)/shalf(js+1)
755  t2 = ohs*(bu1-bu0)*sfull(js) + 0.25_dp*(bu0+bu1)/sfull(js)
756  bv0 = bsubvmns(:,js )/shalf(js)
757  bv1 = bsubvmns(:,js+1)/shalf(js+1)
758  t3 = ohs*(bv1-bv0)*sfull(js)+0.25_dp*(bv0+bv1)/sfull(js)
759  ELSEWHERE
760  t1 = 0.5_dp*(bsubsmnc(:,js+1) + bsubsmnc(:,js))
761  t2 = ohs*(bsubumns(:,js+1)-bsubumns(:,js))
762  t3 = ohs*(bsubvmns(:,js+1)-bsubvmns(:,js))
763  END WHERE
764  currumns(:,js) = xn_nyq(:)*t1 - t3
765  currvmns(:,js) = xm_nyq(:)*t1 + t2
766  END DO
767 
768  WHERE (xm_nyq .LE. 1)
769  currvmns(:,1) = 2*currvmns(:,2) - currvmns(:,3)
770  currumns(:,1) = 2*currumns(:,2) - currumns(:,3)
771  ELSEWHERE
772  currvmns(:,1) = 0
773  currumns(:,1) = 0
774  END WHERE
775  currumns(:,ns) = 2*currumns(:,ns-1) - currumns(:,ns-2)
776  currvmns(:,ns) = 2*currvmns(:,ns-1) - currvmns(:,ns-2)
777  currumns = currumns/mu0; currvmns = currvmns/mu0
778 
779  END SUBROUTINE compute_currents

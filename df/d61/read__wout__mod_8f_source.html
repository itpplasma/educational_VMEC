678  SUBROUTINE compute_currents(ierror)
679  USE stel_constants, ONLY: mu0
680  IMPLICIT NONE
681  INTEGER, INTENT(out) :: ierror
682 !-----------------------------------------------
683 ! L o c a l V a r i a b l e s
684 !-----------------------------------------------
685  INTEGER :: js
686  REAL(rprec) :: ohs, hs, shalf(ns), sfull(ns)
687  REAL(rprec), DIMENSION(mnmax_nyq) :: bu1, bu0, bv1, bv0, t1, t2,
688  & t3
689 !-----------------------------------------------
690 !
691 ! Computes current harmonics for currXmn == sqrt(g)*JsupX, X = u,v
692 ! [Corrected above "JsubX" to "JsupX", JDH 2010-08-16]
693 
694 ! NOTE: bsub(s,u,v)mn are on HALF radial grid
695 ! (in earlier versions, bsubsmn was on FULL radial grid)
696 
697 !
698  ohs = (ns-1)
699  hs = 1._dp/ohs
700 
701  DO js = 2, ns
702  shalf(js) = sqrt(hs*(js-1.5_dp))
703  sfull(js) = sqrt(hs*(js-1))
704  END DO
705 
706  ALLOCATE (currumnc(mnmax_nyq,ns), currvmnc(mnmax_nyq,ns), &
707  & stat=ierror)
708  IF (ierror .ne. 0) RETURN
709 
710  DO js = 2, ns-1
711  WHERE (mod(int(xm_nyq),2) .EQ. 1)
712  t1 = 0.5_dp*(shalf(js+1)*bsubsmns(:,js+1) + &
713  & shalf(js) *bsubsmns(:,js)) /sfull(js)
714  bu0 = bsubumnc(:,js )/shalf(js)
715  bu1 = bsubumnc(:,js+1)/shalf(js+1)
716  t2 = ohs*(bu1-bu0)*sfull(js)+0.25_dp*(bu0+bu1)/sfull(js)
717  bv0 = bsubvmnc(:,js )/shalf(js)
718  bv1 = bsubvmnc(:,js+1)/shalf(js+1)
719  t3 = ohs*(bv1-bv0)*sfull(js)+0.25_dp*(bv0+bv1)/sfull(js)
720  ELSEWHERE
721  t1 = 0.5_dp*(bsubsmns(:,js+1)+bsubsmns(:,js))
722  t2 = ohs*(bsubumnc(:,js+1)-bsubumnc(:,js))
723  t3 = ohs*(bsubvmnc(:,js+1)-bsubvmnc(:,js))
724  ENDWHERE
725  currumnc(:,js) = -xn_nyq(:)*t1 - t3
726  currvmnc(:,js) = -xm_nyq(:)*t1 + t2
727  END DO
728 
729  WHERE (xm_nyq .LE. 1)
730  currvmnc(:,1) = 2*currvmnc(:,2) - currvmnc(:,3)
731  currumnc(:,1) = 2*currumnc(:,2) - currumnc(:,3)
732  ELSEWHERE
733  currvmnc(:,1) = 0
734  currumnc(:,1) = 0
735  ENDWHERE
736 
737  currumnc(:,ns) = 2*currumnc(:,ns-1) - currumnc(:,ns-2)
738  currvmnc(:,ns) = 2*currvmnc(:,ns-1) - currvmnc(:,ns-2)
739  currumnc = currumnc/mu0; currvmnc = currvmnc/mu0
740 
741  IF (.NOT.lasym) RETURN
742 
743  ALLOCATE (currumns(mnmax_nyq,ns), currvmns(mnmax_nyq,ns), &
744  & stat=ierror)
745 
746  DO js = 2, ns-1
747  WHERE (mod(int(xm_nyq),2) .EQ. 1)
748  t1 = 0.5_dp*(shalf(js+1)*bsubsmnc(:,js+1) &
749  & + shalf(js) *bsubsmnc(:,js)) / sfull(js)
750  bu0 = bsubumns(:,js )/shalf(js+1)
751  bu1 = bsubumns(:,js+1)/shalf(js+1)
752  t2 = ohs*(bu1-bu0)*sfull(js) + 0.25_dp*(bu0+bu1)/sfull(js)
753  bv0 = bsubvmns(:,js )/shalf(js)
754  bv1 = bsubvmns(:,js+1)/shalf(js+1)
755  t3 = ohs*(bv1-bv0)*sfull(js)+0.25_dp*(bv0+bv1)/sfull(js)
756  ELSEWHERE
757  t1 = 0.5_dp*(bsubsmnc(:,js+1) + bsubsmnc(:,js))
758  t2 = ohs*(bsubumns(:,js+1)-bsubumns(:,js))
759  t3 = ohs*(bsubvmns(:,js+1)-bsubvmns(:,js))
760  END WHERE
761  currumns(:,js) = xn_nyq(:)*t1 - t3
762  currvmns(:,js) = xm_nyq(:)*t1 + t2
763  END DO
764 
765  WHERE (xm_nyq .LE. 1)
766  currvmns(:,1) = 2*currvmns(:,2) - currvmns(:,3)
767  currumns(:,1) = 2*currumns(:,2) - currumns(:,3)
768  ELSEWHERE
769  currvmns(:,1) = 0
770  currumns(:,1) = 0
771  END WHERE
772  currumns(:,ns) = 2*currumns(:,ns-1) - currumns(:,ns-2)
773  currvmns(:,ns) = 2*currvmns(:,ns-1) - currvmns(:,ns-2)
774  currumns = currumns/mu0; currvmns = currvmns/mu0
775 
776  END SUBROUTINE compute_currents

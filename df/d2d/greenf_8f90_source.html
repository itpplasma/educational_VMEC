9SUBROUTINE greenf(delgr, delgrp, ip)
10 USE vacmod
11 use dbgout
12 IMPLICIT NONE
13
14 INTEGER, INTENT(in) :: ip
15 REAL(rprec), DIMENSION(nuv), INTENT(out) :: delgr, delgrp
16
17 INTEGER, DIMENSION(2) :: ilow, ihigh
18 INTEGER :: ivoff, iskip, iuoff, i, kp, nloop, ivoff0
19 REAL(rprec):: xip, yip, ftemp, htemp ! xper, yper, sxsave, sysave
20
21 REAL(rprec), dimension(nvper) :: xper, yper, sxsave, sysave
22 integer, dimension(nvper,nuv) :: idx_tanu, idx_tanv
23 REAL(rprec), dimension(nvper,nuv) :: all_tanu, all_tanv
24

26 ! ON ENTRANCE, IP IS THE INDEX OF THE PRIMED MESH POINT (lies in 1st field period)
27 !
28 ! ON EXIT:
29 ! DELGR IS THE DIFFERENCE OF "GREEN'S FUNCTION" AND ANALYTIC APPROXIMATION, SUMMED OVER ALL FIELD PERIODS
30 ! DELGRP IS THE DIFFERENCE OF DERIVATIVE OF "GREEN'S FUNCTION" AND ANALYTIC APPROXIMATION.
31 !
32 ! BOTH THESE QUANTITIES ARE COMPUTED FOR ALL UNPRIMED U,V POINTS IN ONE FIELD PERIOD,
33 ! FOR THIS FIXED PRIMED POINT (IP).
34
35 ! COMPUTE OFFSETS FOR U,V ANGLE DIFFERENCES AND CONSTANTS
36
37 ! first round: up before singularity
38 ilow(1) = 1
39 ihigh(1) = ip - 1
40
41 ! second round: from after singularity onwards
42 ilow(2) = ip + 1
43 ihigh(2) = nuv
44
45 ivoff0 = nuv - (ip - 1)
46
47 iskip = (ip - 1)/nv ! implicit floor by conversion to integer
48 iuoff = nuv - nv*iskip
49
50 ! x == r*COS(ip), in 1st field period
51 xip = rcosuv(ip)
52
53 ! y == r*SIN(ip), in 1st field period
54 yip = rsinuv(ip)
55
56 delgr = 0
57 delgrp = 0
58
59 ! COMPUTE FIELD-PERIOD INVARIANT VECTORS
60 ! NOTE: |x - x'|**2 = gsave - 2*[x*x' + y*y']
61 DO i = 1, nuv
62 gsave(i) = rzb2(ip) + rzb2(i) - 2*z1b(ip)*z1b(i)
63 dsave(i) = drv(ip) + z1b(i)*snz(ip)
64 END DO
65
66 ! SUM OVER FIELD-PERIODS (NVPER=NFPER) OR INTEGRATE OVER NV (NVPER=64) IF NV == 1
67 !
68 ! NOTE THE SURFACE NORMAL SNORM == Xu cross Xv = NP*[SNR, SNV, SNZ]
69 ! IS PERIODIC ON EACH FIELD PERIOD: NOTE THE LOOP OVER KP IS A REDUCTION ON delgr, delgrp
70 DO kp = 1, nvper
71 ! add in offset due to toroidal module
72 ivoff = ivoff0 + 2*nu*(kp-1)
73
74 ! x(ip) in field period kp
75 xper(kp) = xip*cosper(kp) - yip*sinper(kp)
76
77 ! y(ip) in field period kp
78 yper(kp) = yip*cosper(kp) + xip*sinper(kp)
79
80 sxsave(kp) = (snr(ip)*xper(kp) - snv(ip)*yper(kp))/r1b(ip)
81 sysave(kp) = (snr(ip)*yper(kp) + snv(ip)*xper(kp))/r1b(ip)
82
83 IF (kp.EQ.1 .OR. nv.EQ.1) THEN
84 ! Tokamak: always
85 ! Stellarator: first toroidal module
86!
87 ! INITIALIZE ANALYTIC APPROXIMATIONS GA1, GA2
88 DO i = 1, nuv
89
90 idx_tanu(kp, i) = i+iuoff
91 idx_tanv(kp, i) = i+ivoff
92 all_tanu(kp, i) = tanu(i+iuoff)
93 all_tanv(kp, i) = tanv(i+ivoff)
94
95 ga1(kp,i) = tanu(i+iuoff)*( guu_b(ip)*tanu(i+iuoff) + guv_b(ip)*tanv(i+ivoff)) &
96 + gvv_b(ip)*tanv(i+ivoff)*tanv(i+ivoff)
97 ga2(kp,i) = tanu(i+iuoff)*( auu(ip)*tanu(i+iuoff) + auv(ip)*tanv(i+ivoff)) &
98 + avv(ip)*tanv(i+ivoff)*tanv(i+ivoff)
99 END DO
100
101 DO nloop = 1, 2
102 IF (kp.GT.1 .AND. nloop.EQ.2) then
103 ! Tokamak (kp>1): only need to skip exactly singular point if in same module
104 ! --> first round already goes to nuv, since ihigh(1) was updated below
105 cycle
106 end if
107
108 ! loop over grid points; ilow, ihigh used to skip point at which exact singularity occurs: ip==i
109 DO i = ilow(nloop), ihigh(nloop)
110 ga2(kp,i) = ga2(kp,i)/ga1(kp,i)
111 ga1(kp,i) = one/sqrt(ga1(kp,i))
112 ftemp = one/(gsave(i) - 2*(xper(kp)*rcosuv(i) + yper(kp)*rsinuv(i)))
113 htemp = sqrt(ftemp)
114 delgrp(i) = delgrp(i) + ftemp*htemp*(rcosuv(i)*sxsave(kp) + rsinuv(i)*sysave(kp) + dsave(i)) - ga2(kp,i)*ga1(kp,i)
115 delgr(i) = delgr(i) + htemp - ga1(kp,i)
116 END DO
117 END DO
118
119 ! update the upper bound of the first loop
120 ! --> in Tokamak case, skip exact singularity only if in first toroidal "module"
121 ihigh(1) = nuv
122
123 ELSE
124 ! Tokamak: never
125 ! Stellarator: all toroidal modules after first one
126
127 DO i = 1,nuv
128 ftemp = one/(gsave(i) - 2*(xper(kp)*rcosuv(i) + yper(kp)*rsinuv(i)))
129 htemp = sqrt(ftemp)
130 delgrp(i) = delgrp(i) + ftemp*htemp*(rcosuv(i)*sxsave(kp) + rsinuv(i)*sysave(kp) + dsave(i))
131 delgr(i) = delgr(i) + htemp
132 END DO
133 ENDIF
134 END DO
135
136 IF (nv.EQ.1) THEN
137 ! Tokamak: toroidal summation is actually an integration
138 ! --> need to divide by step length!
139 delgrp = delgrp/nvper
140 delgr = delgr /nvper
141 END IF
142

144!
145! call add_real_1d("xper", nvper, xper)
146! call add_real_1d("yper", nvper, yper)
147! call add_real_1d("sxsave", nvper, sxsave)
148! call add_real_1d("sysave", nvper, sysave)
149!
150! call add_int_2d("idx_tanu", nvper, nuv, idx_tanu)
151! call add_int_2d("idx_tanv", nvper, nuv, idx_tanv)
152! call add_real_2d("all_tanu", nvper, nuv, all_tanu)
153! call add_real_2d("all_tanv", nvper, nuv, all_tanv)
154!
155! call close_dbg_out()
156! end if

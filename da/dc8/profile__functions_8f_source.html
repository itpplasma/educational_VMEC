11 FUNCTION pcurr (xx)
12! Function to compute the current profile
13
14! Variables declared in vmec_input:
15! ac array (0:20) of coefficients
16! ac_aux_s Auxiliary array s, s-values used for splines
17! ac_aux_f Auxiliary array f, function-values used for splines
18! bloat used to expand the current profile
19! pcurr_type character, specifies the parameterization of the profile
20! | - X for parametrization of I-prime(s), _ for I(s)
21! gauss_trunc X Truncated Gaussian - I-prime
22! two_power X Two powers - ac(0) * (1 - s ** ac(1)) ** ac(2)
23! two_power_gs X Two powers with gaussian peaks -
24! ac(0) * ((1 - s ** ac(1)) ** ac(2))*(1 + Sum[ac(i)*Exp(-(s - ac(i+1))/ac(i+2)) ** 2])
25! sum_atan _ sum of arctangents
26! power_series_I _ Power series for I(s) (NOT default)
27! Akima_spline_Ip X Akima spline for I-prime(s)
28! Akima_spline_I _ Akima spline for I(s)
29! cubic_spline_Ip X cubic spline for I-prime(s)
30! cubic_spline_I _ cubic spline for I(s)
31! pedestal _ Pedestal profile
32! rational _ Rational function (ratio of polynomials)
33! line_segment_Ip X Line segments for I-prime(s)
34! line_segment_I _ Line segments for I(s)
35! power_series X Power Series for I-prime(s) (Default)
36
37! Local Variables
38! i integer counter
39! ioff offset for ac array. Should be zero
40! iflag error flag for spline call
41! xx real argument
42! x constrained to be between 0 and 1
43! xp variable for Gauss_legendre quadrature
44! pcurr_type_lc character, pcurr_type -> lower case
45! gli index for Gauss-Legendre quadrature loop
46! gln order of Gauss-Legendre quadrature
47! glx array of abscissa values for Gauss-Legendre quadrature
48! glw array of wieghts for Gauss-Legendre quadrature
49
50! Note that the profile that is parameterized is often I-prime, whereas
51! I(x) (= Integral_from_0_to_x I-prime(s) ds) is the function that pcurr
52! returns. For the default case of a power series, the integral can be
53! computed analytically. For other cases, a numerical quadrature is done,
54! using a 10-point Gauss-Legendre quadrature.
55 USE stel_kinds
56 USE stel_constants, ONLY: zero, one, pi
57 USE vmec_input, ONLY: ac, bloat, pcurr_type, ac_aux_s, ac_aux_f
58 USE line_segment
59 USE functions
60 IMPLICIT NONE
61! ac assumed to be dimensioned (0:n), with n >= 20
62!-----------------------------------------------
63 INTEGER :: i, ioff, iflag
64 REAL(rprec) :: xx, pcurr, x, xp, temp_num, temp_denom
65 CHARACTER(len=20) :: pcurr_type_lc
66
67 ! knots and weights for 10-point Gauss-Legendre quadrature
68 INTEGER, PARAMETER :: gln = 10
69 INTEGER :: gli
70 REAL(rprec), DIMENSION(gln), PARAMETER :: glx = (/ &
71 & 0.01304673574141414, 0.06746831665550774, 0.1602952158504878, &
72 & 0.2833023029353764, 0.4255628305091844, 0.5744371694908156, &
73 & 0.7166976970646236, 0.8397047841495122, 0.9325316833444923, &
74 & 0.9869532642585859 /)
75 REAL(rprec), DIMENSION(gln), PARAMETER :: glw = (/ &
76 & 0.03333567215434407, 0.0747256745752903, 0.1095431812579910, &
77 & 0.1346333596549982, 0.1477621123573764, 0.1477621123573764, &
78 & 0.1346333596549982, 0.1095431812579910, 0.0747256745752903, &
79 & 0.03333567215434407 /)
80
81 REAL(rprec) :: g1,g2,g3,g4,a8,a12
82
83!-----------------------------------------------
84!
85! NOTE: AC COEFFICIENTS OBTAINED IN THREED1 FILE
86! BY MATCHING TO <JTOR> * dV/dPHI ~ SUM[x^(i+1) * ac(i)/(i+1)], i=0...UBOUND(ac)
87!
88! Start of executable code
89
90 ! limit radial location to 1, even if bloat factor expands profile further
91 x = min(abs(xx * bloat), one)
92 ioff = lbound(ac,1) ! Expected to be zero.
93
94 pcurr = 0
95
96! Convert to Lower Case, to avoid typo issues
97 pcurr_type_lc = pcurr_type
98 CALL tolower(pcurr_type_lc)
99 SELECT CASE(trim(pcurr_type_lc))
100
101 CASE ('gauss_trunc')
102! Truncated Gaussian
103! I-prime(s) = ac(0) * (exp(-(s/ac(1))**2) - exp(-(1/ac(1))**2)
104! Gauss-Legendre Quadrature to get I(s)
105 DO gli = 1,gln
106 xp = x * glx(gli)
107 pcurr = pcurr + glw(gli) * ac(0) * (exp(-(xp / ac(1)) ** 2) &
108 & - exp(-( 1 / ac(1)) ** 2))
109 END DO
110 pcurr = pcurr * x ! correct for x interval
111
112 CASE ('two_power')
113! Two power profile
114! I-prime(s) = [1 - s**ac(0)]**ac(1) !! Old as of 2010-05-26
115! I-prime(s) = ac(0) * [1 - s**ac(1)]**ac(2) !! Current as of 2010-05-26
116! Gauss-Legendre Quadrature to get I(s)
117 !DO gli = 1,gln
118 ! xp = x * glx(gli)
119 ! pcurr = pcurr + glw(gli) * two_power(xp, ac)
120 !END DO
121 !pcurr = pcurr * x ! correct for x interval
122 DO gli = 1,gln
123 xp = x * glx(gli)
124 pcurr = pcurr + glw(gli) * ac(0) * ((1-xp**ac(1))**ac(2))
125 END DO
126 pcurr = pcurr * x ! correct for x interval
127
128! CASE ('two_power_gs')
129!! Two power with Gaussian peak profile
130!! I-prime(s) = ac(0) * [1 - s**ac(1)]**ac(2) * (1 + Sum[ac(i)*Exp(-(s - ac(i+1))/ac(i+2)) ** 2])
131!! Gauss-Legendre Quadrature to get I(s)
132! DO gli = 1,gln
133! xp = x * glx(gli)
134! pcurr = pcurr + glw(gli) * two_power_gs(xp, ac)
135! END DO
136! pcurr = pcurr * x ! correct for x interval
137
138 CASE('sum_atan')
139! I(s) - not I-prime(s)
140! pcurr = ac(0) + &
141! & ac(1) * (2/pi) * atan(ac(2)*x**ac(3)/(1-x)**ac(4)) + &
142! & ac(5) * (2/pi) * atan(ac(6)*x**ac(7)/(1-x)**ac(8)) + &
143! & ac(9) * (2/pi) * atan(ac(10)*x**ac(11)/(1-x)**ac(12)) + &
144! & ac(13) * (2/pi) * atan(ac(14)*x**ac(15)/(1-x)**ac(16)) + &
145! & ac(17) * (2/pi) * atan(ac(18)*x**ac(19)/(1-x)**ac(20))
146 IF (x .ge. one) THEN
147 pcurr = ac(0) + ac(1) + ac(5) + ac(9) + ac(13) + ac(17)
148 ELSE
149 pcurr = ac(0) + &
150 & (2/pi) * (ac(1) * atan(ac(2)*x**ac(3)/(1-x)**ac(4)) + &
151 & ac(5) * atan(ac(6)*x**ac(7)/(1-x)**ac(8)) + &
152 & ac(9) * atan(ac(10)*x**ac(11)/(1-x)**ac(12)) + &
153 & ac(13) * atan(ac(14)*x**ac(15)/(1-x)**ac(16)) + &
154 & ac(17) * atan(ac(18)*x**ac(19)/(1-x)**ac(20)))
155 ENDIF
156
157 CASE('power_series_I','power_series_i') ! former ipcurr=3
158! I(s) - not I-prime(s)
159! I(s) = Sum(i,0,-)[ac(i) * s ** (i + 1)]
160! Note that coefficient interpretation here is different from other
161! polynomial parameterizations, where the terms are a_(i) * s ** i
162! Note that I(0) = 0.
163 DO i = ubound(ac,1), ioff, -1
164 pcurr = (pcurr + ac(i))*x
165 END DO
166
167 CASE('Akima_spline_I','akima_spline_i') ! former ipcurr=11
168! I(s) with Akima splines
169! Akima-spline for current profile
170! i = minloc(scugrid(2:ndatafmax),dim=1) ! this is where zeros are again
171 i = minloc(ac_aux_s(2:),dim=1) ! this is where zeros are again
172 IF (i < 4) stop 'pcurr: check s-grid for curr-grid values!'
173! call spline_akima(x,pcurr,scugrid,cugrid,i,iflag)
174 CALL spline_akima(x,pcurr,ac_aux_s,ac_aux_f,i,iflag)
175 IF(iflag < 0) &
176 & stop 'pcurr: outside value from spline_akima requested'
177
178 CASE('Akima_spline_Ip','akima_spline_ip') ! former ipcurr=12
179! I(s) from I-prime(s)-values
180! uses an integrated Akima-spline to deliver the values of the toroidal current
181! i = minloc(scdgrid(2:ndatafmax),dim=1) ! this is where zeros are again
182 i = minloc(ac_aux_s(2:),dim=1) ! this is where zeros are again
183 IF(i < 4)stop 'pcurr: check s-grid for curr.dens.-grid values!'
184! call spline_akima_int(x,pcurr,scdgrid,cdgrid,i,iflag)
185 CALL spline_akima_int(x,pcurr,ac_aux_s,ac_aux_f,i,iflag)
186 IF (iflag < 0) THEN
187 WRITE(6,*)"x = ", x
188 stop 'pcurr: outside value from spline_akima_int requested'
189 ENDIF
190! Note that spline_akima_int integrates with ac_aux_s(1) as the lower limit.
191! Assumption here is that lower limit is s = 0. Check this assumption.
192 IF (abs(ac_aux_s(1)) .gt. 0.0001) THEN
193 WRITE(*,*) ' Error in profile_functions, Akima_spline_Ip'
194 WRITE(*,*) .gt.' ABS(ac_aux_s(1)) 0.0001'
195 WRITE(*,*) ' Fix This. Stopping'
196 stop
197 ENDIF
198
199 CASE('cubic_spline_I','cubic_spline_i') ! former ipcurr=13
200! I(s) with Cubic splines
201! Cubic-spline for current profile
202! i = minloc(scugrid(2:ndatafmax),dim=1) ! this is where zeros are again
203 i = minloc(ac_aux_s(2:),dim=1) ! this is where zeros are again
204 IF(i < 4) stop 'pcurr: check s-grid for curr-grid values!'
205! call spline_cubic(x,pcurr,scugrid,cugrid,i,iflag)
206 CALL spline_cubic(x,pcurr,ac_aux_s,ac_aux_f,i,iflag)
207 IF (iflag < 0) THEN
208 IF(iflag == -1) THEN
209 stop'pcurr: outside value from spline_cubic requested'
210 ELSEIF(iflag == -2) THEN
211 stop 'pcurr: identical scugrid-values in spline_cubic'
212 ELSE
213 stop 'pcurr: unknown error from spline_cubic'
214 ENDIF
215 ENDIF
216
217 CASE('cubic_spline_Ip','cubic_spline_ip') ! former ipcurr=14
218! I(s) from I-prime(s)-values
219! uses an integrated Cubic-spline to deliver the values of the toroidal current
220! i = minloc(scdgrid(2:ndatafmax),dim=1) ! this is where zeros are again
221 i = minloc(ac_aux_s(2:),dim=1) ! this is where zeros are again
222 IF(i < 4) stop 'pcurr: check s-grid for curr.dens.-grid values!'
223! call spline_cubic_int(x,pcurr,scdgrid,cdgrid,i,iflag)
224 CALL spline_cubic_int(x,pcurr,ac_aux_s,ac_aux_f,i,iflag)
225 IF (iflag < 0)THEN
226 IF (iflag == -1) THEN
227 stop'pcurr: outside value from spline_cubic_int requested'
228 ELSEIF (iflag == -2) THEN
229 stop 'pcurr: identical scdgrid-values in spline_cubic_int'
230 ELSE
231 stop 'pcurr: unknown error from spline_cubic_int'
232 ENDIF
233 ENDIF
234! Note that spline_cubic_int integrates with ac_aux_s(1) as the lower limit.
235! Assumption here is that lower limit is s = 0. Check this assumption.
236 IF (abs(ac_aux_s(1)) .gt. 0.0001) THEN
237 WRITE(*,*) ' Error in profile_functions, cubic_spline_Ip'
238 WRITE(*,*) .gt.' ABS(ac_aux_s(1)) 0.0001'
239 WRITE(*,*) ' Fix This. Stopping'
240 stop
241 ENDIF
242
243 CASE ('pedestal')
244! Parameterization for I(s)
245! Added by SPH 2010-05-26
246 DO i = 7, ioff, -1
247 pcurr = x*pcurr + ac(i)/(i-ioff+1)
248 END DO
249 pcurr = x*pcurr
250
251 i=8
252 IF (ac(i+3) .le. 0._dp ) THEN
253 ac(i:i+4) = 0
254 ac(i+3) = 1.0e30_dp
255 ELSE
256 ac(i+4) = 1.0_dp/
257 1 (tanh(2*ac(i+2)/ac(i+3))-tanh(2*(ac(i+2)-1)/ac(i+3)))
258 END IF
259
260 a8 =max(ac(i+8),0.01_dp)
261 a12=max(ac(i+12),0.01_dp)
262
263 g1= (x-ac(i+7))/a8
264 g3= (-ac(i+7))/a8
265 g2= (x-ac(i+11))/a12
266 g4= (-ac(i+11))/a12
267 pcurr = pcurr + ac(i+4) * ac(i+0) *
268 1 ( tanh( 2*ac(i+2)/ac(i+3) )
269 2 -tanh( 2*(ac(i+2)-sqrt(x))/ac(i+3) ) )
270 3 +ac(i+5)*( tanh(g1) - tanh(g3) )
271 4 +ac(i+9)*( tanh(g2) - tanh(g4) )
272
273
274 CASE('rational') !
275! I(s) - not I-prime(s). No need for Gaussian quadrature.
276! Ratio of polynomials.
277! Numerator coefficients: ac(LBOUND) - ac(9)
278! Denominator coefficients: ac(10) - ac(UBOUND(ac,1))
279 temp_num = zero
280 temp_denom = zero
281 DO i = 9, ioff, -1
282 temp_num = x * temp_num + ac(i)
283 END DO
284 DO i = ubound(ac,1), 10, -1
285 temp_denom = x * temp_denom + ac(i)
286 END DO
287 IF (temp_denom .ne. zero) THEN
288 pcurr = temp_num / temp_denom
289 ELSE
290 pcurr = huge(pcurr)
291 ENDIF
292
293 CASE('line_segment_Ip', 'line_segment_ip')
294!! I(s) from I-prime(s)-values. Integrated line segments to determine the plasma
295!! current.
296 i = minloc(ac_aux_s(2:),dim=1)
297 CALL line_seg_int(x,pcurr,ac_aux_s,ac_aux_f,i)
298
299 CASE('line_segment_I', 'line_segment_i')
300!! I(s) values. Linearly interpolated line segments to determine the plasma
301!! current.
302 i = minloc(ac_aux_s(2:),dim=1)
303 CALL line_seg(x,pcurr,ac_aux_s,ac_aux_f,i)
304
305 CASE DEFAULT
306! Power series
307! I-prime(s) = Sum(i,0,-)[ac(i) * s ** i]
308! Analytic integration to get I(s)
309 DO i = ubound(ac,1), ioff, -1
310 pcurr = x*pcurr + ac(i)/(i-ioff+1)
311 END DO
312 IF (trim(pcurr_type_lc) .ne. 'power_series') THEN
313 WRITE(*,*) 'Unrecognized pcurr_type:', pcurr_type
314 WRITE(*,*) ' *** CHECK YOUR INPUT ***'
315 WRITE(*,*) 'Changing pcurr_type from ''', &
316 & trim(pcurr_type), '''to ''power_series''.'
317 pcurr_type = 'power_series'
318 END IF
319 pcurr = x*pcurr
320 END SELECT
321
322 END FUNCTION pcurr

326 FUNCTION piota (x)
327! Function to compute the iota/q profile.
328
329! Variables declared in vmec_input:
330! ai array (0:20) of coefficients
331! ai_aux_s Auxiliary array s, s-values used for splines
332! ai_aux_f Auxiliary array f, function-values used for splines
333! piota_type character, specifies the parameterization of the profile
334! sum_atan Sum of atan functions plus offset.
335! Akima_spline Akima spline
336! cubic_spline cubic spline
337! rational rational (ratio of polynomials)
338! nice_quadratic quadratic with rerranged coefficients.
339! line_segment Linearly interpolated line segments
340! power_series Power Series (Default)
341
342! Local Variables
343! i integer counter
344! iflag error flag for spline call
345! x real argument
346! x constrained to be between 0 and 1
347! piota_type_lc character, piota_type -> lower case
348
349 USE stel_kinds
350 USE stel_constants, ONLY: zero, one, pi
351 USE vmec_input, ONLY: ai, piota_type, ai_aux_s, ai_aux_f
352 USE line_segment
353 IMPLICIT NONE
354! ai assumed to be dimensioned (0:n), with n >= 20
355!-----------------------------------------------
356 INTEGER :: i, iflag, ioff
357 REAL(rprec) :: x, piota, temp_num, temp_denom
358 CHARACTER(len=20) :: piota_type_lc
359!-----------------------------------------------
360 piota = 0
361 ioff = lbound(ai,1) ! Expected to be zero.
362
363! Convert to Lower Case, to avoid typo issues
364 piota_type_lc = piota_type
365 CALL tolower(piota_type_lc)
366 SELECT CASE(trim(piota_type_lc))
367
368 CASE('sum_atan')
369! Sum atan functions mapped to [0:1], with an ai(0) offset
370! piota = ai(0) + &
371! & ai(1) * (2/pi) * atan(ai(2)*x**ai(3)/(1-x)**ai(4)) + &
372! & ai(5) * (2/pi) * atan(ai(6)*x**ai(7)/(1-x)**ai(8)) + &
373! & ai(9) * (2/pi) * atan(ai(10)*x**ai(11)/(1-x)**ai(12)) + &
374! & ai(13) * (2/pi) * atan(ai(14)*x**ai(15)/(1-x)**ai(16)) + &
375! & ai(17) * (2/pi) * atan(ai(18)*x**ai(19)/(1-x)**ai(20))
376 IF (x .ge. one) THEN
377 piota = ai(0) + ai(1) + ai(5) + ai(9) + ai(13) + ai(17)
378 ELSE
379 piota = ai(0) + &
380 & (2/pi) * (ai(1) * atan(ai(2)*x**ai(3)/(1-x)**ai(4)) + &
381 & ai(5) * atan(ai(6)*x**ai(7)/(1-x)**ai(8)) + &
382 & ai(9) * atan(ai(10)*x**ai(11)/(1-x)**ai(12)) + &
383 & ai(13) * atan(ai(14)*x**ai(15)/(1-x)**ai(16)) + &
384 & ai(17) * atan(ai(18)*x**ai(19)/(1-x)**ai(20)))
385 ENDIF
386
387 CASE('Akima_spline','akima_spline') ! former ipiota=11
388! Akima-spline (iogrid + siogrid)
389! i = minloc(siogrid(2:ndatafmax),dim=1) ! this is where zeros are again
390 i = minloc(ai_aux_s(2:),dim=1) ! this is where zeros are again
391 if(i < 4) stop 'piota: check s-grid for iota-grid values!'
392! call spline_akima(x,piota,siogrid,iogrid,i,iflag)
393 call spline_akima(x,piota,ai_aux_s,ai_aux_f,i,iflag)
394 if(iflag < 0) &
395 & stop'piota: outside value from spline_akima requested'
396
397 CASE('cubic_spline') ! former ipiota=13
398! cubic-spline (iogrid + siogrid)
399! i = minloc(siogrid(2:ndatafmax),dim=1) ! this is where zeros are again
400 i = minloc(ai_aux_s(2:),dim=1) ! this is where zeros are again
401 if(i < 4) stop 'piota: check s-grid for iota-grid values!'
402! call spline_cubic(x,piota,siogrid,iogrid,i,iflag)
403 call spline_cubic(x,piota,ai_aux_s,ai_aux_f,i,iflag)
404 if(iflag < 0)then
405 if(iflag == -1) then
406 stop'piota: outside value from spline_cubic requested'
407 elseif(iflag == -2) then
408 stop'piota: identical siogrid-values in spline_cubic'
409 else
410 stop 'piota: unknown error from spline_cubic'
411 endif
412 endif
413
414 CASE('rational') !
415! Ratio of polynomials.
416! Numerator coefficients: ai(LBOUND) - ai(9)
417! Denominator coefficients: ai(10) - ai(UBOUND)
418 temp_num = zero
419 temp_denom = zero
420 DO i = 9, ioff, -1
421 temp_num = x * temp_num + ai(i)
422 END DO
423 DO i = ubound(ai,1), 10, -1
424 temp_denom = x * temp_denom + ai(i)
425 END DO
426 IF (temp_denom .ne. zero) THEN
427 piota = temp_num / temp_denom
428 ELSE
429 piota = huge(piota)
430 ENDIF
431
432 CASE('nice_quadratic') !
433! Quadratic with slightly rearranged coefficients.
434! iota(s) = a0(1-s) + a1 s + 4 a2 s (1 - s)
435! a0 is the iota value at s = 0
436! a1 is the iota value at s = 1
437! a2 is the shift of iota(1/2) from the straight line value
438! (Thus, a2 = 0 gives a linear iota profile.
439 piota = ai(0) * (one - x) + ai(1) * x + 4 * ai(2) * &
440 & x * (one - x)
441
442 CASE('line_segment')
443!! Linearly interpolated line segments to determine the iotabar profile.
444 i = minloc(ai_aux_s(2:),dim=1)
445 CALL line_seg(x,piota,ai_aux_s,ai_aux_f,i)
446
447 CASE default
448! Power Series is the default
449 DO i = ubound(ai,1), lbound(ai,1), -1
450 piota = x*piota + ai(i)
451 END DO
452 IF (trim(piota_type_lc) .ne. 'power_series') THEN
453 WRITE(*,*) 'Unrecognized piota_type:', piota_type
454 WRITE(*,*) ' *** CHECK YOUR INPUT ***'
455 WRITE(*,*) 'Changing piota_type from ''', &
456 & trim(piota_type), '''to ''power_series''.'
457 piota_type = 'power_series'
458 END IF
459 END SELECT
460
461 END FUNCTION piota

465 FUNCTION pmass (xx)
466! Function to compute the mass (pressure) profile
467
468! Variables declared in vmec_input:
469! am array (0:20) of coefficients
470! am_aux_s Auxiliary array s, s-values used for splines
471! am_aux_f Auxiliary array f, function-values used for splines
472! bloat used to expand the profile
473! pmass_type character, specifies the parameterization of the profile
474! gauss_trunc Truncated Gaussian
475! two_power am(0) * [1 - s**am(1)]**am(2)
476! two_power_gs am(0) * [1 - s**am(1)]**am(2)*
477! (1 + Sum[am(i)*Exp(-((x - am(i+1))/am(i+2))**2)])
478! two_Lorentz two Lorentz-type functions, mapped to [0:1]
479! Akima_spline Akima-spline (magrid[-> am_aux_f] + smagrid[-> am_aux_s])
480! cubic_spline Cubic-spline (magrid[-> am_aux_f] + smagrid[-> am_aux_s])
481! pedestal Pedestal profile
482! rational rational (ratio of polynomials)
483! line_segment Linearly interpolated line segments
484! power_series Power Series (Default)
485
486! Local Variables
487! i integer counter
488! iflag error flag for spline call
489! xx real argument
490! x constrained to be between 0 and 1
491! pmass_type_lc character, pmass_type -> lower case
492
493 USE stel_kinds
494 USE stel_constants, ONLY: zero, one
495 USE vmec_input, ONLY: am, bloat, pres_scale, pmass_type, &
496 & am_aux_s, am_aux_f
497! am is assumed to be dimensioned starting at zero.
498 USE vparams, ONLY: mu0
499 USE line_segment
500 USE functions
501 IMPLICIT NONE
502!-----------------------------------------------
503 INTEGER :: i, iflag, ioff
504 REAL(rprec) :: xx, pmass, x, temp_num, temp_denom
505 CHARACTER(len=20) :: pmass_type_lc
506!-----------------------------------------------
507! NOTE: On entry, am is in pascals. pmass internal units are mu0*pascals (B**2 units)

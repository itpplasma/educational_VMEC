321  FUNCTION piota (x)
322 ! Function to compute the iota/q profile.
323 
324 ! Variables declared in vmec_input:
325 ! ai array (0:20) of coefficients
326 ! ai_aux_s Auxiliary array s, s-values used for splines
327 ! ai_aux_f Auxiliary array f, function-values used for splines
328 ! piota_type character, specifies the parameterization of the profile
329 ! sum_atan Sum of atan functions plus offset.
330 ! Akima_spline Akima spline
331 ! cubic_spline cubic spline
332 ! rational rational (ratio of polynomials)
333 ! nice_quadratic quadratic with rerranged coefficients.
334 ! line_segment Linearly interpolated line segments
335 ! power_series Power Series (Default)
336 ! True - ai parameterization specifies q-profile (= 1 / iota)
337 ! False - ai parameterization is for iota-profile
338 
339 ! Local Variables
340 ! i integer counter
341 ! iflag error flag for spline call
342 ! x real argument
343 ! x constrained to be between 0 and 1
344 ! piota_type_lc character, piota_type -> lower case
345 
346  USE stel_kinds
347  USE stel_constants, ONLY: zero, one, pi
348  USE vmec_input, ONLY: ai, piota_type, ai_aux_s, ai_aux_f
349  USE line_segment
350  IMPLICIT NONE
351 ! ai assumed to be dimensioned (0:n), with n >= 20
352 !-----------------------------------------------
353  INTEGER :: i, iflag, ioff
354  REAL(rprec) :: x, piota, temp_num, temp_denom
355  CHARACTER(len=20) :: piota_type_lc
356 !-----------------------------------------------
357  piota = 0
358  ioff = lbound(ai,1) ! Expected to be zero.
359 
360 ! Convert to Lower Case, to avoid typo issues
361  piota_type_lc = piota_type
362  CALL tolower(piota_type_lc)
363  SELECT CASE(trim(piota_type_lc))
364 
365  CASE('sum_atan')
366 ! Sum atan functions mapped to [0:1], with an ai(0) offset
367 ! piota = ai(0) + &
368 ! & ai(1) * (2/pi) * atan(ai(2)*x**ai(3)/(1-x)**ai(4)) + &
369 ! & ai(5) * (2/pi) * atan(ai(6)*x**ai(7)/(1-x)**ai(8)) + &
370 ! & ai(9) * (2/pi) * atan(ai(10)*x**ai(11)/(1-x)**ai(12)) + &
371 ! & ai(13) * (2/pi) * atan(ai(14)*x**ai(15)/(1-x)**ai(16)) + &
372 ! & ai(17) * (2/pi) * atan(ai(18)*x**ai(19)/(1-x)**ai(20))
373  IF (x .ge. one) THEN
374  piota = ai(0) + ai(1) + ai(5) + ai(9) + ai(13) + ai(17)
375  ELSE
376  piota = ai(0) + &
377  & (2/pi) * (ai(1) * atan(ai(2)*x**ai(3)/(1-x)**ai(4)) + &
378  & ai(5) * atan(ai(6)*x**ai(7)/(1-x)**ai(8)) + &
379  & ai(9) * atan(ai(10)*x**ai(11)/(1-x)**ai(12)) + &
380  & ai(13) * atan(ai(14)*x**ai(15)/(1-x)**ai(16)) + &
381  & ai(17) * atan(ai(18)*x**ai(19)/(1-x)**ai(20)))
382  ENDIF
383 
384  CASE('Akima_spline','akima_spline') ! former ipiota=11
385 ! Akima-spline (iogrid + siogrid)
386 ! i = minloc(siogrid(2:ndatafmax),dim=1) ! this is where zeros are again
387  i = minloc(ai_aux_s(2:),dim=1) ! this is where zeros are again
388  if(i < 4) stop 'piota: check s-grid for iota-grid values!'
389 ! call spline_akima(x,piota,siogrid,iogrid,i,iflag)
390  call spline_akima(x,piota,ai_aux_s,ai_aux_f,i,iflag)
391  if(iflag < 0) &
392  & stop'piota: outside value from spline_akima requested'
393 
394  CASE('cubic_spline') ! former ipiota=13
395 ! cubic-spline (iogrid + siogrid)
396 ! i = minloc(siogrid(2:ndatafmax),dim=1) ! this is where zeros are again
397  i = minloc(ai_aux_s(2:),dim=1) ! this is where zeros are again
398  if(i < 4) stop 'piota: check s-grid for iota-grid values!'
399 ! call spline_cubic(x,piota,siogrid,iogrid,i,iflag)
400  call spline_cubic(x,piota,ai_aux_s,ai_aux_f,i,iflag)
401  if(iflag < 0)then
402  if(iflag == -1) then
403  stop'piota: outside value from spline_cubic requested'
404  elseif(iflag == -2) then
405  stop'piota: identical siogrid-values in spline_cubic'
406  else
407  stop 'piota: unknown error from spline_cubic'
408  endif
409  endif
410 
411  CASE('rational') !
412 ! Ratio of polynomials.
413 ! Numerator coefficients: ai(LBOUND) - ai(9)
414 ! Denominator coefficients: ai(10) - ai(UBOUND)
415  temp_num = zero
416  temp_denom = zero
417  DO i = 9, ioff, -1
418  temp_num = x * temp_num + ai(i)
419  END DO
420  DO i = ubound(ai,1), 10, -1
421  temp_denom = x * temp_denom + ai(i)
422  END DO
423  IF (temp_denom .ne. zero) THEN
424  piota = temp_num / temp_denom
425  ELSE
426  piota = huge(piota)
427  ENDIF
428 
429  CASE('nice_quadratic') !
430 ! Quadratic with slightly rearranged coefficients.
431 ! iota(s) = a0(1-s) + a1 s + 4 a2 s (1 - s)
432 ! a0 is the iota value at s = 0
433 ! a1 is the iota value at s = 1
434 ! a2 is the shift of iota(1/2) from the straight line value
435 ! (Thus, a2 = 0 gives a linear iota profile.
436  piota = ai(0) * (one - x) + ai(1) * x + 4 * ai(2) * &
437  & x * (one - x)
438 
439  CASE('line_segment')
440 !! Linearly interpolated line segments to determine the iotabar profile.
441  i = minloc(ai_aux_s(2:),dim=1)
442  CALL line_seg(x,piota,ai_aux_s,ai_aux_f,i)
443 
444  CASE default
445 ! Power Series is the default
446  DO i = ubound(ai,1), lbound(ai,1), -1
447  piota = x*piota + ai(i)
448  END DO
449  IF (trim(piota_type_lc) .ne. 'power_series') THEN
450  WRITE(*,*) 'Unrecognized piota_type:', piota_type
451  WRITE(*,*) ' *** CHECK YOUR INPUT ***'
452  WRITE(*,*) 'Changing piota_type from ''', &
453  & trim(piota_type), '''to ''power_series''.'
454  piota_type = 'power_series'
455  END IF
456  END SELECT
457 
458  END FUNCTION piota

464 
465 ! Variables declared in vmec_input:
466 ! am array (0:20) of coefficients
467 ! am_aux_s Auxiliary array s, s-values used for splines
468 ! am_aux_f Auxiliary array f, function-values used for splines
469 ! bloat used to expand the profile

471 ! gauss_trunc Truncated Gaussian
472 ! two_power am(0) * [1 - s**am(1)]**am(2)

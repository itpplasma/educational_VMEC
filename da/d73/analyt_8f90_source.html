2 SUBROUTINE analyt(grpmn, bvec, ivacskip, lasym, m_map, n_map, grpmn_m_map, grpmn_n_map)
3  USE vacmod, vm_grpmn => grpmn
4  IMPLICIT NONE
5 
6  INTEGER, INTENT(in) :: ivacskip
7  REAL(rprec), INTENT(out) :: grpmn(nuv2*mnpd2)
8  REAL(rprec), INTENT(out) :: bvec(mnpd2)
9  real(rprec), intent(out) :: m_map(mnpd2)
10  real(rprec), intent(out) :: n_map(mnpd2)
11  real(rprec), intent(out) :: grpmn_m_map(nuv2*mnpd2)
12  real(rprec), intent(out) :: grpmn_n_map(nuv2*mnpd2)
13  integer, intent(in) :: lasym
14 
15  INTEGER :: l, n, m
16  REAL(rprec) :: fl, fl1, sign1
17 
18  ! ALL EQUATIONS REFER TO THE PAPER BY P. MERKEL (PKM)
19  ! IN J. COMPUT. PHYSICS 66, p83 (1986)
20  !
21  ! IN GOING BETWEEN THE COMPLEX NOTATION OF (PKM) AND OUR REAL FORM,
22  ! NOTE THAT THE INTEGRALS (APPENDIX, PKM) Imn AND Kmn ARE BOTH REAL.
23  ! THUS, THE SIN(mu-nv) INTEGRALS OF THE SINGULAR PIECE (ANALYTIC CONTRIBUTION)
24  ! VANISHES.
25  !
26  ! THE REQUIRED SOURCE-TERM INTEGRALS ARE (Eq.2.16-2.17):
27  !
28  ! BVECS(m,n) = Int< SIN(mu' - nv') han(u',v') >
29  ! BVECC(m,n) = Int< COS(mu' - nv') han(u',v') >
30  !
31  ! Where Int<...> means integration over u (theta) and v (zeta) and
32  ! summation over field periods. These can be written in terms of PKM integrals
33  ! Imn(a,b,c), where a(u,v) = guu (g theta-theta), etc.:
34  !
35  ! BVECS(m,n) = ALP * Int<SIN(mu' - nv') * F * Im,-n(a,b,c)>
36  ! BVECC(m,n) = ALP * Int<COS(mu' - nv') * F * Im,-n(a,b,c)>
37  !
38  ! Here, F = - BNORM(u',v') is defined in Eq.(2.13), and ALP = (2*pi/nfp).
39  !
40  ! Similarly, the analytic part of the matrix A(m,n;m',n') can be written:
41  !
42  ! A(m,n;m',n') = (2*pi/nfp) * Int<SIN(mu' - nv')*SIN(m'u' - n'v')
43  ! [Km,-n](a',b',c';A',B',C')>
44  !
45  ! On EXIT, GRPMN(ip,m,n) = ALP * SIN(ip,m,n) * K[m,-n](ip)
46  !
47  ! COMPUTE ALL QUANTITIES INDEPENDENT OF THE MODE INDICES L,M,N
48  ! NOTE: 2b = guv_b HAS FACTOR OF 2 BUILT IN (see SUBROUTINE SURFACE)
49  !
50  ! ADP(M): a +(-)2b + c
51  ! CMA: c - a
52  ! DELTA: 4*(ac - b**2)
53  ! AZP(M): A +(-)2*B + C
54  ! CMA1: C - A
55  ! R1P(M): Coefficient of l*Tl+(-) in eq (A17)
56  ! R0P(M): Coefficient of l*T(l-1)+(-) in eq (A17)
57  ! RA1P(M):Coefficient of Tl+(-) in eq (A17)
58 
59  adp = guu_b + guv_b + gvv_b
60  adm = guu_b - guv_b + gvv_b
61  cma = gvv_b - guu_b
62  sqrtc = two*sqrt(gvv_b)
63  sqrta = two*sqrt(guu_b)
64  sqad1u = sqrt(adp)
65  sqad2u = sqrt(adm)
66 
67  ! INITIALIZE VECTORS
68  bvec = 0
69  IF (ivacskip .eq. 0) THEN
70  grpmn = 0
71 
72  delt1u = adp*adm - cma*cma
73  azp1u = auu + auv + avv
74  azm1u = auu - auv + avv
75  cma11u = avv - auu
76  r1p = (azp1u*(delt1u - cma*cma)/adp - azm1u*adp + two*cma11u*cma)/delt1u
77  r1m = (azm1u*(delt1u - cma*cma)/adm - azp1u*adm + two*cma11u*cma)/delt1u
78  r0p = (-azp1u*adm*cma/adp - azm1u*cma + two*cma11u*adm)/delt1u
79  r0m = (-azm1u*adp*cma/adm - azp1u*cma + two*cma11u*adp)/delt1u
80  ra1p = azp1u/adp
81  ra1m = azm1u/adm
82  ENDIF
83 
84  ! INITIALIZE T0+ and T0-
85  !
86  ! TLP(M): TL+(-)
87  ! TLP(M)1:T(L-1)+(-)
88  ! TLP(M)2:T(L-2)+(-)
89  tlp1 = 0
90  tlm1 = 0
91  tlp = one/sqad1u*log((sqad1u*sqrtc + adp + cma)/(sqad1u*sqrta - adp + cma))
92  tlm = one/sqad2u*log((sqad2u*sqrtc + adm + cma)/(sqad2u*sqrta - adm + cma))
93  tlpm = tlp + tlm
94 
95  ! BEGIN L-SUM IN EQ (A14) TO COMPUTE Imn (and Kmn) INTEGRALS
96  ! NOTE THAT IN THE LOOP OVER L BELOW: L == |m - n| + 2L_A14
97  ! THUS, L BELOW IS THE INDEX OF THE T+- (S+-)
98  sign1 = 1
99  fl1 = 0
100 
101  lloop: DO l = 0, mf + nf
102  fl = fl1
103 
104  ! COMPUTE SL+ and SL- , Eq (A17)
105  ! SLP(M): SL+(-)
106  IF (ivacskip .eq. 0) THEN
107  slp = (r1p*fl + ra1p)*tlp + r0p*fl*tlp1 - (r1p + r0p)/sqrtc + sign1*(r0p - r1p)/sqrta
108  slm = (r1m*fl + ra1m)*tlm + r0m*fl*tlm1 - (r1m + r0m)/sqrtc + sign1*(r0m - r1m)/sqrta
109  slpm = slp + slm
110 
111  ! here, tlp/m and slp/m are available for the current value of l
112  ! --> save into matrix for debugging
113  all_tlp(:,l) = tlp
114  all_tlm(:,l) = tlm
115  all_slp(:,l) = slp
116  all_slm(:,l) = slm
117  ENDIF
118 
119  ! BEGIN MODE NUMBER (m,n) LOOP
120  DO n = 0, nf
121  DO m = 0, mf
122  ! always call analysum, analysum2 to get bvec fill pattern (m_map, n_map)
123  ! IF (cmns(l,m,n) .eq. zero) CYCLE
124 
125  IF (n.eq.0 .or. m.eq.0) THEN
126  ! 1. n = 0 and m >= 0 OR n > 0 and m = 0
127  CALL analysum (grpmn, bvec, slpm, tlpm, m, n, l, ivacskip, lasym, m_map, n_map, &
128  grpmn_m_map, grpmn_n_map)
129  ELSE
130  ! 2. n>=1 and m>=1
131  CALL analysum2 (grpmn, bvec, m, n, l, ivacskip, lasym, m_map, n_map, &
132  grpmn_m_map, grpmn_n_map)
133  ENDIF
134  END DO
135  END DO
136 
137  ! UPDATE "TL's" (FOR L -> L+1) USING EQ (A15)
138  tlp2 = tlp1
139  tlm2 = tlm1
140  tlp1 = tlp
141  tlm1 = tlm
142 
143  ! next l
144  fl1 = fl1 + 1
145 
146  ! (-1)**l (next l now)
147  sign1 = -sign1
148 
149  tlp = ((sqrtc + sign1*sqrta) - (two*fl1 - one)*cma*tlp1 - fl*adm*tlp2)/(adp*fl1)
150  tlm = ((sqrtc + sign1*sqrta) - (two*fl1 - one)*cma*tlm1 - fl*adp*tlm2)/(adm*fl1)
151  tlpm = tlp + tlm
152  END DO lloop
153 
154 END SUBROUTINE analyt

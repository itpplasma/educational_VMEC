15 USE vacmod, vm_grpmn => grpmn
16 IMPLICIT NONE

18 INTEGER, INTENT(in) :: ivacskip
19 REAL(rprec), INTENT(out) :: grpmn(nuv2*mnpd2)
20 REAL(rprec), INTENT(out) :: bvec(mnpd2)
21 real(rprec), intent(out) :: m_map(mnpd2)
22 real(rprec), intent(out) :: n_map(mnpd2)
23 real(rprec), intent(out) :: grpmn_m_map(nuv2*mnpd2)
24 real(rprec), intent(out) :: grpmn_n_map(nuv2*mnpd2)
25 integer, intent(in) :: lasym
26
27 INTEGER :: l, n, m
28 REAL(rprec) :: fl, fl1, sign1
29
30 ! ALL EQUATIONS REFER TO THE PAPER BY P. MERKEL (PKM)
31 ! IN J. COMPUT. PHYSICS 66, p83 (1986)
32 !
33 ! IN GOING BETWEEN THE COMPLEX NOTATION OF (PKM) AND OUR REAL FORM,
34 ! NOTE THAT THE INTEGRALS (APPENDIX, PKM) Imn AND Kmn ARE BOTH REAL.
35 ! THUS, THE SIN(mu-nv) INTEGRALS OF THE SINGULAR PIECE (ANALYTIC CONTRIBUTION)
36 ! VANISHES.
37 !
38 ! THE REQUIRED SOURCE-TERM INTEGRALS ARE (Eq.2.16-2.17):
39 !
40 ! BVECS(m,n) = Int< SIN(mu' - nv') han(u',v') >
41 ! BVECC(m,n) = Int< COS(mu' - nv') han(u',v') >
42 !
43 ! Where Int<...> means integration over u (theta) and v (zeta) and
44 ! summation over field periods. These can be written in terms of PKM integrals
45 ! Imn(a,b,c), where a(u,v) = guu (g theta-theta), etc.:
46 !
47 ! BVECS(m,n) = ALP * Int<SIN(mu' - nv') * F * Im,-n(a,b,c)>
48 ! BVECC(m,n) = ALP * Int<COS(mu' - nv') * F * Im,-n(a,b,c)>
49 !
50 ! Here, F = - BNORM(u',v') is defined in Eq.(2.13), and ALP = (2*pi/nfp).
51 !
52 ! Similarly, the analytic part of the matrix A(m,n;m',n') can be written:
53 !
54 ! A(m,n;m',n') = (2*pi/nfp) * Int<SIN(mu' - nv')*SIN(m'u' - n'v')
55 ! [Km,-n](a',b',c';A',B',C')>
56 !
57 ! On EXIT, GRPMN(ip,m,n) = ALP * SIN(ip,m,n) * K[m,-n](ip)
58 !
59 ! COMPUTE ALL QUANTITIES INDEPENDENT OF THE MODE INDICES L,M,N
60 ! NOTE: 2b = guv_b HAS FACTOR OF 2 BUILT IN (see SUBROUTINE SURFACE)
61 !
62 ! ADP(M): a +(-)2b + c
63 ! CMA: c - a
64 ! DELTA: 4*(ac - b**2)
65 ! AZP(M): A +(-)2*B + C
66 ! CMA1: C - A
67 ! R1P(M): Coefficient of l*Tl+(-) in eq (A17)
68 ! R0P(M): Coefficient of l*T(l-1)+(-) in eq (A17)
69 ! RA1P(M):Coefficient of Tl+(-) in eq (A17)
70
71 adp = guu_b + guv_b + gvv_b
72 adm = guu_b - guv_b + gvv_b
73 cma = gvv_b - guu_b
74 sqrtc = two*sqrt(gvv_b)
75 sqrta = two*sqrt(guu_b)
76 sqad1u = sqrt(adp)
77 sqad2u = sqrt(adm)
78
79 ! INITIALIZE VECTORS
80 bvec = 0
81 IF (ivacskip .eq. 0) THEN
82 grpmn = 0
83
84 delt1u = adp*adm - cma*cma
85 azp1u = auu + auv + avv
86 azm1u = auu - auv + avv
87 cma11u = avv - auu
88 r1p = (azp1u*(delt1u - cma*cma)/adp - azm1u*adp + two*cma11u*cma)/delt1u
89 r1m = (azm1u*(delt1u - cma*cma)/adm - azp1u*adm + two*cma11u*cma)/delt1u
90 r0p = (-azp1u*adm*cma/adp - azm1u*cma + two*cma11u*adm)/delt1u
91 r0m = (-azm1u*adp*cma/adm - azp1u*cma + two*cma11u*adp)/delt1u
92 ra1p = azp1u/adp
93 ra1m = azm1u/adm
94 ENDIF
95
96 ! INITIALIZE T0+ and T0-
97 !
98 ! TLP(M): TL+(-)
99 ! TLP(M)1:T(L-1)+(-)
100 ! TLP(M)2:T(L-2)+(-)
101 tlp1 = 0
102 tlm1 = 0
103 tlp = one/sqad1u*log((sqad1u*sqrtc + adp + cma)/(sqad1u*sqrta - adp + cma))
104 tlm = one/sqad2u*log((sqad2u*sqrtc + adm + cma)/(sqad2u*sqrta - adm + cma))
105 tlpm = tlp + tlm
106
107 ! BEGIN L-SUM IN EQ (A14) TO COMPUTE Imn (and Kmn) INTEGRALS
108 ! NOTE THAT IN THE LOOP OVER L BELOW: L == |m - n| + 2L_A14
109 ! THUS, L BELOW IS THE INDEX OF THE T+- (S+-)
110 sign1 = 1
111 fl1 = 0
112
113 lloop: DO l = 0, mf + nf
114 fl = fl1
115
116 ! COMPUTE SL+ and SL- , Eq (A17)
117 ! SLP(M): SL+(-)
118 IF (ivacskip .eq. 0) THEN
119 slp = (r1p*fl + ra1p)*tlp + r0p*fl*tlp1 - (r1p + r0p)/sqrtc + sign1*(r0p - r1p)/sqrta
120 slm = (r1m*fl + ra1m)*tlm + r0m*fl*tlm1 - (r1m + r0m)/sqrtc + sign1*(r0m - r1m)/sqrta
121 slpm = slp + slm
122
123 ! here, tlp/m and slp/m are available for the current value of l
124 ! --> save into matrix for debugging
125 all_tlp(:,l) = tlp
126 all_tlm(:,l) = tlm
127 all_slp(:,l) = slp
128 all_slm(:,l) = slm
129 ENDIF
130
131 ! BEGIN MODE NUMBER (m,n) LOOP
132 DO n = 0, nf
133 DO m = 0, mf
134 ! always call analysum, analysum2 to get bvec fill pattern (m_map, n_map)
135 ! IF (cmns(l,m,n) .eq. zero) CYCLE
136
137 IF (n.eq.0 .or. m.eq.0) THEN
138 ! 1. n = 0 and m >= 0 OR n > 0 and m = 0
139 CALL analysum (grpmn, bvec, slpm, tlpm, m, n, l, ivacskip, lasym, m_map, n_map, &
140 grpmn_m_map, grpmn_n_map)
141 ELSE
142 ! 2. n>=1 and m>=1
143 CALL analysum2 (grpmn, bvec, m, n, l, ivacskip, lasym, m_map, n_map, &
144 grpmn_m_map, grpmn_n_map)
145 ENDIF
146 END DO
147 END DO
148
149 ! UPDATE "TL's" (FOR L -> L+1) USING EQ (A15)
150 tlp2 = tlp1
151 tlm2 = tlm1
152 tlp1 = tlp
153 tlm1 = tlm
154
155 ! next l
156 fl1 = fl1 + 1
157
158 ! (-1)**l (next l now)
159 sign1 = -sign1
160
161 tlp = ((sqrtc + sign1*sqrta) - (two*fl1 - one)*cma*tlp1 - fl*adm*tlp2)/(adp*fl1)
162 tlm = ((sqrtc + sign1*sqrta) - (two*fl1 - one)*cma*tlm1 - fl*adp*tlm2)/(adm*fl1)
163 tlpm = tlp + tlm
164 END DO lloop
165
166END SUBROUTINE analyt

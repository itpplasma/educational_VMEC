244SUBROUTINE write_indata_namelist (iunit, istat)
245 IMPLICIT NONE
246
247 INTEGER, INTENT(in) :: iunit
248 INTEGER, INTENT(inout) :: istat
249 INTEGER :: iftol,i,n,m
250 INTEGER, DIMENSION(1) :: ins
251 CHARACTER(LEN=*), PARAMETER :: outboo = "(2X,A,1X,'=',1X,L1)"
252 CHARACTER(LEN=*), PARAMETER :: outint = "(2X,A,1X,'=',1X,I0)"
253 CHARACTER(LEN=*), PARAMETER :: outint1 = "(2X,A,1X,'=',1X,I1.1)"
254 CHARACTER(LEN=*), PARAMETER :: outint2 = "(2X,A,1X,'=',1X,I2.2)"
255 CHARACTER(LEN=*), PARAMETER :: outint3 = "(2X,A,1X,'=',1X,I3.3)"
256 CHARACTER(LEN=*), PARAMETER :: outint4 = "(2X,A,1X,'=',1X,I4.4)"
257 CHARACTER(LEN=*), PARAMETER :: outint5 = "(2X,A,1X,'=',1X,I5.5)"
258 CHARACTER(LEN=*), PARAMETER :: outint6 = "(2X,A,1X,'=',1X,I6.6)"
259 CHARACTER(LEN=*), PARAMETER :: outflt="(2X,A,1X,'=',1X,ES22.12E3)"
260 CHARACTER(LEN=*), PARAMETER :: outexp="(2X,A,1X,'=',1X,ES22.12E3)"
261 IF (istat < 0) RETURN
262 WRITE(iunit,'(A)') '!----- Runtime Parameters -----'
263 WRITE(iunit,'(A)') '&INDATA'
264 WRITE(iunit,outflt) 'DELT',delt
265 WRITE(iunit,outint) 'NSTEP',nstep
266 WRITE(iunit,outflt) 'TCON0',tcon0
267 ins = maxloc(ns_array)
268 WRITE(iunit,'(a,(1p,4i14))') ' NS_ARRAY = ',(ns_array(i), i=1,ins(1))
269 iftol = 1
270 DO WHILE(ftol_array(iftol).ne.0 .and. iftol.lt.100)
271 iftol = iftol + 1
272 END DO
273 WRITE(iunit,'(a,(1p,4e14.6))')' FTOL_ARRAY = ',(ftol_array(i), i=1,iftol - 1)
274 ins = minloc(niter_array)
275 IF (ins(1) > 1) WRITE(iunit,'(a,(1p,4i14))') ' NITER_ARRAY = ',(niter_array(i), i=1,ins(1)-1)
276 WRITE(iunit,'(A)') '!----- Grid Parameters -----'
277 WRITE(iunit,outboo) 'LASYM',lasym
278 WRITE(iunit,outint4) 'NFP',nfp
279 WRITE(iunit,outint4) 'MPOL',mpol
280 WRITE(iunit,outint4) 'NTOR',ntor
281 WRITE(iunit,outflt) 'PHIEDGE',phiedge
282 WRITE(iunit,'(A)') '!----- Free Boundary Parameters -----'
283 WRITE(iunit,outboo) 'LFREEB',lfreeb
284 IF (lfreeb) THEN
285 WRITE (iunit, '(2x,3a)') "MGRID_FILE = '",trim(mgrid_file),"'"
286 WRITE(iunit,outint4) 'NZETA',nzeta
287 DO n=1,SIZE(extcur)
288 IF (extcur(n) == 0) cycle
289 WRITE(iunit,'(2X,A,I3.3,A,ES22.12E3)') 'EXTCUR(',n,') = ',extcur(n)
290 END DO
291 WRITE(iunit,outint4) 'NVACSKIP',nvacskip
292 END IF
293 WRITE(iunit,'(A)') '!----- Pressure Parameters -----'
294 WRITE(iunit,outflt) 'GAMMA',gamma
295 WRITE(iunit,outflt) 'BLOAT',bloat
296 WRITE(iunit,outflt) 'SPRES_PED',spres_ped
297 WRITE(iunit,outflt) 'PRES_SCALE',pres_scale
298 WRITE(iunit,'(2x,3a)') "PMASS_TYPE = '",trim(pmass_type),"'"
299 WRITE(iunit,'(a,(1p,4e22.14))')' AM = ', (am(i-1), i=1,SIZE(am))
300 i = minloc(am_aux_s(2:),dim=1)
301 IF (i > 4) THEN
302 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AM_AUX_S = ', (am_aux_s(n), n=1,i)
303 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AM_AUX_F = ', (am_aux_f(n), n=1,i)
304 END IF
305
306 WRITE(iunit,'(A)') '!----- Current/Iota Parameters -----'
307 WRITE(iunit,outexp) 'CURTOR',curtor
308 WRITE(iunit,outint) 'NCURR',ncurr
309 WRITE (iunit, '(2x,3a)') "PIOTA_TYPE = '",trim(piota_type),"'"
310 WRITE (iunit,'(a,(1p,4e22.14))') ' AI = ',(ai(n-1), n=1,SIZE(ai))
311 i = minloc(ai_aux_s(2:),dim=1)
312 IF (i > 4) THEN
313 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AI_AUX_S = ', (ai_aux_s(n), n=1,i)
314 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AI_AUX_F = ', (ai_aux_f(n), n=1,i)
315 END IF
316 WRITE (iunit, '(2x,3a)') "PCURR_TYPE = '",trim(pcurr_type),"'"
317 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AC = ',(ac(n-1), n=1,SIZE(ac))
318 i = minloc(ac_aux_s(2:),dim=1)
319 IF (i > 4) THEN
320 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AC_AUX_S = ', (ac_aux_s(n), n=1,i)
321 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AC_AUX_F = ', (ac_aux_f(n), n=1,i)
322 END IF
323
324 WRITE(iunit,'(A)') '!----- Axis Parameters ----- '
325 WRITE (iunit,'(a,(1p,4e22.14))') ' RAXIS_CC = ',(raxis_cc(n), n=0,ntor)
326 IF (lasym) then
327 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' RAXIS_CS = ',(raxis_cs(n), n=0,ntor)
328 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' ZAXIS_CC = ',(zaxis_cc(n), n=0,ntor)
329 end if
330 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' ZAXIS_CS = ',(zaxis_cs(n), n=0,ntor)
331
332 WRITE(iunit,'(A)') '!----- Boundary Parameters -----'
333 DO m = 0, mpol - 1
334 DO n = -ntor, ntor
335 IF ((rbc(n,m).ne.0) .or. (zbs(n,m).ne.0)) THEN
336 WRITE(iunit,'(2(A,I4.3,A,I3.3,A,ES22.12E3))') &
337 ' RBC(',n,',',m,') = ',rbc(n,m),' ZBS(',n,',',m,') = ',zbs(n,m)
338 IF (lasym) then
339 WRITE(iunit,'(2(A,I4.3,A,I3.3,A,ES22.12E3))') &
340 ' RBS(',n,',',m,') = ',rbs(n,m),' ZBC(',n,',',m,') = ',zbc(n,m)
341 end if
342 END IF
343 END DO
344 END DO
345
346 WRITE(iunit,'(A)') '/'
347
348 RETURN
349
350END SUBROUTINE write_indata_namelist

371SUBROUTINE write_indata_namelist (iunit, istat)
372 IMPLICIT NONE
373
374 INTEGER, INTENT(in) :: iunit
375 INTEGER, INTENT(inout) :: istat
376 INTEGER :: iftol,i,n,m
377 INTEGER, DIMENSION(1) :: ins
378 CHARACTER(LEN=*), PARAMETER :: outboo = "(2X,A,1X,'=',1X,L1)"
379 CHARACTER(LEN=*), PARAMETER :: outint = "(2X,A,1X,'=',1X,I0)"
380 CHARACTER(LEN=*), PARAMETER :: outint1 = "(2X,A,1X,'=',1X,I1.1)"
381 CHARACTER(LEN=*), PARAMETER :: outint2 = "(2X,A,1X,'=',1X,I2.2)"
382 CHARACTER(LEN=*), PARAMETER :: outint3 = "(2X,A,1X,'=',1X,I3.3)"
383 CHARACTER(LEN=*), PARAMETER :: outint4 = "(2X,A,1X,'=',1X,I4.4)"
384 CHARACTER(LEN=*), PARAMETER :: outint5 = "(2X,A,1X,'=',1X,I5.5)"
385 CHARACTER(LEN=*), PARAMETER :: outint6 = "(2X,A,1X,'=',1X,I6.6)"
386 CHARACTER(LEN=*), PARAMETER :: outflt="(2X,A,1X,'=',1X,ES22.12E3)"
387 CHARACTER(LEN=*), PARAMETER :: outexp="(2X,A,1X,'=',1X,ES22.12E3)"
388 IF (istat < 0) RETURN
389 WRITE(iunit,'(A)') '!----- Runtime Parameters -----'
390 WRITE(iunit,'(A)') '&INDATA'
391 WRITE(iunit,outflt) 'DELT',delt
392 WRITE(iunit,outint) 'NSTEP',nstep
393 WRITE(iunit,outflt) 'TCON0',tcon0
394 ins = maxloc(ns_array)
395 WRITE(iunit,'(a,(1p,4i14))') ' NS_ARRAY = ',(ns_array(i), i=1,ins(1))
396 iftol = 1
397 DO WHILE(ftol_array(iftol).ne.0 .and. iftol.lt.100)
398 iftol = iftol + 1
399 END DO
400 WRITE(iunit,'(a,(1p,4e14.6))')' FTOL_ARRAY = ',(ftol_array(i), i=1,iftol - 1)
401 ins = minloc(niter_array)
402 IF (ins(1) > 1) WRITE(iunit,'(a,(1p,4i14))') ' NITER_ARRAY = ',(niter_array(i), i=1,ins(1)-1)
403 WRITE(iunit,'(A)') '!----- Grid Parameters -----'
404 WRITE(iunit,outboo) 'LASYM',lasym
405 WRITE(iunit,outint4) 'NFP',nfp
406 WRITE(iunit,outint4) 'MPOL',mpol
407 WRITE(iunit,outint4) 'NTOR',ntor
408 WRITE(iunit,outflt) 'PHIEDGE',phiedge
409 WRITE(iunit,'(A)') '!----- Free Boundary Parameters -----'
410 WRITE(iunit,outboo) 'LFREEB',lfreeb
411 IF (lfreeb) THEN
412 WRITE (iunit, '(2x,3a)') "MGRID_FILE = '",trim(mgrid_file),"'"
413 WRITE(iunit,outint4) 'NZETA',nzeta
414 DO n=1,SIZE(extcur)
415 IF (extcur(n) == 0) cycle
416 WRITE(iunit,'(2X,A,I3.3,A,ES22.12E3)') 'EXTCUR(',n,') = ',extcur(n)
417 END DO
418 WRITE(iunit,outint4) 'NVACSKIP',nvacskip
419 END IF
420 WRITE(iunit,'(A)') '!----- Pressure Parameters -----'
421 WRITE(iunit,outflt) 'GAMMA',gamma
422 WRITE(iunit,outflt) 'BLOAT',bloat
423 WRITE(iunit,outflt) 'SPRES_PED',spres_ped
424 WRITE(iunit,outflt) 'PRES_SCALE',pres_scale
425 WRITE(iunit,'(2x,3a)') "PMASS_TYPE = '",trim(pmass_type),"'"
426 WRITE(iunit,'(a,(1p,4e22.14))')' AM = ', (am(i-1), i=1,SIZE(am))
427 i = minloc(am_aux_s(2:),dim=1)
428 IF (i > 4) THEN
429 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AM_AUX_S = ', (am_aux_s(n), n=1,i)
430 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AM_AUX_F = ', (am_aux_f(n), n=1,i)
431 END IF
432
433 WRITE(iunit,'(A)') '!----- Current/Iota Parameters -----'
434 WRITE(iunit,outexp) 'CURTOR',curtor
435 WRITE(iunit,outint) 'NCURR',ncurr
436 WRITE (iunit, '(2x,3a)') "PIOTA_TYPE = '",trim(piota_type),"'"
437 WRITE (iunit,'(a,(1p,4e22.14))') ' AI = ',(ai(n-1), n=1,SIZE(ai))
438 i = minloc(ai_aux_s(2:),dim=1)
439 IF (i > 4) THEN
440 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AI_AUX_S = ', (ai_aux_s(n), n=1,i)
441 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AI_AUX_F = ', (ai_aux_f(n), n=1,i)
442 END IF
443 WRITE (iunit, '(2x,3a)') "PCURR_TYPE = '",trim(pcurr_type),"'"
444 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AC = ',(ac(n-1), n=1,SIZE(ac))
445 i = minloc(ac_aux_s(2:),dim=1)
446 IF (i > 4) THEN
447 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AC_AUX_S = ', (ac_aux_s(n), n=1,i)
448 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' AC_AUX_F = ', (ac_aux_f(n), n=1,i)
449 END IF
450
451 WRITE(iunit,'(A)') '!----- Axis Parameters ----- '
452 WRITE (iunit,'(a,(1p,4e22.14))') ' RAXIS_CC = ',(raxis_cc(n), n=0,ntor)
453 IF (lasym) then
454 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' RAXIS_CS = ',(raxis_cs(n), n=0,ntor)
455 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' ZAXIS_CC = ',(zaxis_cc(n), n=0,ntor)
456 end if
457 WRITE (iunit,'(a,(1p,4ES22.12E3))') ' ZAXIS_CS = ',(zaxis_cs(n), n=0,ntor)
458
459 WRITE(iunit,'(A)') '!----- Boundary Parameters -----'
460 DO m = 0, mpol - 1
461 DO n = -ntor, ntor
462 IF ((rbc(n,m).ne.0) .or. (zbs(n,m).ne.0)) THEN
463 WRITE(iunit,'(2(A,I4.3,A,I3.3,A,ES22.12E3))') &
464 ' RBC(',n,',',m,') = ',rbc(n,m),' ZBS(',n,',',m,') = ',zbs(n,m)
465 IF (lasym) then
466 WRITE(iunit,'(2(A,I4.3,A,I3.3,A,ES22.12E3))') &
467 ' RBS(',n,',',m,') = ',rbs(n,m),' ZBC(',n,',',m,') = ',zbc(n,m)
468 end if
469 END IF
470 END DO
471 END DO
472
473 WRITE(iunit,'(A)') '/'
474
475 RETURN

477END SUBROUTINE write_indata_namelist

244 REAL(dp), PARAMETER :: FThreshold = 1.e-6_dp
245 REAL(dp) :: temp(ns,0:ntor)
246
247 ! COMPUTE INTERNAL gr, gz
248 ! NOTE: internal gz => 0 for both values of lconm1 (although gz is different)
249 ! FOR lconm1=T, gcr(internal) = gcr+gcz, gcz(internal) = gcr-gcz->0
250 IF (lconm1) THEN
251 temp = gcr
252 gcr = osqrt2*(gcr + gcz)
253 gcz = osqrt2*(temp - gcz)
254 END IF
255
256 !v8.50: ADD iter2<2 so reset=<WOUT_FILE> works
257 IF (fsqz.LT.fthreshold .OR. iter2.LT.2) then
258! write(*,*) "zero z-force in constrain_m1"
259
260 ! ensure that the m=1 constraint is satisfied exactly
261 ! --> the corresponding m=1 coeffs of R,Z are constrained to be zero
262 ! and thus must not be "forced" (by the time evol using gc) away from zero
263 gcz = 0
264 end if

266END SUBROUTINE constrain_m1

15
16 INTEGER :: i
17
18 ! exterior Neumann problem
19
20 IF (.not.ALLOCATED(bvac)) stop 'BVAC unallocated in bextern'
21
22 ! THIS ROUTINE COMPUTES THE B DOT DS ARISING FROM EXTERNAL COILS AND INTERNAL PLASMA CURRENT
23 ! NOTE THAT BEXN = - BEX * DS IS THE EFFECTIVE SOURCE TERM
24 !
25 ! COMPUTE B FROM COILS ON THE PLASMA BOUNDARY
26 CALL becoil(r1b, z1b, bvac(1,1), bvac(1,2), bvac(1,3))
27
28 ! COMPUTE B (ON PLASMA BOUNDARY) FROM NET TOROIDAL PLASMA CURRENT
29 ! THE NET CURRENT IS MODELLED AS A WIRE AT THE MAGNETIC AXIS, AND THE
30 ! BIOT-SAVART LAW IS USED TO COMPUTE THE FIELD AT THE PLASMA SURFACE
31 !
32 ! USE BEXU, BEXV, BEXN AS TEMPORARY STORAGE FOR BX, BY, BZ
33 CALL belicu (plascur, bexu, bexv, bexn, cosuv, sinuv, r1b, z1b)
34 DO i = 1, nuv2
35 brad(i) = brad(i) + bexu(i)*cosuv(i) + bexv(i)*sinuv(i)
36 bphi(i) = bphi(i) - bexu(i)*sinuv(i) + bexv(i)*cosuv(i)
37 bz(i) = bz(i) + bexn(i)
38 END DO
39
40 ! COMPUTE COVARIANT COMPONENTS OF EXTERNAL FIELD: BEXU = B0 dot dx/du,
41 ! BEXV = B0 dot dx/dv. HERE, BEXN = -B0*SURF_NORM CORRESPONDS TO THE
42 ! "exterior Neumann problem" convention of PKM (sign flipped as noted in PKM)
43 ! THUS, THE UNIT NORMAL SHOULD POINT INTO THE PLASMA (OUTWARD FROM VACUUM),
44 ! WHICH IT DOES FOR A NEGATIVE JACOBIAN (SIGNGS) SYSTEM
45 DO i = 1, nuv2
46 bexu(i) = rub(i)*brad(i) + zub(i)*bz(i)
47 bexv(i) = rvb(i)*brad(i) + zvb(i)*bz(i) + r1b(i)*bphi(i)
48 bexn(i) =-(brad(i)*snr(i) + bphi(i)*snv(i) + bz(i)*snz(i))
49 END DO
50
51 ! COMPUTE NORMALIZED [(2*pi)**2], READY-TO-INTEGRATE (WINT FACTOR) SOURCE TERM
52 ! NOTE: BEXN == NP*F = -B0 dot [Xu cross Xv] NP (see PKM, Eq. 2.13)
53 bexni(:nuv2) = wint(:nuv2)*bexn(:nuv2)*pi2*pi2
54
55END SUBROUTINE bextern

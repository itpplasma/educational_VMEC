91 IF (ier_flag .eq. 0) then

93
94 ! COMPUTE NS-INVARIANT ARRAYS

103 ' BEGIN FORCE ITERATIONS',/,1x,23('-'),/)

105 do jacob_off = 0, 1
106 ! jacob_off=1 indicates that an initial run with ns=3 shall be inserted
107 ! before the user-provided ns values from ns_array are processed
108 ! in the multi-grid run
109
110 ! convergence flag: initially not converged yet

112
113 IF (lfreeb .and. jacob_off.eq.1) then

117 end if

121 ! multi-grid iterations: loop over ns_array
122 ! jacob_off=0,1 is required to insert one ns=3 run before
123 ! starting to work with the user-provided ns_array
124 ! if the first ns value from ns_array gave a bad jacobian
125 iterations: DO igrid = 1-jacob_off, multi_ns_grid
126
127 ! retrieve settings for (ns, ftol, niter) for current multi-grid iteration
128 IF (igrid .lt. 1) THEN

131 ! TRY TO GET NON-SINGULAR JACOBIAN ON A 3 PT RADIAL MESH

139
140 ELSE

143 IF (nsval .lt. ns_min) then
144 ! skip entries that have less flux surfaces than previous iteration

146 end if
147
148 ! update ns_min --> reduction in number of flux surfaces not allowed

151 ftolv = ftol_array(igrid)
152 niterv = niter_array(igrid)
153 END IF

155 IF (ns_old .le. nsval) then

159 end if
160
161 ! *HERE* is the *ACTUAL* call to the equilibrium solver !
162 CALL eqsolve (ier_flag)
163
164 ! break the multi-grid sequence if current number of flux surfaced
165 ! did not reach convergence
166 IF (ier_flag.ne.norm_term_flag .and. ier_flag.ne.successful_term_flag) then
167 EXIT
168 end if
169
170
171 ! If this point is reached, the current multi-grid step should have properly converged.
172 ! Now dump the current state vector for debugging.
173 if (open_dbg_context("multigrid_result")) then
174
175 call add_int("ns", nsval)
176 call add_int("iter2", iter2)
177 call add_int("ncurr", ncurr)
178 call add_real("phiedge", phiedge)
179 call add_int("ntmax", ntmax)
180
181 call add_real_1d("pres", ns-1, pres(2:ns))
182 call add_real_1d("iotas", ns-1, iotas(2:ns))
183 call add_real_1d("chips", ns-1, chips(2:ns))
184
185 call add_real_5d("xc", 3, ntmax, ns, ntor1, mpol, xc, order=(/ 3, 4, 5, 2, 1 /) )
186 call add_real_5d("scalxc", 3, ntmax, ns, ntor1, mpol, scalxc, order=(/ 3, 4, 5, 2, 1 /) )
187
188 call close_dbg_out()
189 end if
190
191 END DO iterations
192
193
194 ! if did not converge only because jacobian was bad
195 ! and the intermediate ns=3 run was not performed yet (jacob_off is still == 0),
196 ! retry the whole thing again
197 IF (ier_flag.ne.bad_jacobian_flag) THEN
198 exit ! jacob_off loop
199 ! otherwise, retry with initial ns=3 to fix bad jacobian
200 END IF
201
202 ! if ier_flag .eq. bad_jacobian_flag, repeat once again with ns=3 before
203 end do ! jacob_off = 0, 1
204
205 end if ! (ier_flag .eq. 0) after readin
206
207
208
209 ! write output files
210 CALL fileout (ier_flag)
211
212
213
214
215
216
217 ! free memory
218 IF (ALLOCATED(cosmu)) then
219 DEALLOCATE(cosmu, sinmu, cosmum, sinmum, cosmui, cosmumi, &
220 sinmui, sinmumi, cosnv, sinnv, cosnvn, sinnvn, &
221 cosmui3, cosmumi3, cos01, sin01, stat=istat1)
222 IF (istat1 .ne. 0) print *, warning // "#1"
223 end if
224
225 IF (ALLOCATED(xm)) then
226 DEALLOCATE (xm, xn, ixm, xm_nyq, xn_nyq, &
227 jmin3, mscale, nscale, uminus, stat=istat1)
228 IF (istat1 .ne. 0) print *, warning // "#2"
229 end if
230
231

234 CALL free_mem_ns
235 CALL free_mem_nunv
236
237 CALL free_mgrid (istat1)
238 IF (istat1.ne.0) THEN
239 print *,'problem in free_mgrid'
240 print *,' istat1 = ',istat1
241 ENDIF
242
243 ! verbose error message
244 SELECT CASE (ier_flag)
245 CASE (bad_jacobian_flag)
246 ! Bad jacobian even after axis reset and ns->3
247 WRITE ( 6, '(/,1x,a)') bad_jacobian
248 WRITE (nthreed, '(/,1x,a)') bad_jacobian
249 CASE DEFAULT
250 END SELECT

253 IF (nthreed .gt. 0) CLOSE (nthreed)

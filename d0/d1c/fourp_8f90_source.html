20 DO ku = 1,nu2
21 g1 = 0
22 g2 = 0

24 ! PERFORM KV (TOROIDAL ANGLE) TRANSFORM (OVER UNPRIMED MESH IN EQ. 2.14)
25 ! THUS, THE (m,n) INDEX HERE CORRESPONDS TO THE FIRST INDEX OF AMATRIX
26 !
27 ! NOTE: THE .5 FACTOR (IN COSN,SINN) ACCOUNTS FOR THE SUM IN KERNEL.
28 ! ON ENTRY THE FIRST TIME, GRPMN IS SIN,COS * Kmn(analytic)
29 !
30 ! THE 3rd INDEX OF GRPMN IS THE PRIMED U,V MESH COORDINATE
31 DO kv = 1,nv
32 DO n = 0,nf
33 cosn = p5*onp*cosv(n,kv)
34 sinn = p5*onp*sinv(n,kv)
35 iuv = kv+nv*(ku-1)
36 ireflect = imirr(iuv)
37 DO isym = 1, ndim ! ndim == 1 for stellarator-symmetry, 2 for asymmetric (lasym=T)
38 DO ip = 1,nuv2
39 IF (isym .eq. 1) THEN ! only contrib for stellarator-symmetry
40 ! anti-symmetric part (u,v -> -u,-v)
41
42 kernel = grp(iuv,ip) - grp(ireflect,ip)
43
44! if (iuv .eq. ireflect) then
45! print *, "reflecting on self at iuv=", iuv, " ip=", ip, " n=", n, " ku=", ku, " kv=", kv, &
46! " => kernel=", kernel
47! end if
48
49! if (n.eq.0) then
50! ! fourp_kernel.dat
51! print *, iuv, ip, kernel
52! end if
53 ELSE IF (isym .eq. 2) THEN
54 ! symmetric part
55 kernel = grp(iuv,ip) + grp(ireflect,ip)
56 END IF
57
58
59

61 g2(ip,n,isym)=g2(ip,n,isym) + sinn*kernel
62 END DO
63 END DO
64 END DO
65 END DO
66
67 ! PERFORM KU (POLOIDAL ANGLE) TRANFORM [COMPLETE SIN(mu-nv) / COS(mu-nv) TRANSFORM]
68 DO m = 0,mf
69 DO isym = 1, ndim
70 IF (isym .EQ. 1) THEN
71 cosm = -cosui(m,ku)
72 sinm = sinui(m,ku)
73 ELSE IF (isym .EQ. 2) THEN
74 sinm = cosui(m,ku)
75 cosm = sinui(m,ku)
76 END IF
77 DO n= 0,nf
78 DO ip = 1,nuv2
79 gcos = g1(ip,n,isym)*sinm
80 gsin = g2(ip,n,isym)*cosm
81 grpmn(m, n,ip,isym) = grpmn(m, n,ip,isym) + gcos + gsin
82 IF (n .NE. 0) THEN
83 grpmn(m,-n,ip,isym) = grpmn(m,-n,ip,isym) + gcos - gsin
84 ENDIF
85 END DO
86 END DO
87 END DO
88 END DO
89
90 END DO

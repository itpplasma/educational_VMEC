29 ! COMPUTE TRANFORM OF ANALYTIC SOURCE AND KERNEL.
30 ! ON EXIT:
31 ! BVEC CONTAINS THE TRANSFORM OF THE ANALYTIC SOURCE AND
32 ! GRPMN CONTAINS THE TRANSFORM OF THE NORMAL DERIVATIVE OF THE GREENS FUNCTION [PKM, EQ.(2.15)]
33 ! GReen's function Primed (normal derivative...) and Fourier-transformed to MN mode numbers --> "GR P MN"

35
36 IF (ivacskip .ne. 0) THEN
37 ! FOR ivacskip != 0, USE PREVIOUSLY COMPUTED bvecsav FOR SPEED
38
39 ! Here, bvecsav contains the previous non-singular contribution to the "final" bvec from fouri.
40 ! For ivacskip != 0, this contribution is used from the cache in bvecsav.
41 bvec = bvec + bvecsav
42 ELSE
43 ! Here, bvecsav stores the singular part of bvec from analyt
44 ! to be subtracted from the "final" bvec computed below by fouri
45 bvecsav = bvec
46
47 ! COMPUTE SURFACE INTEGRALS OF SOURCE AND GREENS FUNCTION NEEDED
48 ! FOR SPECTRAL DECOMPOSITION OF POTENTIAL INTEGRAL EQUATION
49 ! NOTE: SOURCE IS THE RHS OF EQ.(3.2), KERNEL IS THE LHS OF EQ (3.2).
50 ! IP IS THE INDEX OF THE PRIMED VARIABLE MESH.
51 gstore = 0
52 DO ip = 1, nuv2
53
54 ! COMPUTE DIFFERENCE BETWEEN THE EXACT AND ANALYTIC GREENS FUNCTION AND GRADIENT
55 ! [FIRST TERMS IN EQ.(2.14, 2.16)].
56 CALL greenf (green(1,ip), greenp(1,ip), ip)
57
58! ! debugging: need to fix greenf for Tokamak first...
59! stop
60
61 ! PERFORM INTEGRAL (SUM) OVER PRIMED MESH OF NON-SINGULAR SOURCE TERM
62 ! [(h-hsing)(u,v,u',v') == bexni(ip)*green(u,v; ip) in Eq. 2.16]
63 ! AND STORE IT - FOR UNPRIMED MESH VALUES - IN GSTORE
64 gstore = gstore + bexni(ip)*green(:,ip)
65 END DO
66
67 if (open_dbg_context("vac1n_greenf", id=icall)) then
68
69 call add_real_4d("green", nv, nu, nv, nu3, green)
70 call add_real_4d("greenp", nv, nu, nv, nu3, greenp)
71
72 call add_real_2d("gstore", nv, nu, gstore)
73
74 call close_dbg_out()
75 end if
76
77 ! PERFORM FOURIER INTEGRAL OF GRADIENT KERNEL (GREENP) OVER THE UNPRIMED MESH
78 ! AND STORE IN GRPMN (NOTE THAT GRPMN IS ADDED TO THE ANALYTIC PIECE IN EQ. 2.14,
79 ! - COMPUTED IN ANALYT - WHICH HAS THE APPROPRIATE SIN, COS FACTORS ALREADY)
80 CALL fourp (grpmn, greenp)
81
82 ! COMPUTE FOURIER INTEGRAL OF GRADIENT (GRPMN) OVER PRIMED MESH IN EQ. 2.14
83 ! AND SOURCE (GSTORE) OVER UNPRIMED MESH IN EQ. 2.16
84 CALL fouri (grpmn, gstore, amatrix, amatsav, bvec, wint, lasym)
85
86 ! debugging: focus on Fourier transforms in fouri for now
87 ! return
88
89 ! SAVE NON-SINGULAR CONTRIBUTION TO BVEC (IN BVECSAV)
90 bvecsav(:mnpd2) = bvec - bvecsav(:mnpd2)
91
92 ENDIF
93
94 amatrix = amatsav

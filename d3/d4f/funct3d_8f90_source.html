399 call add_real_3d("gss", ns, ntor1, mpol, extra1(:,1))

401 call close_dbg_out()
402 end if

404 ! COMPUTE MHD FORCES ON INTEGER-MESH
405 CALL forces
406
407 ! SYMMETRIZE FORCES (in u-v space): NOTE - gc IS SMALL BY FACTOR 2 IF lasym=T
408 IF (lasym) THEN
409 CALL symforce (armn, brmn, crmn, azmn, bzmn, czmn, blmn, clmn, rcon, zcon, &
410 r1, ru, rv, z1, zu, zv, extra3, extra4, extra1, extra2 )
411
412 ! NOT NECESSARY (EVEN THOUGH CORRECT) --> why?
413 ! gc = 2*gc
414 END IF
415
416 ! FOURIER-TRANSFORM MHD FORCES TO (M,N)-SPACE
417 ! Note that gc is immediately zeroed on entry,
418 ! so above use of gc is only for temporary values inside alias() !
419 CALL tomnsps (gc, &
420 armn, brmn, crmn, &
421 azmn, bzmn, czmn, &
422 blmn, clmn, rcon, zcon)
423 call tomnsps_con(gc_con, brmn_con, bzmn_con, rcon, zcon)
424 IF (lasym) then
425 CALL tomnspa (gc, &
426 r1, ru, rv, &
427 z1, zu, zv, &
428 extra3, extra4, extra1, extra2)
429 call tomnspa_con(gc_con, brmn_con, bzmn_con, extra1, extra2)
430 end if
431
432 ! COMPUTE FORCE RESIDUALS (RAW AND PRECONDITIONED)
433 gc = gc * scalxc !!IS THIS CORRECT: SPH010214?
434 gc_con = gc_con * scalxc
435 gc_mhd = gc - gc_con
436
437 fsqrz = fsqr + fsqz
438 old_fsqz = fsqz
439
440 CALL residue (gc, gc(1+irzloff), gc(1+2*irzloff), fsqrz, old_fsqz)
441 call residue_con(gc_con, gc_con(1+irzloff), gc_con(1+2*irzloff), fsqrz, old_fsqz)
442 call residue_mhd(gc_mhd, gc_mhd(1+irzloff), gc_mhd(1+2*irzloff), fsqrz, old_fsqz)
443
444 IF (iter2.eq.1 .and. (fsqr+fsqz+fsql).gt.1.e2_dp) then
445 ! first iteration and gigantic force residuals --> what is going one here?
446 first = 4 ! fatal error
447 end if
448
449! ELSE
450 ! iequi == 1 --> skip above remainder of funct3d
451 END IF

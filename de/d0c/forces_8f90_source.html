14  INTEGER :: l
15  INTEGER :: ndim
16  REAL(rprec), DIMENSION(:), POINTER :: bsqr

21  ! ON ENTRY, ARMN=ZU, BRMN=ZS, AZMN=RU, BZMN=RS, LU=R*BSQ, LV = BSQ*SQRT(G)/R12
22  ! HERE, XS (X=Z,R) DO NOT INCLUDE DERIVATIVE OF EXPLICIT SQRT(S)
23  ! BSQ = |B|**2/2 + p
24  ! GIJ = (BsupI * BsupJ) * SQRT(G) (I,J = U,V)
25  ! IT IS ESSENTIAL THAT LU,LV AT j=1 ARE ZERO INITIALLY
26  !
27  ! SOME OF THE BIGGER LOOPS WERE SPLIT TO FACILITATE CACHE HITS, PIPELINING ON RISCS
28  !
29  ! ORIGIN OF VARIOUS TERMS
30  !
31  ! LU : VARIATION OF DOMINANT RADIAL-DERIVATIVE TERMS IN JACOBIAN
32  !
33  ! LV : VARIATION OF R-TERM IN JACOBIAN
34  !
35  ! GVV: VARIATION OF R**2-TERM AND Rv**2,Zv**2 IN gvv
36  !
37  ! GUU, GUV: VARIATION OF Ru, Rv, Zu, Zv IN guu, guv
38 
39  ndim = 1+nrzt
40 
41  ! POINTER ALIASES
42  bsqr => extra1(:,1)
43  gvvs => extra2(:,1)
44  guvs => extra3(:,1)
45  guus => extra4(:,1)
46 
47  lu_e(1:ndim:ns) = 0
48  lv_e(1:ndim:ns) = 0
49  guu(1:ndim:ns) = 0
50  guv(1:ndim:ns) = 0
51  gvv(1:ndim:ns) = 0
52  guus = guu*shalf
53  guvs = guv*shalf
54  gvvs = gvv*shalf
55 
56  armn_e = ohs*zu12 * lu_e
57  azmn_e =-ohs*ru12 * lu_e
58  brmn_e = brmn_e * lu_e
59  bzmn_e =-bzmn_e * lu_e
60  bsqr = dshalfds*lu_e/shalf
61 
62  armn_o(1:ndim) = armn_e(1:ndim) *shalf
63  azmn_o(1:ndim) = azmn_e(1:ndim) *shalf
64  brmn_o(1:ndim) = brmn_e(1:ndim) *shalf
65  bzmn_o(1:ndim) = bzmn_e(1:ndim) *shalf
66 
67  ! CONSTRUCT CYLINDRICAL FORCE KERNELS
68  ! NOTE: presg(ns+1) == 0, AND WILL BE "FILLED IN" AT EDGE FOR FREE-BOUNDARY BY RBSQ
69  DO l = 1, nrzt
70  guu(l) = p5*(guu(l) + guu(l+1))
71  gvv(l) = p5*(gvv(l) + gvv(l+1))
72  bsqr(l) = bsqr(l) + bsqr(l+1)
73  guus(l) = p5*(guus(l) + guus(l+1))
74  gvvs(l) = p5*(gvvs(l) + gvvs(l+1))
75  armn_e(l) = armn_e(l+1) - armn_e(l) + p5*(lv_e(l) + lv_e(l+1))
76  azmn_e(l) = azmn_e(l+1) - azmn_e(l)
77  brmn_e(l) = p5*(brmn_e(l) + brmn_e(l+1))
78  bzmn_e(l) = p5*(bzmn_e(l) + bzmn_e(l+1))
79  END DO
80 
81  armn_e(:nrzt) = armn_e(:nrzt) - (gvvs(:nrzt)*r1(:nrzt,1) + gvv(:nrzt)*r1(:nrzt,0))
82  brmn_e(:nrzt) = brmn_e(:nrzt) + bsqr(:nrzt)*z1(:nrzt,1) - (guus(:nrzt)*ru(:nrzt,1) + guu(:nrzt)*ru(:nrzt,0))
83  bzmn_e(:nrzt) = bzmn_e(:nrzt) - (bsqr(:nrzt)*r1(:nrzt,1) + guus(:nrzt)*zu(:nrzt,1) + guu(:nrzt)*zu(:nrzt,0))
84  lv_e(1:ndim) = lv_e(1:ndim)*shalf(1:ndim)
85  lu_o(1:ndim) = dshalfds*lu_e(1:ndim)
86 
87  DO l = 1, nrzt
88  armn_o(l) = armn_o(l+1) - armn_o(l) - zu(l,0)*bsqr(l) + p5*(lv_e(l) + lv_e(l+1))
89  azmn_o(l) = azmn_o(l+1) - azmn_o(l) + ru(l,0)*bsqr(l)
90  brmn_o(l) = p5*(brmn_o(l) + brmn_o(l+1))
91  bzmn_o(l) = p5*(bzmn_o(l) + bzmn_o(l+1))
92  lu_o(l) = lu_o(l) + lu_o(l+1)
93  END DO
94 
95  guu(1:nrzt) = guu(1:nrzt) * sqrts(1:nrzt)**2
96  bsqr(1:nrzt) = gvv(1:nrzt) * sqrts(1:nrzt)**2
97 
98  armn_o(:nrzt) = armn_o(:nrzt) - (zu(:nrzt,1)*lu_o(:nrzt) + bsqr(:nrzt)*r1(:nrzt,1) + gvvs(:nrzt)*r1(:nrzt,0))
99  azmn_o(:nrzt) = azmn_o(:nrzt) + ru(:nrzt,1)*lu_o(:nrzt)
100  brmn_o(:nrzt) = brmn_o(:nrzt) + z1(:nrzt,1)*lu_o(:nrzt) - (guu(:nrzt)*ru(:nrzt,1) + guus(:nrzt)*ru(:nrzt,0))
101  bzmn_o(:nrzt) = bzmn_o(:nrzt) - (r1(:nrzt,1)*lu_o(:nrzt) + guu(:nrzt)*zu(:nrzt,1) + guus(:nrzt)*zu(:nrzt,0))
102 
103  IF (lthreed) THEN
104  DO l = 1, nrzt
105  guv(l) = p5*(guv(l) + guv(l+1))
106  guvs(l) = p5*(guvs(l) + guvs(l+1))
107  END DO
108 
109  brmn_e(:nrzt) = brmn_e(:nrzt) - (guv(:nrzt)*rv(:nrzt,0) + guvs(:nrzt)*rv(:nrzt,1))
110  bzmn_e(:nrzt) = bzmn_e(:nrzt) - (guv(:nrzt)*zv(:nrzt,0) + guvs(:nrzt)*zv(:nrzt,1))
111  crmn_e(:nrzt) = guv(:nrzt) *ru(:nrzt,0) + gvv(:nrzt) *rv(:nrzt,0) + gvvs(:nrzt)*rv(:nrzt,1) + guvs(:nrzt)*ru(:nrzt,1)
112  czmn_e(:nrzt) = guv(:nrzt) *zu(:nrzt,0) + gvv(:nrzt) *zv(:nrzt,0) + gvvs(:nrzt)*zv(:nrzt,1) + guvs(:nrzt)*zu(:nrzt,1)
113  guv(:nrzt) = guv(:nrzt) *sqrts(:nrzt)*sqrts(:nrzt)
114  brmn_o(:nrzt) = brmn_o(:nrzt) - (guvs(:nrzt)*rv(:nrzt,0) + guv(:nrzt)*rv(:nrzt,1))
115  bzmn_o(:nrzt) = bzmn_o(:nrzt) - (guvs(:nrzt)*zv(:nrzt,0) + guv(:nrzt)*zv(:nrzt,1))
116  crmn_o(:nrzt) = guvs(:nrzt)*ru(:nrzt,0) + gvvs(:nrzt)*rv(:nrzt,0) + bsqr(:nrzt)*rv(:nrzt,1) + guv(:nrzt) *ru(:nrzt,1)
117  czmn_o(:nrzt) = guvs(:nrzt)*zu(:nrzt,0) + gvvs(:nrzt)*zv(:nrzt,0) + bsqr(:nrzt)*zv(:nrzt,1) + guv(:nrzt) *zu(:nrzt,1)
118  ENDIF
119 
120  ! ASSIGN EDGE FORCES (JS = NS) FOR FREE BOUNDARY CALCULATION
121  IF (ivac .ge. 1) THEN
122  armn_e(ns:nrzt:ns) = armn_e(ns:nrzt:ns) + zu0(ns:nrzt:ns)*rbsq(1:nznt)
123  armn_o(ns:nrzt:ns) = armn_o(ns:nrzt:ns) + zu0(ns:nrzt:ns)*rbsq(1:nznt)
124  azmn_e(ns:nrzt:ns) = azmn_e(ns:nrzt:ns) - ru0(ns:nrzt:ns)*rbsq(1:nznt)
125  azmn_o(ns:nrzt:ns) = azmn_o(ns:nrzt:ns) - ru0(ns:nrzt:ns)*rbsq(1:nznt)
126  ENDIF
127 
128 ! #ifndef _HBANGLE
129  ! COMPUTE CONSTRAINT FORCE KERNELS
130  rcon(:nrzt,0) = (rcon(:nrzt,0) - rcon0(:nrzt)) * gcon(:nrzt)
131  zcon(:nrzt,0) = (zcon(:nrzt,0) - zcon0(:nrzt)) * gcon(:nrzt)
132 
133  brmn_e(:nrzt) = brmn_e(:nrzt) + rcon(:nrzt,0)
134  bzmn_e(:nrzt) = bzmn_e(:nrzt) + zcon(:nrzt,0)
135  brmn_o(:nrzt) = brmn_o(:nrzt) + rcon(:nrzt,0) * sqrts(:nrzt)
136  bzmn_o(:nrzt) = bzmn_o(:nrzt) + zcon(:nrzt,0) * sqrts(:nrzt)
137 
138  rcon(:nrzt,0) = ru0(:nrzt) * gcon(:nrzt)
139  zcon(:nrzt,0) = zu0(:nrzt) * gcon(:nrzt)
140  rcon(:nrzt,1) = rcon(:nrzt,0) * sqrts(:nrzt)
141  zcon(:nrzt,1) = zcon(:nrzt,0) * sqrts(:nrzt)
142 ! #end /* ndef _HBANGLE */
143 
144 END SUBROUTINE forces

58 ! SCALE VP, PHIPS TO REAL UNITS (VOLUME, TOROIDAL FLUX DERIVATIVES)
59 ! AND PUT GSQRT IN ABS UNITS (SIGNGS MAY BE NEGATIVE)
60 ! NOTE: VP has (coming into this routine) the sign of the jacobian multiplied out
61 ! i.e., vp = signgs*<gsqrt>
62 ! THE SHEAR TERM MUST BE MULTIPLIED BY THE SIGN OF THE JACOBIAN
63 ! (OR A BETTER SOLUTION IS TO RETAIN THE JACOBIAN SIGN IN ALL TERMS, INCLUDING
64 ! VP, THAT DEPEND EXPLICITLY ON THE JACOBIAN. WE CHOOSE THIS LATTER METHOD...)
65 !
66 ! COMING INTO THIS ROUTINE, THE JACOBIAN(gsqrt) = 1./(grad-s . grad-theta X grad-zeta)
67 ! WE CONVERT THIS FROM grad-s to grad-phi DEPENDENCE BY DIVIDING gsqrt by PHIP_real
68 !
69 ! NOTE: WE ARE USING 0 < s < 1 AS THE FLUX VARIABLE, BEING CAREFUL
70 ! TO KEEP d(phi)/ds == PHIP_real FACTORS WHERE REQUIRED
71 ! THE V'' TERM IS d2V/d(PHI)**2, PHI IS REAL TOROIDAL FLUX
72 !
73 ! SHEAR = d(iota)/d(phi) : FULL MESH
74 ! VPP = d(vp)/d(phi) : FULL MESH
75 ! PRESP = d(pres)/d(phi) : FULL MESH (PRES IS REAL PRES*mu0)
76 ! IP = d(Itor)/d(phi) : FULL MESH
77 !
78 ! ON ENTRY, BDOTJ = Jacobian * J*B ON THE FULL RADIAL GRID
79 ! BSQ = 0.5*|B**2| + p IS ON THE HALF RADIAL GRID
80
81 ns1 = ns - 1
82 IF (ns1 .le. 0) RETURN
83 hs = one/ns1
84 sign_jac = zero
85 IF (gsqrt(ns,1) .ne. zero) then
86 sign_jac = abs(gsqrt(ns,1))/gsqrt(ns,1)
87 end if
88
89 IF (sign_jac .eq. zero) RETURN
90 phip_real = twopi * phips * sign_jac
91
92 ! NOTE: phip_real should be > 0 to get the correct physical sign of REAL-space gradients
93 ! For example, grad-p, grad-Ip, etc. However, with phip_real defined this way,
94 ! Mercier will be correct
95 vp_real(2:ns) = sign_jac*(twopi*twopi)*vp(2:ns)/phip_real(2:ns) !!dV/d(PHI) on half mesh
96
97 ! COMPUTE INTEGRATED TOROIDAL CURRENT
98 DO i = 2,ns
99 torcur(i)=sign_jac*twopi*sum(bsubu(i:nrzt:ns)*wint(i:nrzt:ns))
100 END DO
101
102 ! COMPUTE SURFACE AVERAGE VARIABLES ON FULL RADIAL MESH
103 DO i = 2,ns1
104 phip_real(i) = p5*(phip_real(i+1) + phip_real(i))
105 denom = one/(hs*phip_real(i))
106 shear(i) = (iotas(i+1) - iotas(i))*denom !!d(iota)/d(PHI)
107 vpp(i) = (vp_real(i+1) - vp_real(i))*denom !!d(VP)/d(PHI)
108 presp(i) = (pres(i+1) - pres(i))*denom !!d(p)/d(PHI)
109 ip(i) = (torcur(i+1) - torcur(i))*denom !!d(Itor)/d(PHI)
110 END DO
111
112 ! COMPUTE GPP == |grad-phi|**2 = PHIP**2*|grad-s|**2 (on full mesh)
113 ! Gsqrt_FULL = JACOBIAN/PHIP == jacobian based on flux (on full mesh)
114 DO i = 2, ns1
115 gsqrt_full(i,:) = p5*(gsqrt(i,:) + gsqrt(i+1,:))
116 bdotj(i,:) = bdotj(i,:)/gsqrt_full(i,:)
117 gsqrt_full(i,:) = gsqrt_full(i,:)/phip_real(i)
118 sj(i) = hs*(i-1)
119 sqs = sqrt(sj(i))
120 rtf(:) = rt(i,:,0) + sqs*rt(i,:,1)
121 ztf(:) = zt(i,:,0) + sqs*zt(i,:,1)
122 gtt(:) = rtf(:)*rtf(:) + ztf(:)*ztf(:)
123 rzf(:) = rz(i,:,0) + sqs*rz(i,:,1)
124 zzf(:) = zz(i,:,0) + sqs*zz(i,:,1)
125 r1f(:) = r1(i,:,0) + sqs*r1(i,:,1)
126 gpp(i,:) = gsqrt_full(i,:)**2/(gtt(:)*r1f(:)**2 + &
127 (rtf(:)*zzf(:) - rzf(:)*ztf(:))**2) ! 1/gpp
128 END DO
129
130 ! COMPUTE SURFACE AVERAGES OVER dS/|grad-PHI|**3 => |Jac| du dv / |grad-PHI|**2
131 ! WHERE Jac = gsqrt/phip_real
132 DO i = 2,ns
133 b2(i,:) = two*(bsq(i,:) - pres(i))
134 END DO
135
136 DO i = 2,ns1
137 b2i(:) = p5*(b2(i+1,:) + b2(i,:))
138 ob2(:) = gsqrt_full(i,:)/b2i(:)
139 tpp(i) = sum(ob2(:)*wint(i:nrzt:ns)) ! <1/B**2>
140 ob2(:) = b2i(:) * gsqrt_full(i,:) * gpp(i,:)
141 tbb(i) = sum(ob2(:)*wint(i:nrzt:ns)) ! <b*b/|grad-phi|**3>
142 jdotb(:) = bdotj(i,:) * gpp(i,:) * gsqrt_full(i,:)
143 tjb(i) = sum(jdotb(:)*wint(i:nrzt:ns)) ! <j*b/|grad-phi|**3>
144 jdotb(:) = jdotb(:) * bdotj(i,:) / b2i(:)
145 tjj(i) = sum(jdotb(:)*wint(i:nrzt:ns)) ! <(j*b)2/b**2*|grad-phi|**3>
146 END DO
147
148 DEALLOCATE (gpp, gsqrt_full, b2, stat=i)
149
150 ! REFERENCE: BAUER, BETANCOURT, GARABEDIAN, MHD Equilibrium and Stability of Stellarators
151 ! We break up the Omega-subs into a positive shear term (Dshear) and a net current term, Dcurr
152 ! Omega_subw == Dwell and Omega-subd == Dgeod (geodesic curvature, Pfirsch-Schluter term)
153 !
154 ! Include (eventually) Suydam for reference (cylindrical limit)
155
156 WRITE(nmerc,90)
15790 FORMAT(6x,'S',10x,'PHI',9x,'IOTA',8x,'SHEAR',7x,' VP ',8x,'WELL', &
158 8x,'ITOR',7x,'ITOR''',7x,'PRES',7x,'PRES''',/,120('-'))
159
160 DO i = 2,ns1
161 sqs = p5*(vp_real(i) + vp_real(i+1))*sign_jac
162 IF (sqs .eq. zero) cycle
163 WRITE(nmerc,100) sj(i), hs*sum(phip_real(2:i)), &
164 p5*(iotas(i+1)+iotas(i)), shear(i)/sqs, &
165 sqs, -vpp(i)*sign_jac, &
166 p5*(torcur(i) + torcur(i+1)), ip(i)/sqs, &
167 p5*(pres(i) + pres(i+1)), presp(i)/sqs
168 END DO
169
170100 FORMAT(1p,10e12.4)
171
172 WRITE(nmerc,190)
173190 FORMAT(/,6x,'S',8x,'DMerc',8x,'DShear',7x,'DCurr',7x,'DWell',7x,'Dgeod',/,100('-'))
174
175 DO i = 2,ns1
176 tpp(i) = (twopi*twopi)*tpp(i)
177 tjb(i) = (twopi*twopi)*tjb(i)
178 tbb(i) = (twopi*twopi)*tbb(i)
179 tjj(i) = (twopi*twopi)*tjj(i)
180 dshear(i) = shear(i) * shear(i)/4
181 dcurr(i) =-shear(i) * (tjb(i) - ip(i) *tbb(i))
182 dwell(i) = presp(i) * (vpp(i) - presp(i) *tpp(i))*tbb(i)
183 dgeod(i) = tjb(i) *tjb(i) - tbb(i) *tjj(i)
184 dmerc(i) = dshear(i) + dcurr(i) + dwell(i) + dgeod(i)
185 WRITE(nmerc,100) sj(i), dmerc(i), dshear(i), dcurr(i), dwell(i), dgeod(i)
186 END DO
187
188 CLOSE (nmerc)

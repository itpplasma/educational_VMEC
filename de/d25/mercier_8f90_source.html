2 SUBROUTINE mercier(gsqrt, bsq, bdotj, iotas, wint, &
3  r1, rt, rz, zt, zz, bsubu, vp, phips, pres, ns, nznt)
4  USE safe_open_mod
5  USE vmercier
6  USE vmec_input, ONLY: input_extension
7  USE vparams, ONLY: one, zero, twopi, nmercier0
8  IMPLICIT NONE
9 
10  INTEGER, INTENT(in) :: ns, nznt
11  REAL(rprec), DIMENSION(ns,nznt), INTENT(in) :: gsqrt, bsq
12  REAL(rprec), DIMENSION(ns,nznt), INTENT(inout) :: bdotj
13  REAL(rprec), DIMENSION(ns*nznt), INTENT(in) :: wint, bsubu
14  REAL(rprec), DIMENSION(ns,nznt,0:1), INTENT(in) :: r1, rt, rz, zt, zz
15  REAL(rprec), DIMENSION(ns), INTENT(in) :: iotas, vp, phips, pres
16 
17  REAL(rprec), PARAMETER :: p5 = 0.5_dp, two = 2
18 
19  INTEGER :: ns1, i, imercier0, nmerc = nmercier0, nrzt
20  REAL(rprec) :: sign_jac, hs, sqs, denom
21  REAL(rprec), DIMENSION(:,:), ALLOCATABLE :: gpp, gsqrt_full, b2
22  REAL(rprec), DIMENSION(nznt) :: gtt, rtf, ztf, &
23  rzf, zzf, r1f, jdotb, ob2, b2i
24  REAL(rprec), DIMENSION(ns) :: vp_real, phip_real, &
25  shear, vpp, presp, torcur, ip, sj, tpp, tjb, tbb, tjj
26  CHARACTER(LEN=120) mercier_file
27 
28  nrzt = ns*nznt
29  mercier_file = 'mercier.'//trim(input_extension)
30  CALL safe_open (nmerc, imercier0, mercier_file, 'replace', 'formatted')
31  IF (imercier0 .ne. 0) RETURN
32 
33  ALLOCATE (gpp(ns,nznt), gsqrt_full(ns,nznt), b2(ns,nznt), stat=i)
34  IF (i .ne. 0) stop 'allocation error in Mercier'
35 
36  ! SCALE VP, PHIPS TO REAL UNITS (VOLUME, TOROIDAL FLUX DERIVATIVES)
37  ! AND PUT GSQRT IN ABS UNITS (SIGNGS MAY BE NEGATIVE)
38  ! NOTE: VP has (coming into this routine) the sign of the jacobian multiplied out
39  ! i.e., vp = signgs*<gsqrt>
40  ! THE SHEAR TERM MUST BE MULTIPLIED BY THE SIGN OF THE JACOBIAN
41  ! (OR A BETTER SOLUTION IS TO RETAIN THE JACOBIAN SIGN IN ALL TERMS, INCLUDING
42  ! VP, THAT DEPEND EXPLICITLY ON THE JACOBIAN. WE CHOOSE THIS LATTER METHOD...)
43  !
44  ! COMING INTO THIS ROUTINE, THE JACOBIAN(gsqrt) = 1./(grad-s . grad-theta X grad-zeta)
45  ! WE CONVERT THIS FROM grad-s to grad-phi DEPENDENCE BY DIVIDING gsqrt by PHIP_real
46  !
47  ! NOTE: WE ARE USING 0 < s < 1 AS THE FLUX VARIABLE, BEING CAREFUL
48  ! TO KEEP d(phi)/ds == PHIP_real FACTORS WHERE REQUIRED
49  ! THE V'' TERM IS d2V/d(PHI)**2, PHI IS REAL TOROIDAL FLUX
50  !
51  ! SHEAR = d(iota)/d(phi) : FULL MESH
52  ! VPP = d(vp)/d(phi) : FULL MESH
53  ! PRESP = d(pres)/d(phi) : FULL MESH (PRES IS REAL PRES*mu0)
54  ! IP = d(Itor)/d(phi) : FULL MESH
55  !
56  ! ON ENTRY, BDOTJ = Jacobian * J*B ON THE FULL RADIAL GRID
57  ! BSQ = 0.5*|B**2| + p IS ON THE HALF RADIAL GRID
58 
59  ns1 = ns - 1
60  IF (ns1 .le. 0) RETURN
61  hs = one/ns1
62  sign_jac = zero
63  IF (gsqrt(ns,1) .ne. zero) then
64  sign_jac = abs(gsqrt(ns,1))/gsqrt(ns,1)
65  end if
66 
67  IF (sign_jac .eq. zero) RETURN
68  phip_real = twopi * phips * sign_jac
69 
70  ! NOTE: phip_real should be > 0 to get the correct physical sign of REAL-space gradients
71  ! For example, grad-p, grad-Ip, etc. However, with phip_real defined this way,
72  ! Mercier will be correct
73  vp_real(2:ns) = sign_jac*(twopi*twopi)*vp(2:ns)/phip_real(2:ns) !!dV/d(PHI) on half mesh
74 
75  ! COMPUTE INTEGRATED TOROIDAL CURRENT
76  DO i = 2,ns
77  torcur(i)=sign_jac*twopi*sum(bsubu(i:nrzt:ns)*wint(i:nrzt:ns))
78  END DO
79 
80  ! COMPUTE SURFACE AVERAGE VARIABLES ON FULL RADIAL MESH
81  DO i = 2,ns1
82  phip_real(i) = p5*(phip_real(i+1) + phip_real(i))
83  denom = one/(hs*phip_real(i))
84  shear(i) = (iotas(i+1) - iotas(i))*denom !!d(iota)/d(PHI)
85  vpp(i) = (vp_real(i+1) - vp_real(i))*denom !!d(VP)/d(PHI)
86  presp(i) = (pres(i+1) - pres(i))*denom !!d(p)/d(PHI)
87  ip(i) = (torcur(i+1) - torcur(i))*denom !!d(Itor)/d(PHI)
88  END DO
89 
90  ! COMPUTE GPP == |grad-phi|**2 = PHIP**2*|grad-s|**2 (on full mesh)
91  ! Gsqrt_FULL = JACOBIAN/PHIP == jacobian based on flux (on full mesh)
92  DO i = 2, ns1
93  gsqrt_full(i,:) = p5*(gsqrt(i,:) + gsqrt(i+1,:))
94  bdotj(i,:) = bdotj(i,:)/gsqrt_full(i,:)
95  gsqrt_full(i,:) = gsqrt_full(i,:)/phip_real(i)
96  sj(i) = hs*(i-1)
97  sqs = sqrt(sj(i))
98  rtf(:) = rt(i,:,0) + sqs*rt(i,:,1)
99  ztf(:) = zt(i,:,0) + sqs*zt(i,:,1)
100  gtt(:) = rtf(:)*rtf(:) + ztf(:)*ztf(:)
101  rzf(:) = rz(i,:,0) + sqs*rz(i,:,1)
102  zzf(:) = zz(i,:,0) + sqs*zz(i,:,1)
103  r1f(:) = r1(i,:,0) + sqs*r1(i,:,1)
104  gpp(i,:) = gsqrt_full(i,:)**2/(gtt(:)*r1f(:)**2 + &
105  (rtf(:)*zzf(:) - rzf(:)*ztf(:))**2) ! 1/gpp
106  END DO
107 
108  ! COMPUTE SURFACE AVERAGES OVER dS/|grad-PHI|**3 => |Jac| du dv / |grad-PHI|**2
109  ! WHERE Jac = gsqrt/phip_real
110  DO i = 2,ns
111  b2(i,:) = two*(bsq(i,:) - pres(i))
112  END DO
113 
114  DO i = 2,ns1
115  b2i(:) = p5*(b2(i+1,:) + b2(i,:))
116  ob2(:) = gsqrt_full(i,:)/b2i(:)
117  tpp(i) = sum(ob2(:)*wint(i:nrzt:ns)) ! <1/B**2>
118  ob2(:) = b2i(:) * gsqrt_full(i,:) * gpp(i,:)
119  tbb(i) = sum(ob2(:)*wint(i:nrzt:ns)) ! <b*b/|grad-phi|**3>
120  jdotb(:) = bdotj(i,:) * gpp(i,:) * gsqrt_full(i,:)
121  tjb(i) = sum(jdotb(:)*wint(i:nrzt:ns)) ! <j*b/|grad-phi|**3>
122  jdotb(:) = jdotb(:) * bdotj(i,:) / b2i(:)
123  tjj(i) = sum(jdotb(:)*wint(i:nrzt:ns)) ! <(j*b)2/b**2*|grad-phi|**3>
124  END DO
125 
126  DEALLOCATE (gpp, gsqrt_full, b2, stat=i)
127 
128  ! REFERENCE: BAUER, BETANCOURT, GARABEDIAN, MHD Equilibrium and Stability of Stellarators
129  ! We break up the Omega-subs into a positive shear term (Dshear) and a net current term, Dcurr
130  ! Omega_subw == Dwell and Omega-subd == Dgeod (geodesic curvature, Pfirsch-Schluter term)
131  !
132  ! Include (eventually) Suydam for reference (cylindrical limit)
133 
134  WRITE(nmerc,90)
135 90 FORMAT(6x,'S',10x,'PHI',9x,'IOTA',8x,'SHEAR',7x,' VP ',8x,'WELL', &
136  8x,'ITOR',7x,'ITOR''',7x,'PRES',7x,'PRES''',/,120('-'))
137 
138  DO i = 2,ns1
139  sqs = p5*(vp_real(i) + vp_real(i+1))*sign_jac
140  IF (sqs .eq. zero) cycle
141  WRITE(nmerc,100) sj(i), hs*sum(phip_real(2:i)), &
142  p5*(iotas(i+1)+iotas(i)), shear(i)/sqs, &
143  sqs, -vpp(i)*sign_jac, &
144  p5*(torcur(i) + torcur(i+1)), ip(i)/sqs, &
145  p5*(pres(i) + pres(i+1)), presp(i)/sqs
146  END DO
147 
148 100 FORMAT(1p,10e12.4)
149 
150  WRITE(nmerc,190)
151 190 FORMAT(/,6x,'S',8x,'DMerc',8x,'DShear',7x,'DCurr',7x,'DWell',7x,'Dgeod',/,100('-'))
152 
153  DO i = 2,ns1
154  tpp(i) = (twopi*twopi)*tpp(i)
155  tjb(i) = (twopi*twopi)*tjb(i)
156  tbb(i) = (twopi*twopi)*tbb(i)
157  tjj(i) = (twopi*twopi)*tjj(i)
158  dshear(i) = shear(i) * shear(i)/4
159  dcurr(i) =-shear(i) * (tjb(i) - ip(i) *tbb(i))
160  dwell(i) = presp(i) * (vpp(i) - presp(i) *tpp(i))*tbb(i)
161  dgeod(i) = tjb(i) *tjb(i) - tbb(i) *tjj(i)
162  dmerc(i) = dshear(i) + dcurr(i) + dwell(i) + dgeod(i)
163  WRITE(nmerc,100) sj(i), dmerc(i), dshear(i), dcurr(i), dwell(i), dgeod(i)
164  END DO
165 
166  CLOSE (nmerc)
167 
168 END SUBROUTINE mercier

75  ax(js,1) = ax(js,1) + ptau(lk)*t1*t1
76  ax(js,2) = ax(js,2) + ptau(lk)*(-t1+t3)*(t1+t2)
77  ax(js,3) = ax(js,3) + ptau(lk)*(t1+t2)*(t1+t2)
78  ax(js,4) = ax(js,4) + ptau(lk)*(-t1+t3)*(-t1+t3)
79  END DO

91  END DO

98  axm(js,2) = ax(js,2) * sm(js) * sp(js-1)

104  cx(js) = cx(js) + cx(js+1)
105  temp(js) = signgs*(temp(js) + temp(js+1))
106  END DO
107 
108  eqfactor(2:ns-1) = axd(2:ns-1,2)*hs*hs/temp(2:ns-1)
109  eqfactor(1) = 0; eqfactor(ns) = 0
110  axm(ns+1,:) = 0; axd(ns+1,:) = 0
111  bxm(ns+1,:) = 0; bxd(ns+1,:) = 0
112

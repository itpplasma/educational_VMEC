22 INTEGER, PARAMETER :: limpts = 61
23 REAL(rprec), PARAMETER :: p5 = 0.5_dp, two = 2

25 INTEGER :: iv, iu, iu_r, ivminus, nlim, ns12, klim, n
26! REAL(rprec), DIMENSION(nzeta) :: rcom, zcom
27! REAL(rprec), DIMENSION(nzeta,ntheta1) :: r1b, z1b, rub, zub
28! REAL(rprec), DIMENSION(nzeta,ntheta1) :: r12, z12

40 integer :: ivmax

42 if (.not.lasym) then
43 ivmax = nzeta/2+1
44 else
45 ivmax = nzeta
46 end if
47
48 allocate(rcom(nzeta))
49 allocate(zcom(nzeta))
50 allocate(r1b(nzeta,ntheta1)); r1b = cbig
51 allocate(z1b(nzeta,ntheta1)); z1b = cbig
52 allocate(rub(nzeta,ntheta1)); rub = cbig
53 allocate(zub(nzeta,ntheta1)); zub = cbig
54 allocate(r12(nzeta,ntheta1)); r12 = cbig
55 allocate(z12(nzeta,ntheta1)); z12 = cbig
56 allocate(rs(nzeta,ntheta1)); rs = cbig
57 allocate(zs(nzeta,ntheta1)); zs = cbig
58 allocate(tau(nzeta,ntheta1)); tau = cbig
59 allocate(ru12(nzeta,ntheta1)); ru12 = cbig
60 allocate(zu12(nzeta,ntheta1)); zu12 = cbig
61 allocate(tau0(nzeta,ntheta1)); tau0 = cbig

63 ! COMPUTES GUESS FOR MAGNETIC AXIS IF USER GUESS
64 ! LEADS TO INITIAL SIGN CHANGE OF JACOBIAN. DOES A GRID
65 ! SEARCH (irgrid, izgrid) IN EACH PHI-PLANE FOR POINTS WHICH
66 ! YIELD A VALUE FOR THE JACOBIAN WITH THE CORRECT SIGN (SIGNGS)
67 ! CHOOSES THE AXIS POSITION SO THE MIN VALUE OF THE JACOBIAN IS MAXIMIZED
68
69 ns12 = (ns+1)/2
70
71 planes: DO iv = 1, nzeta
72
73 IF (.not.lasym .and. iv.gt.nzeta/2+1) THEN
74 rcom(iv) = rcom(nzeta+2-iv)
75 zcom(iv) =-zcom(nzeta+2-iv)
76
77 ! For a stellarator-symmetric case, the rest of the loop is skipped for iv>nzeta/2+1.
78 cycle
79 END IF
80
81 r1b(iv,:ntheta3) = r1(ns,iv,:,0) + r1(ns,iv,:,1)
82 z1b(iv,:ntheta3) = z1(ns,iv,:,0) + z1(ns,iv,:,1)
83 r12(iv,:ntheta3) = r1(ns12,iv,:,0) + r1(ns12,iv,:,1)*sqrts(ns12)
84 z12(iv,:ntheta3) = z1(ns12,iv,:,0) + z1(ns12,iv,:,1)*sqrts(ns12)
85 rub(iv,:ntheta3) = ru0(ns,iv,:)
86 zub(iv,:ntheta3) = zu0(ns,iv,:)
87 ru12(iv,:ntheta3) = p5*(ru0(ns,iv,:) + ru0(ns12,iv,:))
88 zu12(iv,:ntheta3) = p5*(zu0(ns,iv,:) + zu0(ns12,iv,:))
89
90 IF (.not.lasym) THEN
91
92 ! USE Z(v,-u) = -Z(twopi-v,u), R(v,-u) = R(twopi-v,u)
93 ! TO DO EXTEND R,Z, etc. OVER ALL THETA (NOT JUST 0,PI)
94
95 ! (twopi-v)
96 ivminus = mod(nzeta + 1 - iv, nzeta) + 1
97 DO iu = 1+ntheta2, ntheta1
98 iu_r = ntheta1 + 2 - iu ! (ntheta1 + 1) - (iu - 1)
99
100 r1b(iv,iu) = r1(ns, ivminus,iu_r,0) + r1( ns, ivminus,iu_r,1)
101 z1b(iv,iu) =-( z1(ns, ivminus,iu_r,0) + z1( ns, ivminus,iu_r,1))
102 r12(iv,iu) = r1(ns12,ivminus,iu_r,0) + r1( ns12,ivminus,iu_r,1)*sqrts(ns12)
103 z12(iv,iu) =-( z1(ns12,ivminus,iu_r,0) + z1( ns12,ivminus,iu_r,1)*sqrts(ns12))
104 rub(iv,iu) =- ru0(ns, ivminus,iu_r)
105 zub(iv,iu) = zu0(ns, ivminus,iu_r)
106 ru12(iv,iu)=-(ru0(ns, ivminus,iu_r) + ru0(ns12,ivminus,iu_r))*p5
107 zu12(iv,iu)= (zu0(ns, ivminus,iu_r) + zu0(ns12,ivminus,iu_r))*p5
108 END DO
109 END IF
110
111 ! Scan over r-z grid for interior point
112 rmin = minval(r1b(iv,:))
113 rmax = maxval(r1b(iv,:))
114 zmin = minval(z1b(iv,:))
115 zmax = maxval(z1b(iv,:))
116 rcom(iv) = (rmax + rmin)/2
117 zcom(iv) = (zmax + zmin)/2
118
119 ! Estimate jacobian based on boundary and 1/2 surface
120 ds = (ns - ns12)*hs
121 DO iu = 1, ntheta1
122 rs(iv,iu) = (r1b(iv,iu) - r12(iv,iu))/ds + r1(1,iv,1,0)
123 zs(iv,iu) = (z1b(iv,iu) - z12(iv,iu))/ds + z1(1,iv,1,0)
124 tau0(iv,iu) = ru12(iv,iu)*zs(iv,iu) - zu12(iv,iu)*rs(iv,iu)
125 END DO
126
127 mintau = 0
128
129 DO nlim = 1, limpts
130 zlim = zmin + ((zmax - zmin)*(nlim-1))/(limpts-1)
131 IF (.not.lasym .and. (iv.eq.1 .or. iv.eq.nzeta/2+1)) THEN
132 zlim = 0
133 IF (nlim .gt. 1) then
134 EXIT
135 end if
136 END IF
137
138 ! Find value of magnetic axis that maximizes the minimum jacobian value
139 DO klim = 1, limpts
140 rlim = rmin + ((rmax - rmin)*(klim-1))/(limpts-1)
141 tau(iv,:) = signgs*(tau0(iv,:) - ru12(iv,:)*zlim + zu12(iv,:)*rlim)
142
143 mintemp = minval(tau(iv,:))
144 IF (mintemp .gt. mintau) THEN
145 mintau = mintemp
146 rcom(iv) = rlim
147 zcom(iv) = zlim
148 ELSE IF (mintemp .eq. mintau) THEN
149 ! If up-down symmetric and lasym=T, need this to pick z = 0
150 IF (abs(zcom(iv)).gt.abs(zlim)) then
151 zcom(iv) = zlim
152 end if
153 END IF
154 END DO
155 END DO
156 END DO planes
157
158 ! FOURIER TRANSFORM RCOM, ZCOM
159 dzeta = two/nzeta
160 DO n = 0, ntor
161 raxis_cc(n) = dzeta*sum(cosnv(:,n)*rcom(:))/nscale(n)
162 zaxis_cs(n) =-dzeta*sum(sinnv(:,n)*zcom(:))/nscale(n)
163 raxis_cs(n) =-dzeta*sum(sinnv(:,n)*rcom(:))/nscale(n)
164 zaxis_cc(n) = dzeta*sum(cosnv(:,n)*zcom(:))/nscale(n)
165 IF (n.eq.0 .or. n.eq.nzeta/2) THEN
166 raxis_cc(n) = p5*raxis_cc(n)
167 zaxis_cc(n) = p5*zaxis_cc(n)
168 END IF
169 END DO
170
171 ! debugging output from guess_axis
172 if (open_dbg_context("guess_axis")) then
173
174 call add_real_2d("r1b" , ivmax, ntheta1, r1b(1:ivmax,:) )
175 call add_real_2d("z1b" , ivmax, ntheta1, z1b(1:ivmax,:) )
176 call add_real_2d("rub" , ivmax, ntheta1, rub(1:ivmax,:) )
177 call add_real_2d("zub" , ivmax, ntheta1, zub(1:ivmax,:) )
178 call add_real_2d("r12" , ivmax, ntheta1, r12(1:ivmax,:) )
179 call add_real_2d("z12" , ivmax, ntheta1, z12(1:ivmax,:) )
180 call add_real_2d("rs" , ivmax, ntheta1, rs(1:ivmax,:) )
181 call add_real_2d("zs" , ivmax, ntheta1, zs(1:ivmax,:) )
182 call add_real_2d("tau" , ivmax, ntheta1, tau(1:ivmax,:) )
183 call add_real_2d("ru12", ivmax, ntheta1, ru12(1:ivmax,:))
184 call add_real_2d("zu12", ivmax, ntheta1, zu12(1:ivmax,:))
185 call add_real_2d("tau0", ivmax, ntheta1, tau0(1:ivmax,:))
186
187 call add_real_1d("rcom", nzeta, rcom)
188 call add_real_1d("zcom", nzeta, zcom)
189
190 call add_real_1d("raxis_cc", ntor+1, raxis_cc)
191 call add_real_1d("zaxis_cs", ntor+1, zaxis_cs)
192 call add_real_1d("raxis_cs", ntor+1, raxis_cs)
193 call add_real_1d("zaxis_cc", ntor+1, zaxis_cc)
194
195 call close_dbg_out()
196 end if
197
198 deallocate(rcom, zcom)
199 deallocate(r1b, z1b, rub, zub, &
200 r12, z12, &
201 rs, zs, tau, ru12, zu12, tau0)

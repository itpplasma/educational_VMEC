22 INTEGER :: iv, iu, iu_r, ivminus, nlim, ns12, klim, n
23 REAL(rprec), DIMENSION(nzeta) :: rcom, zcom
24 REAL(rprec), DIMENSION(ntheta1) :: r1b, z1b, rub, zub
25 REAL(rprec), DIMENSION(ntheta1) :: r12, z12
26 REAL(rprec), DIMENSION(ntheta1) :: rs, zs, tau, ru12, zu12, tau0

31 character(len=255) :: dump_filename

34 ! COMPUTES GUESS FOR MAGNETIC AXIS IF USER GUESS
35 ! LEADS TO INITIAL SIGN CHANGE OF JACOBIAN. DOES A GRID
36 ! SEARCH (irgrid, izgrid) IN EACH PHI-PLANE FOR POINTS WHICH
37 ! YIELD A VALUE FOR THE JACOBIAN WITH THE CORRECT SIGN (SIGNGS)
38 ! CHOOSES THE AXIS POSITION SO THE MIN VALUE OF THE JACOBIAN IS MAXIMIZED
39
40 ns12 = (ns+1)/2
41
42 ! debugging output from guess_axis
43 if (dump_guess_axis) then
44 write(dump_filename, 998) trim(input_extension)
45 open(unit=42, file=trim(dump_filename), status="unknown")
46
47 write(42, *) "# nzeta ntheta1, ns12"
48 write(42, *) nzeta, ntheta1, ns12
49
50 write(42, *) "# iv iu r1b z1b rub zub r12 z12 rs zs tau ru12 zu12 tau0"
51 end if
52
53 planes: DO iv = 1, nzeta
54
55 IF (.not.lasym .and. iv.gt.nzeta/2+1) THEN
56 rcom(iv) = rcom(nzeta+2-iv)
57 zcom(iv) =-zcom(nzeta+2-iv)
58 cycle
59 END IF
60
61 r1b(:ntheta3) = r1(ns,iv,:,0) + r1(ns,iv,:,1)
62 z1b(:ntheta3) = z1(ns,iv,:,0) + z1(ns,iv,:,1)
63 r12(:ntheta3) = r1(ns12,iv,:,0) + r1(ns12,iv,:,1)*sqrts(ns12)
64 z12(:ntheta3) = z1(ns12,iv,:,0) + z1(ns12,iv,:,1)*sqrts(ns12)
65 rub(:ntheta3) = ru0(ns,iv,:)
66 zub(:ntheta3) = zu0(ns,iv,:)
67 ru12(:ntheta3) = p5*(ru0(ns,iv,:) + ru0(ns12,iv,:))
68 zu12(:ntheta3) = p5*(zu0(ns,iv,:) + zu0(ns12,iv,:))
69
70 IF (.not.lasym) THEN
71
72 ! USE Z(v,-u) = -Z(twopi-v,u), R(v,-u) = R(twopi-v,u)
73 ! TO DO EXTEND R,Z, etc. OVER ALL THETA (NOT JUST 0,PI)
74
75 ! (twopi-v)
76 ivminus = mod(nzeta + 1 - iv, nzeta) + 1
77 DO iu = 1+ntheta2, ntheta1
78 iu_r = ntheta1 + 2 - iu ! (ntheta1 + 1) - (iu - 1)
79 r1b(iu) = r1(ns,ivminus,iu_r,0) + r1(ns,ivminus,iu_r,1)
80 z1b(iu) =-(z1(ns,ivminus,iu_r,0) + z1(ns,ivminus,iu_r,1))
81 r12(iu) = r1(ns12,ivminus,iu_r,0) + r1(ns12,ivminus,iu_r,1)*sqrts(ns12)
82 z12(iu) =-(z1(ns12,ivminus,iu_r,0) + z1(ns12,ivminus,iu_r,1)*sqrts(ns12))
83 rub(iu) =-ru0(ns,ivminus,iu_r)
84 zub(iu) = zu0(ns,ivminus,iu_r)
85 ru12(iu)=-p5*(ru0(ns,ivminus,iu_r) + ru0(ns12,ivminus,iu_r))
86 zu12(iu)= p5*(zu0(ns,ivminus,iu_r) + zu0(ns12,ivminus,iu_r))
87 END DO
88
89 END IF
90
91 ! Scan over r-z grid for interior point
92 rmin = minval(r1b)
93 rmax = maxval(r1b)
94 zmin = minval(z1b)
95 zmax = maxval(z1b)
96 rcom(iv) = (rmax + rmin)/2
97 zcom(iv) = (zmax + zmin)/2
98
99 ! Estimate jacobian based on boundary and 1/2 surface
100 ds = (ns - ns12)*hs
101 DO iu = 1, ntheta1
102 rs(iu) = (r1b(iu) - r12(iu))/ds + r1(1,iv,1,0)
103 zs(iu) = (z1b(iu) - z12(iu))/ds + z1(1,iv,1,0)
104 tau0(iu) = ru12(iu)*zs(iu) - zu12(iu)*rs(iu)
105 END DO
106
107 mintau = 0
108
109 DO nlim = 1, limpts
110 zlim = zmin + ((zmax - zmin)*(nlim-1))/(limpts-1)
111 IF (.not.lasym .and. (iv.eq.1 .or. iv.eq.nzeta/2+1)) THEN
112 zlim = 0
113 IF (nlim .gt. 1) then
114 EXIT
115 end if
116 END IF
117
118 ! Find value of magnetic axis that maximizes the minimum jacobian value
119 DO klim = 1, limpts
120 rlim = rmin + ((rmax - rmin)*(klim-1))/(limpts-1)
121 tau = signgs*(tau0 - ru12(:)*zlim + zu12(:)*rlim)
122
123 mintemp = minval(tau)
124 IF (mintemp .gt. mintau) THEN
125 mintau = mintemp
126 rcom(iv) = rlim
127 zcom(iv) = zlim
128 ELSE IF (mintemp .eq. mintau) THEN
129 ! If up-down symmetric and lasym=T, need this to pick z = 0
130 IF (abs(zcom(iv)).gt.abs(zlim)) then
131 zcom(iv) = zlim
132 end if
133 END IF
134 END DO
135 END DO
136
137 if (dump_guess_axis) then
138 do iu = 1, ntheta1
139 write(42, *) iv, iu, &
140 r1b(iu), z1b(iu), rub(iu), zub(iu), &
141 r12(iu), z12(iu), &
142 rs(iu), zs(iu), tau(iu), ru12(iu), zu12(iu), tau0(iu)
143 end do
144 end if
145
146 END DO planes
147
148 ! FOURIER TRANSFORM RCOM, ZCOM
149 dzeta = two/nzeta
150 DO n = 0, ntor
151 raxis_cc(n) = dzeta*sum(cosnv(:,n)*rcom(:))/nscale(n)
152 zaxis_cs(n) =-dzeta*sum(sinnv(:,n)*zcom(:))/nscale(n)
153 raxis_cs(n) =-dzeta*sum(sinnv(:,n)*rcom(:))/nscale(n)
154 zaxis_cc(n) = dzeta*sum(cosnv(:,n)*zcom(:))/nscale(n)
155 IF (n.eq.0 .or. n.eq.nzeta/2) THEN
156 raxis_cc(n) = p5*raxis_cc(n)
157 zaxis_cc(n) = p5*zaxis_cc(n)
158 END IF
159 END DO
160
161 if (dump_guess_axis) then
162 write(42, *) "# iv rcom zcom"
163 do iv=1, nzeta
164 write(42, *) iv, rcom(iv), zcom(iv)
165 end do
166
167 write(42, *) "# n raxis_cc zaxis_cs raxis_cs zaxis_cc"
168 DO n = 0, ntor
169 write(42, *) n, raxis_cc(n), zaxis_cs(n), raxis_cs(n), zaxis_cc(n)
170 end do
171
172 close(42)
173
174 print *, "dumped guess_axis output to '"//trim(dump_filename)//"'"
175 stop
176 end if

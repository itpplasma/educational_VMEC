105 SUBROUTINE read_mgrid (mgrid_file, extcur, nv, nfp, lscreen,

108C-----------------------------------------------
109C D u m m y V a r i a b l e s
110C-----------------------------------------------
111!
112! mgrid_file: full path to mgrid file
113! lscreen : logical controlling output to screen
114! ier_flag ; error flag returned to caller
115! extcur(n) : external current multiplier for bfield(n) components
116!
117 INTEGER, INTENT(out) :: ier_flag
118 INTEGER, INTENT(in) :: nv, nfp
119 LOGICAL, INTENT(in) :: lscreen
120 REAL(rprec), INTENT(in) :: extcur(:)
121 CHARACTER(len=*), INTENT(in) :: mgrid_file

123C L o c a l P a r a m e t e r s
124C-----------------------------------------------
125 CHARACTER(LEN=*), PARAMETER :: mgrid_defarea='$HOME/vmec/MAKEGRID'
126C-----------------------------------------------
127C L o c a l V a r i a b l e s

131!

134 LOGICAL :: lgrid_exist, lfind
135C-----------------------------------------------
136
137 mgrid_path = trim(mgrid_file)
138
139 IF ((mgrid_path .eq. trim(mgrid_path_old)) .and.
140 1 ALLOCATED(curlabel)) THEN
141 print *,' mgrid file previously parsed!'
142 RETURN
143 END IF
144
145 INQUIRE (file=mgrid_path,exist=lgrid_exist,iostat=istat)
146 IF (istat.ne.0 .or. .not.lgrid_exist) THEN
147 IF (lscreen) print *,' MGRID FILE NOT FOUND IN SPECIFIED ',
148 1 'PATH: SEARCHING DEFAULT AREA'
149 ii = index(mgrid_file,'/',back=.true.)
150 istat = index(mgrid_defarea, '$HOME')
151 IF (istat .ne. 0) THEN
152 CALL getenv('HOME', home_dir)
153 IF (istat .gt. 1) THEN
154 home_dir = mgrid_defarea(1:istat-1) // trim(home_dir)
155 1 // mgrid_defarea(istat+5:)
156 ELSE
157 home_dir = trim(home_dir) // mgrid_defarea(istat+5:)
158 END IF
159 ELSE
160 home_dir = mgrid_defarea
161 END IF
162 mgrid_path = trim(home_dir) // mgrid_file(ii+1:)
163 INQUIRE (file=mgrid_path,exist=lgrid_exist,iostat=istat)
164 END IF
165
166 mgrid_path_old = mgrid_path
167
168 ier_flag = 0
169
170 IF (lgrid_exist) THEN
171 IF (lscreen) print '(2x,2a)',
172 1 'Opening vacuum field file: ', trim(mgrid_file)
173!
174! Parse mgrid file name, look for .nc extension (netcdf format)
175!
176 ii = len_trim(mgrid_path) - 2
177 lfind = (mgrid_path(ii:ii+2) == '.nc')
178 IF (lfind) THEN
179 CALL read_mgrid_nc (mgrid_path, extcur, nv, nfp,
180 1 ier_flag)
181 END IF
182
183 IF (np0b .ne. nv) THEN
184 print *,' NZETA=',nv,
185 1 ' NOT EQUAL TO NP0B=',np0b,' IN MGRID FILE'
186 ier_flag = 9
187 ELSE IF (nfper0.ne.nfp) THEN
188 print *,' NFP(READ in) = ',nfp,' DOES NOT AGREE WITH ',
189 1 'NFPER (in vacuum field file) = ',nfper0
190 ier_flag = 9
191 END IF
192
193 END IF
194
195 IF (ier_flag .ne. 0) RETURN
196
197 IF (.not.lgrid_exist .or. ier_flag.ne.0) THEN
198 lfreeb = .false.
199 IF (lscreen) THEN
200 print *, ' Error opening/reading mgrid file in dir: ',
201 1 trim(home_dir)
202 print *, ' User must supply vacuum bfield in mgrid to ',
203 1 'run vmec in free-boundary mode!'
204 print *, ' Proceeding to run vmec in',
205 1 ' fixed boundary mode'
206 END IF
207 END IF

211 SUBROUTINE read_mgrid_nc (filename, extcur, nv, nfp,
212 1 ier_flag)
213 USE ezcdf
214 USE vsvd0, only: nigroup
215 IMPLICIT NONE
216C-----------------------------------------------
217C D u m m y A r g u m e n t s
218C-----------------------------------------------
219 CHARACTER(LEN=*), INTENT(in) :: filename
220 INTEGER, INTENT(in) :: nv, nfp
221 REAL(rprec), INTENT(in) :: extcur(:)
222C-----------------------------------------------
223C L o c a l V a r i a b l e s
224C-----------------------------------------------
225 REAL(rprec), DIMENSION(:,:,:), ALLOCATABLE ::
226 1 brtemp, bztemp, bptemp
227 INTEGER :: ier_flag, ngrid
228 INTEGER :: istat, ig
229 INTEGER, DIMENSION(3) :: dimlens
230 CHARACTER(LEN=100) :: temp
231
232C-----------------------------------------------
233 call cdf_open(ngrid, filename,'r', istat)
234 IF (istat .ne. 0) THEN
235 ier_flag = 9
236 RETURN
237 END IF
238
239!
240! READ IN DATA
241!
242 CALL cdf_read(ngrid, vn_ir, nr0b)
243 CALL cdf_read(ngrid, vn_jz, nz0b)
244 CALL cdf_read(ngrid, vn_kp, np0b)
245 CALL cdf_read(ngrid, vn_nfp, nfper0)
246
247 IF (nfper0.ne.nfp .or. np0b.ne.nv) RETURN
248
249 CALL cdf_read(ngrid, vn_nextcur, nextcur)
250
251 IF (nextcur .eq. 0) THEN
252 print *,' NEXTCUR = 0 IN READING MGRID FILE'
253 ier_flag = 9
254 RETURN
255 ELSE IF (nextcur .gt. nigroup) THEN
256 print *,' NEXTCUR > NIGROUP IN MGRID FILE'
257 ier_flag = 9
258 RETURN
259 END IF
260
261 CALL cdf_read(ngrid, vn_rmin, rminb)
262 CALL cdf_read(ngrid, vn_zmin, zminb)
263 CALL cdf_read(ngrid, vn_rmax, rmaxb)
264 CALL cdf_read(ngrid, vn_zmax, zmaxb)
265
266 delrb = (rmaxb-rminb)/(nr0b-1)
267 delzb = (zmaxb-zminb)/(nz0b-1)
268
269 CALL cdf_inquire(ngrid, vn_coilgrp, dimlens)
270 IF (.NOT. ALLOCATED(curlabel)) THEN
271 ALLOCATE (curlabel(nextcur), stat=istat)
272 ELSE IF (SIZE(curlabel) .ne. nextcur) THEN
273 DEALLOCATE (curlabel)
274 ALLOCATE (curlabel(nextcur), stat=istat)
275 END IF
276!THIS IS A GLITCH WITH cdf_read: must distinguish 1D char array from multi-D
277 IF (nextcur .eq. 1) THEN
278 IF (istat .eq. 0)
279 1 CALL cdf_read(ngrid, vn_coilgrp, curlabel(1))
280 ELSE IF (istat .eq. 0) THEN
281 CALL cdf_read(ngrid, vn_coilgrp, curlabel(1:nextcur))
282 END IF
283
284 IF (istat .ne. 0) stop 'Error allocating CURLABEL in mgrid_mod'
285!
286! READ 3D Br, Bp, Bz ARRAYS FOR EACH COIL GROUP
287!
288 nbvac = nr0b*nz0b*np0b
289 IF (.NOT. ALLOCATED(bvac)) THEN
290 ALLOCATE (bvac(nbvac,3), stat=istat)
291 ELSE IF (SIZE(bvac,1) .ne. nbvac) THEN
292 DEALLOCATE (bvac); ALLOCATE(bvac(nbvac,3), stat=istat)
293 END IF
294 IF (istat .ne. 0) stop 'Error allocating bvac in mgrid_mod'
295
296 bvac = 0
297
298 ALLOCATE (brtemp(nr0b, nz0b, np0b), &
299 & bptemp(nr0b, nz0b, np0b), &
300 & bztemp(nr0b, nz0b, np0b), stat=istat)
301 IF (istat .ne. 0)stop 'Error allocating bXtemp in mgrid_mod'
302 brtemp = 0
303 bptemp = 0
304 bztemp = 0
305
306 DO ig = 1, nextcur
307 WRITE (temp, 1000) vn_br0, ig
308 CALL cdf_read(ngrid, temp, brtemp)
309
310 WRITE (temp, 1000) vn_bp0, ig
311 CALL cdf_read(ngrid, temp, bptemp)
312
313 WRITE (temp, 1000) vn_bz0, ig
314 CALL cdf_read(ngrid, temp, bztemp)
315
316!
317! STORE SUMMED BFIELD (OVER COIL GROUPS) IN BVAC
318!
319 CALL sum_bfield(bvac(:,1), brtemp, extcur(ig), nbvac)
320 CALL sum_bfield(bvac(:,2), bptemp, extcur(ig), nbvac)
321 CALL sum_bfield(bvac(:,3), bztemp, extcur(ig), nbvac)
322 END DO
323
324!
325! MUST ADD EXTERNAL LOOP STUFF LATER
326! MAY DECIDE TO WRITE THAT INFO INTO A SEPARATE FILE
327! FOR NOW, JUST SET DEFAULTS
328!
329 nobser = 0
330 nobd = 0
331 nbsets = 0
332
333 CALL cdf_inquire(ngrid, vn_mgmode, dimlens, ier=istat)
334 IF (istat .eq. 0) THEN
335 CALL cdf_read(ngrid, vn_mgmode, mgrid_mode)
336 ELSE
337 mgrid_mode = 'N'
338 END IF
339
340 CALL cdf_inquire(ngrid, vn_coilcur, dimlens, ier=istat)
341 IF (istat .eq. 0) THEN
342 IF (ALLOCATED(raw_coil_current)) DEALLOCATE(raw_coil_current)
343 ALLOCATE (raw_coil_current(nextcur), stat=istat)
344 IF (istat .ne. 0) stop 'Error allocating RAW_COIL in mgrid_mod'
345 CALL cdf_read(ngrid, vn_coilcur, raw_coil_current)
346 END IF
347
348 CALL cdf_close(ngrid)
349
350 IF (ALLOCATED(brtemp))
351 1 DEALLOCATE (brtemp, bptemp, bztemp)
352
353 CALL assign_bptrs(bvac)
354
3551000 FORMAT(a,'_',i3.3)
356
357 END SUBROUTINE read_mgrid_nc
